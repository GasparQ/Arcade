27 void *dlopen(const char *name, int flags)
28 {
29  shl_t handle;
30  char *library;
31 
32  if (name == NULL || *name == '\0')
33  return NULL;
34 
35  /* By convention, libmikmod will look for "foo.so" while on HP-UX the
36  name would be "foo.sl". Change the last letter here. */
37  library = strdup(name);
38  library[strlen(library) - 1] = 'l';
39 
40  handle = shl_load(library,
41  (flags & RTLD_LAZY ? BIND_DEFERRED : BIND_IMMEDIATE) | DYNAMIC_PATH,
42  0L);
43 
44  free(library);
45

47 }
