70 /* thread handling buffer-updates */
71 /* ARGSUSED */
72 void OS2_UpdateBufferThread(void *dummy)
73 {
74  /* Run at timecritical priority */
75  DosSetPriority(PRTYS_THREAD, PRTYC_TIMECRITICAL, PRTYD_MAXIMUM - 1, 0);
76 
77  while (!FinishPlayback) {
78  ULONG count;
79 static ULONG NextBuffer = 0; /* next fragment to be filled */
80 
81  /* wait for play enable */
82  DosWaitEventSem(Play, SEM_INDEFINITE_WAIT);
83  /* wait for timer event */
84  DosWaitEventSem(Update, SEM_INDEFINITE_WAIT);
85  /* reset timer semaphore */
86  DosResetEventSem(Update, &count);
87 
88  /* fill all free buffers */
89  while (PlayList[NextBuffer].ulOperand3 >=
90  PlayList[NextBuffer].ulOperand2) {
91  MUTEX_LOCK(vars);
92  if (Player_Paused_internal())
93  PlayList[NextBuffer].ulOperand2 =
94  VC_SilenceBytes((SBYTE *)PlayList[NextBuffer].ulOperand1,
95  BufferSize);
96  else
97  PlayList[NextBuffer].ulOperand2 =
98  VC_WriteBytes((SBYTE *)PlayList[NextBuffer].ulOperand1,
99  BufferSize);
100  MUTEX_UNLOCK(vars);
101  /* Reset play offset, although it seems MMOS2 does it
102  automagically */
103  PlayList[NextBuffer].ulOperand3 = 0;
104  NextBuffer = (NextBuffer + 1) % FRAGMENTS;
105  }
106  }
107  /* Tell main thread we're done */
108  ThreadID = 0;
109 }

130 /* checks for availability of an OS/2 MCI-WAVEAUDIO-device */
131 static BOOL OS2_IsPresent(void)
132 {
133  MCI_OPEN_PARMS mciOpenParms;
134  MCI_GENERIC_PARMS mciGenericParms;
135 
136  memset(&mciOpenParms, 0, sizeof(mciOpenParms));
137  mciOpenParms.pszDeviceType = (PSZ) MAKEULONG(MCI_DEVTYPE_WAVEFORM_AUDIO,
138  (USHORT) DeviceIndex);
139 
140  /* try to open WAVEAUDIO-device */
141  if (mciSendCommand(0, MCI_OPEN,
142  MCI_WAIT | MCI_OPEN_SHAREABLE | MCI_OPEN_TYPE_ID,
143  (PVOID) & mciOpenParms, 0) != MCIERR_SUCCESS)
144  return 0;
145 
146  mciSendCommand(mciOpenParms.usDeviceID, MCI_CLOSE, MCI_WAIT,
147  (PVOID) & mciGenericParms, 0);
148  return 1;
149 }

151 static BOOL OS2_Init(void)
152 {
153  MCI_OPEN_PARMS mciOpenParms;
154  MCI_WAVE_SET_PARMS mciWaveSetParms;
155  int i, bpsrate;
156  ULONG ulInterval;
157 
158  if (VC_Init())
159  return 1;
160 
161  DeviceID = 0;
162  ThreadID = 0;
163  Timer = NULLHANDLE;
164  Update = NULLHANDLE;
165  Play = NULLHANDLE;
166  AudioBuffer = NULL;
167 
168  /* Compute the bytes per second output rate */
169  bpsrate = md_mixfreq;
170  if (md_mode & DMODE_STEREO)
171  bpsrate <<= 1;
172  if (md_mode & DMODE_16BITS)
173  bpsrate <<= 1;
174 
175  /* Compute audio buffer size if necessary */
176  if (BufferSize == (ULONG)-1) {
177  int bit;
178 
179  BufferSize = bpsrate >> 3;
180  for (bit = 15; bit >= 12; bit--)
181  if (BufferSize & (1 << bit))
182  break;
183  BufferSize = 1 << bit;
184  }
185 
186  /* WARNING: with big buffer sizes there is an audible click between
187  sub-blocks. Don't know whenever this is a driver issue (I've tested it
188  only with Gravis Ultrasound driver) but its something to take into
189  consideration. That's why above code chooses relatively small buffer
190  sizes (16K for 44KHz, 16 bit stereo). */
191 
192  /* Allocate buffer */
193  if (!(AudioBuffer = _mm_malloc(BufferSize * FRAGMENTS))) {
194  _mm_errno = MMERR_OUT_OF_MEMORY;
195  return 1;
196  }
197 
198  /* create playlist */
199  for (i = 0; i < FRAGMENTS; i++) {
200  PlayList[i].ulCommand = DATA_OPERATION; /* play data */
201  PlayList[i].ulOperand1 = ((ULONG)AudioBuffer) + i * BufferSize;
202  PlayList[i].ulOperand2 = BufferSize; /* size */
203  PlayList[i].ulOperand3 = 0; /* offset */
204  }
205  PlayList[FRAGMENTS].ulCommand = BRANCH_OPERATION; /* jump */
206  PlayList[FRAGMENTS].ulOperand1 = 0;
207  PlayList[FRAGMENTS].ulOperand2 = 0; /* destination */
208  PlayList[FRAGMENTS].ulOperand3 = 0;
209 
210  memset(&mciOpenParms, 0, sizeof(mciOpenParms));
211  mciOpenParms.pszDeviceType = (PSZ) MAKEULONG(MCI_DEVTYPE_WAVEFORM_AUDIO,
212  (USHORT) DeviceIndex);
213  mciOpenParms.pszElementName = (PSZ) & PlayList;
214 
215  /* open WAVEAUDIO-device */
216  if (mciSendCommand(0, MCI_OPEN,
217  MCI_WAIT | MCI_OPEN_TYPE_ID | MCI_OPEN_PLAYLIST,
218  (PVOID) & mciOpenParms, 0)) {
219  _mm_errno = MMERR_OPENING_AUDIO;
220  return 1;
221  }
222  DeviceID = mciOpenParms.usDeviceID;
223 
224  memset(&mciWaveSetParms, 0, sizeof(mciWaveSetParms));
225  mciWaveSetParms.ulSamplesPerSec = md_mixfreq;
226  mciWaveSetParms.usBitsPerSample = (md_mode & DMODE_16BITS) ? 16 : 8;
227  mciWaveSetParms.usChannels = (md_mode & DMODE_STEREO) ? 2 : 1;
228  mciWaveSetParms.ulAudio = MCI_SET_AUDIO_ALL;
229 
230  /* set play-parameters */
231  if (mciSendCommand(DeviceID, MCI_SET,
232  MCI_WAIT | MCI_WAVE_SET_SAMPLESPERSEC |
233  MCI_WAVE_SET_BITSPERSAMPLE | MCI_WAVE_SET_CHANNELS,
234  (PVOID) & mciWaveSetParms, 0)) {
235  _mm_errno = MMERR_OS2_MIXSETUP;
236  return 1;
237  }
238 
239  /* Setup semaphore for playing and for buffer updates */
240  if (DosCreateEventSem(NULL, &Play, DC_SEM_SHARED, FALSE) ||
241  DosCreateEventSem(NULL, &Update, DC_SEM_SHARED, FALSE)) {
242  _mm_errno = MMERR_OS2_SEMAPHORE;
243  return 1;
244  }
245 
246  /* Compute interval in ms (2 possible updates while playing 1 block) */
247  ulInterval = BufferSize * 1000L / bpsrate;
248  /* Setup timer for buffer updates */
249  if (DosStartTimer(ulInterval, (HSEM) Update, &Timer)) {
250  _mm_errno = MMERR_OS2_TIMER;
251  return 1;
252  }
253 
254  /* Create thread for buffer updates */
255  FinishPlayback = FALSE;
256  ThreadID = _beginthread(OS2_UpdateBufferThread, NULL, 0x4000, NULL);
257  if (!ThreadID) {
258  _mm_errno = MMERR_OS2_THREAD;
259  return 1;
260  }
261 
262  return 0;
263 }

265 static void OS2_Exit(void)
266 {
267  MCI_GENERIC_PARMS mciGenericParms;
268 
269  FinishPlayback = TRUE;
270  while (ThreadID) {
271  DosPostEventSem(Play);
272  DosPostEventSem(Update);
273  DosSleep(1);
274  }
275  if (Timer) {
276  DosStopTimer(Timer);
277  Timer = NULLHANDLE;
278  }
279  if (Update) {
280  DosCloseEventSem(Update);
281  Update = NULLHANDLE;
282  }
283  if (Play) {
284  DosCloseEventSem(Play);
285  Play = NULLHANDLE;
286  }
287  VC_Exit();
288  if (DeviceID) {
289  mciGenericParms.hwndCallback = (HWND) NULL;
290  mciSendCommand(DeviceID, MCI_CLOSE, MCI_WAIT,
291  (PVOID) & mciGenericParms, 0);
292  DeviceID = 0;
293  }
294  _mm_free(AudioBuffer);
295 }

297 static BOOL OS2_PlayStart(void)
298 {
299  MCI_PLAY_PARMS mciPlayParms;
300  int i;
301 
302  VC_SilenceBytes((SBYTE *)AudioBuffer, BufferSize * FRAGMENTS);
303  VC_PlayStart();
304 
305  /* Signal update of audio fragments */
306  for (i = 0; i < FRAGMENTS; i++)
307  PlayList[i].ulOperand3 = 0;
308 
309  /* Signal play */
310  DosPostEventSem(Play);
311 
312  /* no MCI_WAIT here, because application program has to continue */
313  memset(&mciPlayParms, 0, sizeof(mciPlayParms));
314  if (mciSendCommand(DeviceID, MCI_PLAY, MCI_FROM, &mciPlayParms, 0)) {
315  _mm_errno = MMERR_OS2_MIXSETUP;
316  return 1;
317  }
318 
319  return 0;
320 }
