106  while(1){

108 
109  if(boundary>0 && vf->offset>=boundary)return(OV_FALSE);
110  more=ogg_sync_pageseek(&vf->oy,og);

126  advance the internal offset past the page end */

658 /* fetch and process a packet. Handles the case where we're at a
659  bitstream boundary and dumps the decoding machine. If the decoding
660  machine is unloaded, it loads it. It also keeps pcm_offset up to
661  date (seek and read both use this. seek uses a special hack with
662  readp).
663 
664  return: <0) error, OV_HOLE (lost packet) or OV_EOF
665  0) need more data (only if readp==0)
666  1) got a packet
667 */
668 
669 static int _fetch_and_process_packet(OggVorbis_File *vf,
670  ogg_packet *op_in,
671  int readp,
672  int spanp){
673  ogg_page og;
674 
675  /* handle one packet. Try to fetch it from current stream state */
676  /* extract packets from page */
677  while(1){
678 
679  if(vf->ready_state==STREAMSET){
680  int ret=_make_decode_ready(vf);
681  if(ret<0)return ret;
682  }
683 
684  /* process a packet if we can. If the machine isn't loaded,
685  neither is a page */
686  if(vf->ready_state==INITSET){
687  while(1) {
688  ogg_packet op;
689  ogg_packet *op_ptr=(op_in?op_in:&op);
690  int result=ogg_stream_packetout(&vf->os,op_ptr);
691  ogg_int64_t granulepos;
692 
693  op_in=NULL;
694  if(result==-1)return(OV_HOLE); /* hole in the data. */
695  if(result>0){
696  /* got a packet. process it */
697  granulepos=op_ptr->granulepos;
698  if(!vorbis_synthesis(&vf->vb,op_ptr)){ /* lazy check for lazy
699  header handling. The
700  header packets aren't
701  audio, so if/when we
702  submit them,
703  vorbis_synthesis will
704  reject them */
705 
706  /* suck in the synthesis data and track bitrate */
707  {
708  int oldsamples=vorbis_synthesis_pcmout(&vf->vd,NULL);
709  /* for proper use of libvorbis within libvorbisfile,
710  oldsamples will always be zero. */
711  if(oldsamples)return(OV_EFAULT);
712 
713  vorbis_synthesis_blockin(&vf->vd,&vf->vb);
714  vf->samptrack+=vorbis_synthesis_pcmout(&vf->vd,NULL)-oldsamples;
715  vf->bittrack+=op_ptr->bytes*8;
716  }
717 
718  /* update the pcm offset. */
719  if(granulepos!=-1 && !op_ptr->e_o_s){
720  int link=(vf->seekable?vf->current_link:0);
721  int i,samples;
722 
723  /* this packet has a pcm_offset on it (the last packet
724  completed on a page carries the offset) After processing
725  (above), we know the pcm position of the *last* sample
726  ready to be returned. Find the offset of the *first*
727 
728  As an aside, this trick is inaccurate if we begin
729  reading anew right at the last page; the end-of-stream
730  granulepos declares the last frame in the stream, and the
731  last packet of the last page may be a partial frame.
732  So, we need a previous granulepos from an in-sequence page
733  to have a reference point. Thus the !op_ptr->e_o_s clause
734  above */
735 
736  if(vf->seekable && link>0)
737  granulepos-=vf->pcmlengths[link*2];
738  if(granulepos<0)granulepos=0; /* actually, this
739  shouldn't be possible
740  here unless the stream
741  is very broken */
742 
743  samples=vorbis_synthesis_pcmout(&vf->vd,NULL);
744 
745  granulepos-=samples;
746  for(i=0;i<link;i++)
747  granulepos+=vf->pcmlengths[i*2+1];
748  vf->pcm_offset=granulepos;
749  }
750  return(1);
751  }
752  }
753  else
754  break;
755  }
756  }
757 
758  if(vf->ready_state>=OPENED){
759  ogg_int64_t ret;
760 
761  while(1){
762  /* the loop is not strictly necessary, but there's no sense in
763  doing the extra checks of the larger loop for the common
764  case in a multiplexed bistream where the page is simply
765  part of a different logical bitstream; keep reading until
766  we get one with the correct serialno */
767 
768  if(!readp)return(0);
769  if((ret=_get_next_page(vf,&og,-1))<0){
770  return(OV_EOF); /* eof. leave unitialized */
771  }
772 
773  /* bitrate tracking; add the header's bytes here, the body bytes
774  are done by packet above */
775  vf->bittrack+=og.header_len*8;
776 
777  if(vf->ready_state==INITSET){
778  if(vf->current_serialno!=ogg_page_serialno(&og)){
779 
780  /* two possibilities:
781  1) our decoding just traversed a bitstream boundary
782  2) another stream is multiplexed into this logical section */
783 
784  if(ogg_page_bos(&og)){
785  /* boundary case */
786  if(!spanp)
787  return(OV_EOF);
788 
789  _decode_clear(vf);
790 
791  if(!vf->seekable){
792  vorbis_info_clear(vf->vi);
793  vorbis_comment_clear(vf->vc);
794  }
795  break;
796 
797  }else
798  continue; /* possibility #2 */
799  }
800  }
801 
802  break;
803  }
804  }
805 
806  /* Do we need to load a new machine before submitting the page? */
807  /* This is different in the seekable and non-seekable cases.
808 
809  In the seekable case, we already have all the header
810  information loaded and cached; we just initialize the machine
811  with it and continue on our merry way.
812 
813  In the non-seekable (streaming) case, we'll only be at a
814  boundary if we just left the previous logical bitstream and
815  we're now nominally at the header of the next bitstream
816  */
817 
818  if(vf->ready_state!=INITSET){
819  int link;
820 
821  if(vf->ready_state<STREAMSET){
822  if(vf->seekable){
823  ogg_uint32_t serialno = ogg_page_serialno(&og);
824 
825  /* match the serialno to bitstream section. We use this rather than
826  offset positions to avoid problems near logical bitstream
827  boundaries */
828 
829  for(link=0;link<vf->links;link++)
830  if(vf->serialnos[link]==serialno)break;
831 
832  if(link==vf->links) continue; /* not the desired Vorbis
833  bitstream section; keep
834  trying */
835 
836  vf->current_serialno=serialno;
837  vf->current_link=link;
838 
839  ogg_stream_reset_serialno(&vf->os,vf->current_serialno);
840  vf->ready_state=STREAMSET;
841 
842  }else{
843  /* we're streaming */
844  /* fetch the three header packets, build the info struct */
845 
846  int ret=_fetch_headers(vf,vf->vi,vf->vc,NULL,NULL,&og);
847  if(ret)return(ret);
848  vf->current_serialno=vf->os.serialno;
849  vf->current_link++;
850  link=0;
851  }
852  }
853  }
854 
855  /* the buffered page is the data we want, and we're ready for it;
856  add it to the stream state */
857  ogg_stream_pagein(&vf->os,&og);
858 
859  }

1214  advance the raw cursor past good packets just to get to the first
1215  with a granulepos. That's not equivalent behavior to beginning
1216  decoding as immediately after the seek position as possible.
1217

1219  the shared vf->os stream state. We use the local state to
1220  scan, and the shared state as a buffer for later decode.
1221

1223  because the granulepos on the last page is not necessarily on a
1224  packet boundary, and we need to make sure the granpos is
1225  correct.

1231  int lastblock=0;
1232  int accblock=0;
1233  int thisblock=0;
1234  int lastflag=0;
1235  int firstflag=0;
1236  ogg_int64_t pagepos=-1;
1237 
1238  ogg_stream_init(&work_os,vf->current_serialno); /* get the memory ready */
1239  ogg_stream_reset(&work_os); /* eliminate the spurious OV_HOLE
1240  return from not necessarily
1241  starting from the beginning */
1242 
1243  while(1){
