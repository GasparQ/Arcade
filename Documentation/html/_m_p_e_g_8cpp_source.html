380 bool MPEG::seekIntoStream(int position)
381 {
382  /* First we stop everything */
383  Stop();
384 
385  /* Go to the desired position into file */
386  if(!system->Seek(position)) return(false);
387 
388  /* Seek first aligned data */
389  if(audiostream && audioaction_enabled)
390  while(audiostream->time() == -1)
391  if ( ! audiostream->next_packet() ) return false;
392  if(videostream && videoaction_enabled)
393  while(videostream->time() == -1)
394  if ( ! videostream->next_packet() ) return false;
395 
396  /* Calculating current play time on audio only makes sense when there
397  is no video */
398  if ( audioaction && !videoaction) {
399  audioaction->Rewind();
400  audioaction->ResetSynchro(system->TimeElapsedAudio(position));
401  }
402  /* And forget what we previouly buffered */
403  else if ( audioaction ) {
404  audioaction->Rewind();
405  audioaction->ResetSynchro(audiostream->time());
406  }
407  if ( videoaction ) {
408  videoaction->Rewind();
409  videoaction->ResetSynchro(videostream->time());
410  }
