88 /* Limits the number of hardware voices to the specified amount.
89  This function should only be used by the low-level drivers. */
90 static void LimitHardVoices(int limit)
91 {
92  int t=0;
93 
94  if (!(md_mode & DMODE_SOFT_SNDFX) && (md_sfxchn>limit)) md_sfxchn=limit;
95  if (!(md_mode & DMODE_SOFT_MUSIC) && (md_sngchn>limit)) md_sngchn=limit;
96 
97  if (!(md_mode & DMODE_SOFT_SNDFX))
98  md_hardchn=md_sfxchn;
99  else
100  md_hardchn=0;
101 
102  if (!(md_mode & DMODE_SOFT_MUSIC)) md_hardchn += md_sngchn;
103 
104  while (md_hardchn>limit) {
105  if (++t & 1) {
106  if (!(md_mode & DMODE_SOFT_SNDFX) && (md_sfxchn>4)) md_sfxchn--;
107  } else {
108  if (!(md_mode & DMODE_SOFT_MUSIC) && (md_sngchn>8)) md_sngchn--;
109  }
110 
111  if (!(md_mode & DMODE_SOFT_SNDFX))
112  md_hardchn=md_sfxchn;
113  else
114  md_hardchn=0;
115 
116  if (!(md_mode & DMODE_SOFT_MUSIC))
117  md_hardchn+=md_sngchn;
118  }
119  md_numchn=md_hardchn+md_softchn;
120 }

478 static BOOL _mm_init(CHAR *cmdline)
479 {
480  UWORD t;
481 
482  _mm_critical = 1;
483 
484  /* if md_device==0, try to find a device number */
485  if(!md_device) {
486  cmdline=NULL;
487 
488  for(t=1,md_driver=firstdriver;md_driver;md_driver=md_driver->next,t++)
489  if(md_driver->IsPresent()) break;
490 
491  if(!md_driver) {
492  _mm_errno = MMERR_DETECTING_DEVICE;
493  if(_mm_errorhandler) _mm_errorhandler();
494  md_driver = &drv_nos;
495  return 1;
496  }
497 
498  md_device = t;
499  } else {
500  /* if n>0, use that driver */
501  for(t=1,md_driver=firstdriver;(md_driver)&&(t!=md_device);md_driver=md_driver->next)
502  t++;
503 
504  if(!md_driver) {
505  _mm_errno = MMERR_INVALID_DEVICE;
506  if(_mm_errorhandler) _mm_errorhandler();
507  md_driver = &drv_nos;
508  return 1;
509  }
510 
511  /* arguments here might be necessary for the presence check to succeed */
512  if(cmdline&&(md_driver->CommandLine))
513  md_driver->CommandLine(cmdline);
514 
515  if(!md_driver->IsPresent()) {
516  _mm_errno = MMERR_DETECTING_DEVICE;
517  if(_mm_errorhandler) _mm_errorhandler();
518  md_driver = &drv_nos;
519  return 1;
520  }
521  }
522 
523  olddevice = md_device;
524  if(md_driver->Init()) {
525  MikMod_Exit_internal();
526  if(_mm_errorhandler) _mm_errorhandler();
527  return 1;
528  }
529 
530  initialized=1;
531  _mm_critical=0;
532 
533  return 0;
534 }

573 /* Reset the driver using the new global variable settings.
574  If the driver has not been initialized, it will be now. */
575 static BOOL _mm_reset(CHAR *cmdline)
576 {
577  BOOL wasplaying = 0;
578 
579  if(!initialized) return _mm_init(cmdline);
580 
581  if (isplaying) {
582  wasplaying = 1;
583  md_driver->PlayStop();
584  }
585 
586  if((!md_driver->Reset)||(md_device != olddevice)) {
587  /* md_driver->Reset was NULL, or md_device was changed, so do a full
588  reset of the driver. */
589  md_driver->Exit();
590  if(_mm_init(cmdline)) {
591  MikMod_Exit_internal();
592  if(_mm_errno)
593  if(_mm_errorhandler) _mm_errorhandler();
594  return 1;
595  }
596  } else {
597  if(md_driver->Reset()) {
598  MikMod_Exit_internal();
599  if(_mm_errno)
600  if(_mm_errorhandler) _mm_errorhandler();
601  return 1;
602  }
603  }
604 
605  if (wasplaying) md_driver->PlayStart();
606  return 0;
607 }

622 /* If either parameter is -1, the current set value will be retained. */
623 BOOL MikMod_SetNumVoices_internal(int music, int sfx)
624 {
625  BOOL resume = 0;
626  int t, oldchn = 0;
627 
628  if((!music)&&(!sfx)) return 1;
629  _mm_critical = 1;
630  if(isplaying) {
631  MikMod_DisableOutput_internal();
632  oldchn = md_numchn;
633  resume = 1;
634  }
635 
636  if(sfxinfo) free(sfxinfo);
637  if(md_sample) free(md_sample);
638  md_sample = NULL;
639  sfxinfo = NULL;
640 
641  if(music!=-1) md_sngchn = music;
642  if(sfx!=-1) md_sfxchn = sfx;
643  md_numchn = md_sngchn + md_sfxchn;
644 
645  LimitHardVoices(md_driver->HardVoiceLimit);
646  LimitSoftVoices(md_driver->SoftVoiceLimit);
647 
648  if(md_driver->SetNumVoices()) {
649  MikMod_Exit_internal();
650  if(_mm_errno)
651  if(_mm_errorhandler!=NULL) _mm_errorhandler();
652  md_numchn = md_softchn = md_hardchn = md_sfxchn = md_sngchn = 0;
653  return 1;
654  }
655 
656  if(md_sngchn+md_sfxchn)
657  md_sample=(SAMPLE**)_mm_calloc(md_sngchn+md_sfxchn,sizeof(SAMPLE*));
658  if(md_sfxchn)
659  sfxinfo = (UBYTE *)_mm_calloc(md_sfxchn,sizeof(UBYTE));
660 
661  /* make sure the player doesn't start with garbage */
662  for(t=oldchn;t<md_numchn;t++) Voice_Stop_internal(t);
663 
664  sfxpool = 0;
665  if(resume) MikMod_EnableOutput_internal();
666  _mm_critical = 0;
667 
668  return 0;
669 }

741 {

743  int c;

885 /* Check if the file is a regular or nonexistant file (or a link to a such a
886  file), and that, should the calling program be setuid, the access rights are
887  reasonable. Returns 1 if it is safe to rewrite the file, 0 otherwise.
888  The goal is to prevent a setuid root libmikmod application from overriding
889  files like /etc/passwd with digital sound... */
890 BOOL MD_Access(CHAR *filename)
891 {
892  struct stat buf;
893 
894  if(!stat(filename,&buf)) {
895  /* not a regular file ? */
896  if(!S_ISREG(buf.st_mode)) return 0;
897  /* more than one hard link to the file ? */
898  if(buf.st_nlink>1) return 0;
899  /* check access rights with the real user and group id */
900  if(getuid()==buf.st_uid) {
901  if(!(buf.st_mode&S_IWUSR)) return 0;
902  } else if(getgid()==buf.st_gid) {
903  if(!(buf.st_mode&S_IWGRP)) return 0;
904  } else
905  if(!(buf.st_mode&S_IWOTH)) return 0;
906  }
907 
908  return 1;
909 }

917  if(setuid(getuid())) return 1;

924  uid=nobody->pw_uid;

926  return 1;
927  if (setuid(uid)) return 1;
928  }
