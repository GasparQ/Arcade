412  }

434  int pos;

439

441  {

443 
444  FD_ZERO(&fdset);
445  FD_SET(pipe_fd[1], &fdset);
446  timeout.tv_sec = 10;
447  timeout.tv_usec = 0;
448  if(select(pipe_fd[1]+1, NULL, &fdset, NULL, &timeout) <= 0) break;
449  if(write(pipe_fd[1], buffer, CD_FRAMESIZE_RAW0) < 0) break;
450  }
451

453  exit(0);
454  }
455
