97  int numParts = 0;
98  Uint16 ppqn;
99 
100  /* Init the arrays */
101  memset(part_poly_max,0,sizeof(part_poly_max));
102  memset(part_to_inst,-1,sizeof(part_to_inst));
103 
104  /* Attempt to load the midi file */
105  evntlist = CreateMIDIEventList(src, &ppqn);
106  if (!evntlist)
107  goto bail;
108 
109  /* Allocate memory for the song struct */
110  song = malloc(sizeof(NativeMidiSong));
111  if (!song)
112  goto bail;
113 
114  /* Build a tune sequence from the event list */
115  song->tuneSequence = BuildTuneSequence(evntlist, ppqn, part_poly_max, part_to_inst, &numParts);
116  if(!song->tuneSequence)
117  goto bail;
118 
119  /* Now build a tune header from the data we collect above, create
120  all parts as needed and assign them the correct instrument.
121  */
122  song->tuneHeader = BuildTuneHeader(part_poly_max, part_to_inst, numParts);
123  if(!song->tuneHeader)
124  goto bail;
125 
126  /* Increment the instance count */
127  gInstaceCount++;
128  if (gTunePlayer == NULL)
129  gTunePlayer = OpenDefaultComponent(kTunePlayerComponentType, 0);
130 
131  /* Finally, free the event list */
132  FreeMIDIEventList(evntlist);
133 
134  if (freerw) {
135  SDL_RWclose(rw);
136  }
137  return song;
138 
139 bail:
140  if (evntlist)
141  FreeMIDIEventList(evntlist);
142 
143  if (song)
144  {
145  if(song->tuneSequence)
146  free(song->tuneSequence);
147 
148  if(song->tuneHeader)
149  DisposePtr((Ptr)song->tuneHeader);
150 
151  free(song);
152  }
153  return NULL;

180 {
181  UInt32 queueFlags = 0;
182  ComponentResult tpError;
183 
184  assert (gTunePlayer != NULL);
185 
186  /* FIXME: is this code even used anymore? */
187  assert (loops == 0);
188 
189  SDL_PauseAudio(1);
190  SDL_UnlockAudio();
191 
192  /* First, stop the currently playing music */
193  native_midi_stop();
194 
195  /* Set up the queue flags */
196  queueFlags = kTuneStartNow;
197 
198  /* Set the time scale (units per second), we want milliseconds */
199  tpError = TuneSetTimeScale(gTunePlayer, 1000);
200  if (tpError != noErr)

203  goto done;
204  }
205 
206  /* Set the header, to tell what instruments are used */
207  tpError = TuneSetHeader(gTunePlayer, (UInt32 *)song->tuneHeader);
208  if (tpError != noErr)
209  {
210  strncpy (gErrorBuffer, "MIDI error during TuneSetHeader", ERROR_BUF_SIZE);
211  goto done;
212  }
213 
214  /* Have it allocate whatever resources are needed */
215  tpError = TunePreroll(gTunePlayer);
216  if (tpError != noErr)
217  {
218  strncpy (gErrorBuffer, "MIDI error during TunePreroll", ERROR_BUF_SIZE);
219  goto done;
220  }
221 
222  /* We want to play at normal volume */
223  tpError = TuneSetVolume(gTunePlayer, 0x00010000);
224  if (tpError != noErr)
225  {
226  strncpy (gErrorBuffer, "MIDI error during TuneSetVolume", ERROR_BUF_SIZE);
227  goto done;
228  }
229 
230  /* Finally, start playing the full song */
231  gCurrentTuneSequence = song->tuneSequence;
232  tpError = TuneQueue(gTunePlayer, (UInt32 *)song->tuneSequence, 0x00010000, 0, 0xFFFFFFFF, queueFlags, NULL, 0);
233  if (tpError != noErr)
234  {
235  strncpy (gErrorBuffer, "MIDI error during TuneQueue", ERROR_BUF_SIZE);
236  goto done;
237  }
238 
239 done:
240  SDL_LockAudio();
241  SDL_PauseAudio(0);
242 }

283 Uint32 *BuildTuneSequence(MIDIEvent *evntlist, int ppqn, int part_poly_max[32], int part_to_inst[32], int *numParts)
284 {
285  int part_poly[32];
286  int channel_to_part[16];
287 
288  int channel_pan[16];
289  int channel_vol[16];
290  int channel_pitch_bend[16];
291 
292  int lastEventTime = 0;
293  int tempo = 500000;
294  double Ippqn = 1.0 / (1000*ppqn);
295  double tick = tempo * Ippqn;
296  MIDIEvent *eventPos = evntlist;
297  MIDIEvent *noteOffPos;
298  Uint32 *tunePos, *endPos;
299  Uint32 *tuneSequence;
300  size_t tuneSize;
301 
302  /* allocate space for the tune header */
303  tuneSize = 5000;
304  tuneSequence = (Uint32 *)malloc(tuneSize * sizeof(Uint32));
305  if (tuneSequence == NULL)
306  return NULL;
307 
308  /* Set starting position in our tune memory */
309  tunePos = tuneSequence;
310  endPos = tuneSequence + tuneSize;
311 
312  /* Initialise the arrays */
313  memset(part_poly,0,sizeof(part_poly));
314 
315  memset(channel_to_part,-1,sizeof(channel_to_part));
316  memset(channel_pan,-1,sizeof(channel_pan));
317  memset(channel_vol,-1,sizeof(channel_vol));
318  memset(channel_pitch_bend,-1,sizeof(channel_pitch_bend));
319 
320  *numParts = 0;
321 
322  /*
323  * Now the major work - iterate over all GM events,
324  * and turn them into QuickTime Music format.
325  * At the same time, calculate the max. polyphony for each part,
326  * and also the part->instrument mapping.
327  */
328  while(eventPos)
329  {
330  int status = (eventPos->status&0xF0)>>4;
331  int channel = eventPos->status&0x0F;
332  int part = channel_to_part[channel];
333  int velocity, pitch;
334  int value, controller;
335  int bend;
336  int newInst;
337 
338  /* Check if we are running low on space... */
339  if((tunePos+16) > endPos)
340  {
341  /* Resize our data storage. */
342  Uint32 *oldTuneSequence = tuneSequence;
343 
344  tuneSize += BUFFER_INCREMENT;
345  tuneSequence = (Uint32 *)realloc(tuneSequence, tuneSize * sizeof(Uint32));
346  if(oldTuneSequence != tuneSequence)
347  tunePos += tuneSequence - oldTuneSequence;
348  endPos = tuneSequence + tuneSize;
349  }
350 
351  switch (status)
352  {
353  case MIDI_STATUS_NOTE_OFF:
354  assert(part>=0 && part<=31);
355 
356  /* Keep track of the polyphony of the current part */
357  part_poly[part]--;
358  break;
359  case MIDI_STATUS_NOTE_ON:
360  if (part < 0)
361  {
362  /* If no part is specified yet, we default to the first instrument, which
363  is piano (or the first drum kit if we are on the drum channel)
364  */
365  int newInst;
366 
367  if (channel == 9)
368  newInst = kFirstDrumkit + 1; /* the first drum kit is the "no drum" kit! */
369  else
370  newInst = kFirstGMInstrument;
371  part = channel_to_part[channel] = *numParts;
372  part_to_inst[(*numParts)++] = newInst;
373  }
374  /* TODO - add support for more than 32 parts using eXtended QTMA events */
375  assert(part<=31);
376 
377  /* Decode pitch & velocity */
378  pitch = eventPos->data[0];
379  velocity = eventPos->data[1];
380 
381  if (velocity == 0)
382  {
383  /* was a NOTE OFF in disguise, so we decrement the polyphony */
384  part_poly[part]--;
385  }
386  else
387  {
388  /* Keep track of the polyphony of the current part */
389  int foo = ++part_poly[part];
390  if (part_poly_max[part] < foo)
391  part_poly_max[part] = foo;
392 
393  /* Now scan forward to find the matching NOTE OFF event */
394  for(noteOffPos = eventPos; noteOffPos; noteOffPos = noteOffPos->next)
395  {
396  if ((noteOffPos->status&0xF0)>>4 == MIDI_STATUS_NOTE_OFF
397  && channel == (eventPos->status&0x0F)
398  && pitch == noteOffPos->data[0])
399  break;
400  /* NOTE ON with velocity == 0 is the same as a NOTE OFF */
401  if ((noteOffPos->status&0xF0)>>4 == MIDI_STATUS_NOTE_ON
402  && channel == (eventPos->status&0x0F)
403  && pitch == noteOffPos->data[0]
404  && 0 == noteOffPos->data[1])
405  break;
406  }
407 
408  /* Did we find a note off? Should always be the case, but who knows... */
409  if (noteOffPos)
410  {
411  /* We found a NOTE OFF, now calculate the note duration */
412  int duration = (int)((noteOffPos->time - eventPos->time)*tick);
413 
414  REST_IF_NECESSARY();
415  /* Now we need to check if we get along with a normal Note Event, or if we need an extended one... */
416  if (duration < 2048 && pitch>=32 && pitch<=95 && velocity>=0 && velocity<=127)
417  {
418  qtma_StuffNoteEvent(*tunePos, part, pitch, velocity, duration);
419  tunePos++;
420  }
421  else
422  {
423  qtma_StuffXNoteEvent(*tunePos, *(tunePos+1), part, pitch, velocity, duration);
424  tunePos+=2;
425  }
426  }
427  }
428  break;
429  case MIDI_STATUS_AFTERTOUCH:
430  /* NYI - use kControllerAfterTouch. But how are the parameters to be mapped? */
431  break;
432  case MIDI_STATUS_CONTROLLER:
433  controller = eventPos->data[0];
434  value = eventPos->data[1];
435 
436  switch(controller)
437  {
438  case 0: /* bank change - igore for now */
439  break;
440  case kControllerVolume:
441  if(channel_vol[channel] != value<<8)
442  {
443  channel_vol[channel] = value<<8;
444  if(part>=0 && part<=31)
445  {
446  REST_IF_NECESSARY();
447  qtma_StuffControlEvent(*tunePos, part, kControllerVolume, channel_vol[channel]);
448  tunePos++;
449  }
450  }
451  break;
452  case kControllerPan:
453  if(channel_pan[channel] != (value << 1) + 256)
454  {
455  channel_pan[channel] = (value << 1) + 256;
456  if(part>=0 && part<=31)
457  {
458  REST_IF_NECESSARY();
459  qtma_StuffControlEvent(*tunePos, part, kControllerPan, channel_pan[channel]);
460  tunePos++;
461  }
462  }
463  break;
464  default:
465  /* No other controllers implemented yet */;
466  break;
467  }
468 
469  break;
470  case MIDI_STATUS_PROG_CHANGE:
471  /* Instrument changed */
472  newInst = eventPos->data[0];
473 
474  /* Channel 9 (the 10th channel) is different, it indicates a drum kit */
475  if (channel == 9)
476  newInst += kFirstDrumkit;
477  else
478  newInst += kFirstGMInstrument;
479  /* Only if the instrument for this channel *really* changed, add a new part. */
480  if(newInst != part_to_inst[part])
481  {
482  /* TODO maybe make use of kGeneralEventPartChange here,
483  to help QT reuse note channels?
484  */
485  part = channel_to_part[channel] = *numParts;
486  part_to_inst[(*numParts)++] = newInst;
487 
488  if(channel_vol[channel] >= 0)
489  {
490  REST_IF_NECESSARY();
491  qtma_StuffControlEvent(*tunePos, part, kControllerVolume, channel_vol[channel]);
492  tunePos++;
493  }
494  if(channel_pan[channel] >= 0)
495  {
496  REST_IF_NECESSARY();
497  qtma_StuffControlEvent(*tunePos, part, kControllerPan, channel_pan[channel]);
498  tunePos++;
499  }
500  if(channel_pitch_bend[channel] >= 0)
501  {
502  REST_IF_NECESSARY();
503  qtma_StuffControlEvent(*tunePos, part, kControllerPitchBend, channel_pitch_bend[channel]);
504  tunePos++;
505  }
506  }
507  break;
508  case MIDI_STATUS_PRESSURE:
509  /* NYI */
510  break;
511  case MIDI_STATUS_PITCH_WHEEL:
512  /* In the midi spec, 0x2000 = center, 0x0000 = - 2 semitones, 0x3FFF = +2 semitones
513  but for QTMA, we specify it as a 8.8 fixed point of semitones
514  TODO: detect "pitch bend range changes" & honor them!
515  */
516  bend = (eventPos->data[0] & 0x7f) | ((eventPos->data[1] & 0x7f) << 7);
517 
518  /* "Center" the bend */
519  bend -= 0x2000;
520 
521  /* Move it to our format: */
522  bend <<= 4;
523 
524  /* If it turns out the pitch bend didn't change, stop here */
525  if(channel_pitch_bend[channel] == bend)
526  break;
527 
528  channel_pitch_bend[channel] = bend;
529  if(part>=0 && part<=31)
530  {
531  /* Stuff a control event */
532  REST_IF_NECESSARY();
533  qtma_StuffControlEvent(*tunePos, part, kControllerPitchBend, bend);
534  tunePos++;
535  }
536  break;
537  case MIDI_STATUS_SYSEX:
538  if (eventPos->status == 0xFF && eventPos->data[0] == 0x51) /* Tempo change */
539  {
540  tempo = (eventPos->extraData[0] << 16) +
541  (eventPos->extraData[1] << 8) +
542  eventPos->extraData[2];
543 
544  tick = tempo * Ippqn;
545  }
546  break;
547  }
548 
549  /* on to the next event */
550  eventPos = eventPos->next;
551  }
552 
553  /* Finally, place an end marker */
554  *tunePos = kEndMarkerValue;
555 
556  return tuneSequence;
557 }

559 Uint32 *BuildTuneHeader(int part_poly_max[32], int part_to_inst[32], int numParts)
560 {
561  Uint32 *myHeader;
562  Uint32 *myPos1, *myPos2; /* pointers to the head and tail long words of a music event */
563  NoteRequest *myNoteRequest;
564  NoteAllocator myNoteAllocator; /* for the NAStuffToneDescription call */
565  ComponentResult myErr = noErr;
566  int part;
567 
568  myHeader = NULL;
569  myNoteAllocator = NULL;
570 
571  /*
572  * Open up the Note Allocator
573  */
574  myNoteAllocator = OpenDefaultComponent(kNoteAllocatorComponentType,0);
575  if (myNoteAllocator == NULL)
576  goto bail;
577 
578  /*
579  * Allocate space for the tune header
580  */
581  myHeader = (Uint32 *)
582  NewPtrClear((numParts * kNoteRequestEventLength + kMarkerEventLength) * sizeof(Uint32));
583  if (myHeader == NULL)
584  goto bail;
585 
586  myPos1 = myHeader;
587 
588  /*
589  * Loop over all parts
590  */
591  for(part = 0; part < numParts; ++part)
592  {
593  /*
594  * Stuff request for the instrument with the given polyphony
595  */
596  myPos2 = myPos1 + (kNoteRequestEventLength - 1); /* last longword of general event */
597  qtma_StuffGeneralEvent(*myPos1, *myPos2, part, kGeneralEventNoteRequest, kNoteRequestEventLength);
598  myNoteRequest = (NoteRequest *)(myPos1 + 1);
599  myNoteRequest->info.flags = 0;
600  /* I'm told by the Apple people that the Quicktime types were poorly designed and it was
601  * too late to change them. On little endian, the BigEndian(Short|Fixed) types are structs
602  * while on big endian they are primitive types. Furthermore, Quicktime failed to
603  * provide setter and getter functions. To get this to work, we need to case the
604  * code for the two possible situations.
605  * My assumption is that the right-side value was always expected to be BigEndian
606  * as it was written way before the Universal Binary transition. So in the little endian
607  * case, OSSwap is used.
608  */
609 #if SDL_BYTEORDER == SDL_LIL_ENDIAN
610  myNoteRequest->info.polyphony.bigEndianValue = OSSwapHostToBigInt16(part_poly_max[part]);
611  myNoteRequest->info.typicalPolyphony.bigEndianValue = OSSwapHostToBigInt32(0x00010000);
612 #else
613  myNoteRequest->info.polyphony = part_poly_max[part];
614  myNoteRequest->info.typicalPolyphony = 0x00010000;
615 #endif
616  myErr = NAStuffToneDescription(myNoteAllocator,part_to_inst[part],&myNoteRequest->tone);
617  if (myErr != noErr)
618  goto bail;
619 
620  /* move pointer to beginning of next event */
621  myPos1 += kNoteRequestEventLength;
622  }
623 
624  *myPos1 = kEndMarkerValue; /* end of sequence marker */
625 
626 
627 bail:
628  if(myNoteAllocator)
629  CloseComponent(myNoteAllocator);
630 
631  /* if we encountered an error, dispose of the storage we allocated and return NULL */
632  if (myErr != noErr) {
633  DisposePtr((Ptr)myHeader);
634  myHeader = NULL;
635  }
636 
637  return myHeader;
638 }
