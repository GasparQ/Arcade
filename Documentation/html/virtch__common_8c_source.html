240 /* Fill the buffer with 'todo' bytes of silence (it depends on the mixing mode
241  how the buffer is filled) */
242 ULONG VC1_SilenceBytes(SBYTE* buf,ULONG todo)
243 {
244  todo=samples2bytes(bytes2samples(todo));
245 
246  /* clear the buffer to zero (16 bits signed) or 0x80 (8 bits unsigned) */
247  if(vc_mode & DMODE_16BITS)
248  memset(buf,0,todo);
249  else
250  memset(buf,0x80,todo);
251 
252  return todo;
253 }

257 /* Writes 'todo' mixed SBYTES (!!) to 'buf'. It returns the number of SBYTES
258  actually written to 'buf' (which is rounded to number of samples that fit
259  into 'todo' bytes). */
260 ULONG VC1_WriteBytes(SBYTE* buf,ULONG todo)
261 {
262  if(!vc_softchn)
263  return VC1_SilenceBytes(buf,todo);
264 
265  todo = bytes2samples(todo);
266  VC1_WriteSamples(buf,todo);
267 
268  return samples2bytes(todo);
269 }

357 SWORD VC1_SampleLoad(struct SAMPLOAD* sload,int type)
358 {
359  SAMPLE *s = sload->sample;
360  int handle;
361  ULONG t, length,loopstart,loopend;
362 
363  if(type==MD_HARDWARE) return -1;
364 
365  /* Find empty slot to put sample address in */
366  for(handle=0;handle<MAXSAMPLEHANDLES;handle++)
367  if(!Samples[handle]) break;
368 
369  if(handle==MAXSAMPLEHANDLES) {
370  _mm_errno = MMERR_OUT_OF_HANDLES;
371  return -1;
372  }
373 
374  /* Reality check for loop settings */
375  if (s->loopend > s->length)
376  s->loopend = s->length;
377  if (s->loopstart >= s->loopend)
378  s->flags &= ~SF_LOOP;
379 
380  length = s->length;
381  loopstart = s->loopstart;
382  loopend = s->loopend;
383 
384  SL_SampleSigned(sload);
385  SL_Sample8to16(sload);
386 
387  if(!(Samples[handle]=(SWORD*)_mm_malloc((length+20)<<1))) {
388  _mm_errno = MMERR_SAMPLE_TOO_BIG;
389  return -1;
390  }
391 
392  /* read sample into buffer */
393  if (SL_Load(Samples[handle],sload,length))
394  return -1;
395 
396  /* Unclick sample */
397  if(s->flags & SF_LOOP) {
398  if(s->flags & SF_BIDI)
399  for(t=0;t<16;t++)
400  Samples[handle][loopend+t]=Samples[handle][(loopend-t)-1];
401  else
402  for(t=0;t<16;t++)
403  Samples[handle][loopend+t]=Samples[handle][t+loopstart];
404  } else
405  for(t=0;t<16;t++)
406  Samples[handle][t+length]=0;
407 
408  return handle;
409 }

423 ULONG VC1_VoiceRealVolume(UBYTE voice)
424 {
425  ULONG i,s,size;
426  int k,j;
427  SWORD *smp;
428  SLONG t;
429 
430  t = vinf[voice].current>>FRACBITS;
431  if(!vinf[voice].active) return 0;
432 
433  s = vinf[voice].handle;
434  size = vinf[voice].size;
435 
436  i=64; t-=64; k=0; j=0;
437  if(i>size) i = size;
438  if(t<0) t = 0;
439  if(t+i > size) t = size-i;
440 
441  i &= ~1; /* make sure it's EVEN. */
442 
443  smp = &Samples[s][t];
444  for(;i;i--,smp++) {
445  if(k<*smp) k = *smp;
446  if(j>*smp) j = *smp;
447  }
448  return abs(k-j);
449 }
