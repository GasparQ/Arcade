241 /* New Note Action Scoring System :
242  --------------------------------
243  1) total-volume (fadevol, chanvol, volume) is the main scorer.
244  2) a looping sample is a bonus x2
245  3) a foreground channel is a bonus x4
246  4) an active envelope with keyoff is a handicap -x2

248 static int MP_FindEmptyChannel(MODULE *mod)
249 {
250  MP_VOICE *a;
251  ULONG t,k,tvol,pp;
252 
253  for (t=0;t<NUMVOICES(mod);t++)
254  if (((mod->voice[t].main.kick==KICK_ABSENT)||
255  (mod->voice[t].main.kick==KICK_ENV))&&
256  Voice_Stopped_internal(t))
257  return t;
258 
259  tvol=0xffffffUL;t=-1;a=mod->voice;
260  for (k=0;k<NUMVOICES(mod);k++,a++) {
261  /* allow us to take over a nonexisting sample */
262  if (!a->main.s)
263  return k;
264 
265  if ((a->main.kick==KICK_ABSENT)||(a->main.kick==KICK_ENV)) {
266  pp=a->totalvol<<((a->main.s->flags&SF_LOOP)?1:0);
267  if ((a->master)&&(a==a->master->slave))
268  pp<<=2;
269 
270  if (pp<tvol) {
271  tvol=pp;
272  t=k;
273  }
274  }
275  }
276 
277  if (tvol>8000*7) return -1;
278  return t;
279 }

380 /* This procedure processes all envelope types, include volume, pitch, and
381  panning. Envelopes are defined by a set of points, each with a magnitude
382  [relating either to volume, panning position, or pitch modifier] and a tick
383  position.
384 
385  Envelopes work in the following manner:
386 
387  (a) Each tick the envelope is moved a point further in its progression. For
388  an accurate progression, magnitudes between two envelope points are
389  interpolated.
390 
391  (b) When progression reaches a defined point on the envelope, values are
392  shifted to interpolate between this point and the next, and checks for
393  loops or envelope end are done.
394

396  Sustain loops are loops that are only active as long as the keyoff flag is
397  clear. When a volume envelope terminates, so does the current fadeout.
398 */

406  b = t->b;

410  * Sustain loop on one point (XM type).
411  * Not processed if KEYOFF.
412  * Don't move and don't interpolate when the point is reached
413  */
414  if ((t->flg & EF_SUSTAIN) && t->susbeg == t->susend &&
415  (!(aout->main.keyoff & KEY_OFF) && p == t->env[t->susbeg].pos)) {
416  v = t->env[t->susbeg].val;
417  } else {

419  * All following situations will require interpolation between
420  * two envelope points.
421  */
422 
423  /*
424  * Sustain loop between two points (IT type).
425  * Not processed if KEYOFF.
426  */
427  /* if we were on a loop point, loop now */
428  if ((t->flg & EF_SUSTAIN) && !(aout->main.keyoff & KEY_OFF) &&
429  a >= t->susend) {
430  a = t->susbeg;
431  b = (t->susbeg==t->susend)?a:a+1;
432  p = t->env[a].pos;
433  v = t->env[a].val;
434  } else
435  /*
436  * Regular loop.
437  * Be sure to correctly handle single point loops.
438  */
439  if ((t->flg & EF_LOOP) && a >= t->end) {
440  a = t->beg;
441  b = t->beg == t->end ? a : a + 1;
442  p = t->env[a].pos;
443  v = t->env[a].val;
444  } else
445  /*
446  * Non looping situations.
447  */
448  if (a != b)
449  v = InterpolateEnv(p, &t->env[a], &t->env[b]);
450  else
451  v = t->env[a].val;
452 
453  /*
454  * Start to fade if the volume envelope is finished.
455  */
456  if (p >= t->env[t->pts - 1].pos) {
457  if (t->flg & EF_VOLENV) {
458  aout->main.keyoff |= KEY_FADE;
459  if (!v)
460  aout->main.fadevol = 0;
461  }
462  } else {
463  p++;
464  /* did pointer reach point b? */
465  if (p >= t->env[b].pos)
466  a = b++; /* shift points a and b */
467  }
468  t->a = a;
469  t->b = b;
470  t->p = p;
471  }
472  }
473  return v;

893 static void DoEEffects(UWORD tick, UWORD flags, MP_CONTROL *a, MODULE *mod,
894  SWORD channel, UBYTE dat)
895 {
896  UBYTE nib = dat & 0xf;
897 
898  switch (dat>>4) {
899  case 0x0: /* hardware filter toggle, not supported */
900  break;
901  case 0x1: /* fineslide up */
902  if (a->main.period)
903  if (!tick)
904  a->tmpperiod-=(nib<<2);
905  break;
906  case 0x2: /* fineslide dn */
907  if (a->main.period)
908  if (!tick)
909  a->tmpperiod+=(nib<<2);
910  break;
911  case 0x3: /* glissando ctrl */
912  a->glissando=nib;
913  break;
914  case 0x4: /* set vibrato waveform */
915  a->wavecontrol&=0xf0;
916  a->wavecontrol|=nib;
917  break;
918  case 0x5: /* set finetune */
919  if (a->main.period) {
920  if (flags&UF_XMPERIODS)
921  a->speed=nib+128;
922  else
923  a->speed=finetune[nib];
924  a->tmpperiod=GetPeriod(flags, (UWORD)a->main.note<<1,a->speed);
925  }
926  break;
927  case 0x6: /* set patternloop */
928  if (tick)
929  break;
930  if (nib) { /* set reppos or repcnt ? */
931  /* set repcnt, so check if repcnt already is set, which means we
932  are already looping */
933  if (a->pat_repcnt)
934  a->pat_repcnt--; /* already looping, decrease counter */
935  else {
936 #if 0
937  /* this would make walker.xm, shipped with Xsoundtracker,
938  play correctly, but it's better to remain compatible
939  with FT2 */
940  if ((!(flags&UF_NOWRAP))||(a->pat_reppos!=POS_NONE))
941 #endif
942  a->pat_repcnt=nib; /* not yet looping, so set repcnt */
943  }
944 
945  if (a->pat_repcnt) { /* jump to reppos if repcnt>0 */
946  if (a->pat_reppos==POS_NONE)
947  a->pat_reppos=mod->patpos-1;
948  if (a->pat_reppos==-1) {
949  mod->pat_repcrazy=1;
950  mod->patpos=0;
951  } else
952  mod->patpos=a->pat_reppos;
953  } else a->pat_reppos=POS_NONE;
954  } else
955  a->pat_reppos=mod->patpos-1; /* set reppos - can be (-1) */
956  break;
957  case 0x7: /* set tremolo waveform */
958  a->wavecontrol&=0x0f;
959  a->wavecontrol|=nib<<4;
960  break;
961  case 0x8: /* set panning */
962  if (mod->panflag) {
963  if (nib<=8) nib<<=4;
964  else nib*=17;
965  a->main.panning=mod->panning[channel]=nib;
966  }
967  break;
968  case 0x9: /* retrig note */
969  /* do not retrigger on tick 0, until we are emulating FT2 and effect
970  data is zero */
971  if (!tick && !((flags & UF_FT2QUIRKS) && (!nib)))
972  break;
973  /* only retrigger if data nibble > 0, or if tick 0 (FT2 compat) */
974  if (nib || !tick) {
975  if (!a->retrig) {
976  /* when retrig counter reaches 0, reset counter and restart
977  the sample */
978  if (a->main.period) a->main.kick=KICK_NOTE;
979  a->retrig=nib;
980  }
981  a->retrig--; /* countdown */
982  }
983  break;
984  case 0xa: /* fine volume slide up */
985  if (tick)
986  break;
987  a->tmpvolume+=nib;
988  if (a->tmpvolume>64) a->tmpvolume=64;
989  break;
990  case 0xb: /* fine volume slide dn */
991  if (tick)
992  break;
993  a->tmpvolume-=nib;
994  if (a->tmpvolume<0)a->tmpvolume=0;
995  break;
996  case 0xc: /* cut note */
997  /* When tick reaches the cut-note value, turn the volume to
998  zero (just like on the amiga) */
999  if (tick>=nib)
1000  a->tmpvolume=0; /* just turn the volume down */
1001  break;
1002  case 0xd: /* note delay */
1003  /* delay the start of the sample until tick==nib */
1004  if (!tick)
1005  a->main.notedelay=nib;
1006  else if (a->main.notedelay)
1007  a->main.notedelay--;
1008  break;
1009  case 0xe: /* pattern delay */
1010  if (!tick)
1011  if (!mod->patdly2)
1012  mod->patdly=nib+1; /* only once, when tick=0 */
1013  break;
1014  case 0xf: /* invert loop, not supported */
1015  break;
1016  }
1017 }
