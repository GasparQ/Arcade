35 static inline int sanityCheckByteorder(void)
36 {
37  int sane = 0;
38  union {
39  Uint32 ui32;
40  Uint8 ui8[4];
41  } swapper;
42 
43  swapper.ui32 = 0x00000001;
44 
45 #if SDL_BYTEORDER == SDL_LIL_ENDIAN
46  sane = (swapper.ui8[0] == 0x01);
47 #else
48  sane = (swapper.ui8[3] == 0x01);
49 #endif
50 
51  if (!sane) {
52  fprintf(stderr,
53  "\n\n"
54  "*************************************************************\n"
55  " SMPEG ERROR: SDL is wrong about this platform's byte order!\n"
56  " You need to fix your SDL install before SMPEG can work!\n"
57  "*************************************************************\n"
58  "\n\n");
59  }
60 
61  return(sane);
62 }

65 /* Create a new SMPEG object from an MPEG file.
66  On return, if 'info' is not NULL, it will be filled with information
67  about the MPEG object.
68  This function returns a new SMPEG object. Use SMPEG_error() to find out
69  whether or not there was a problem building the MPEG stream.
70  The sdl_audio parameter indicates if SMPEG should initialize the SDL audio
71  subsystem. If not, you will have to use the SMPEG_playaudio() function below
72  to extract the decoded data.
73  */
74 SMPEG* SMPEG_new(const char *file, SMPEG_Info* info, int sdl_audio)
75 {
76  SMPEG *mpeg;
77 
78  if (!sanityCheckByteorder())
79  return(NULL);
80 
81  /* Create a new SMPEG object! */
82  mpeg = new SMPEG;
83  mpeg->obj = new MPEG(file, sdl_audio ? true : false);
84 
85  /* Find out the details of the stream, if requested */
86  SMPEG_getinfo(mpeg, info);
87 
88  /* We're done! */
89  return(mpeg);
90 }

152 /* Get current information about an SMPEG object */
153 void SMPEG_getinfo( SMPEG* mpeg, SMPEG_Info* info )
154 {
155  if ( info ) {
156  MPEG_AudioInfo ainfo;
157  MPEG_VideoInfo vinfo;
158  MPEG_SystemInfo sinfo;
159 
160  memset(info, 0, (sizeof *info));
161  if ( mpeg->obj ) {
162  info->has_audio = (mpeg->obj->audiostream != NULL);
163  if ( info->has_audio ) {
164  mpeg->obj->GetAudioInfo(&ainfo);
165  info->audio_current_frame = ainfo.current_frame;
166  sprintf(info->audio_string,
167  "MPEG-%d Layer %d %dkbit/s %dHz %s",
168  ainfo.mpegversion+1,
169  ainfo.layer,
170  ainfo.bitrate,
171  ainfo.frequency,
172  (ainfo.mode == 3) ? "mono" : "stereo");
173  }
174  info->has_video = (mpeg->obj->videostream != NULL);
175  if ( info->has_video ) {
176  mpeg->obj->GetVideoInfo(&vinfo);
177  info->width = vinfo.width;
178  info->height = vinfo.height;
179  info->current_frame = vinfo.current_frame;
180  info->current_fps = vinfo.current_fps;
181  }
182  if(mpeg->obj->system != NULL)
183  {
184  mpeg->obj->GetSystemInfo(&sinfo);
185  info->total_size = sinfo.total_size;
186  info->current_offset = sinfo.current_offset;
187  info->total_time = sinfo.total_time;
188  info->current_time = sinfo.current_time;
189  }
190  else
191  {
192  info->total_size = 0;
193  info->current_offset = 0;
194  }
195  }
196  }
197 }
