104 /* Load a modplug stream from an SDL_RWops object */
105 modplug_data *modplug_new_RW(SDL_RWops *src, int freesrc)
106 {
107  modplug_data *music = NULL;
108  Sint64 offset;
109  size_t sz;
110  char *buf;
111 
112  /* Make sure the modplug library is loaded */
113  if ( !Mix_Init(MIX_INIT_MODPLUG) ) {
114  return NULL;
115  }
116 
117  offset = SDL_RWtell(src);
118  SDL_RWseek(src, 0, RW_SEEK_END);
119  sz = (size_t)(SDL_RWtell(src) - offset);
120  SDL_RWseek(src, offset, RW_SEEK_SET);
121  buf = (char*)SDL_malloc(sz);
122  if (buf) {
123  if (SDL_RWread(src, buf, sz, 1) == 1) {
124  music = (modplug_data*)SDL_malloc(sizeof(modplug_data));
125  if (music) {
126  music->playing = 0;
127  music->file = modplug.ModPlug_Load(buf, sz);
128  if (!music->file) {
129  SDL_free(music);
130  music = NULL;
131  }
132  } else {
133  SDL_OutOfMemory();
134  }
135  }
136  SDL_free(buf);
137  } else {
138  SDL_OutOfMemory();
139  }
140  if (music && freesrc) {
141  SDL_RWclose(src);
142  }
143  return music;
144 }

159 /* Play some of a stream previously started with modplug_play() */
160 int modplug_playAudio(modplug_data *music, Uint8 *stream, int len)
161 {
162  if (current_output_channels > 2) {
163  int small_len = 2 * len / current_output_channels;
164  int i;
165  Uint8 *src, *dst;
166 
167  i=modplug.ModPlug_Read(music->file, stream, small_len);
168  if(i<small_len)
169  {
170  SDL_memset(stream+i,0,small_len-i);
171  music->playing=0;
172  }
173  /* and extend to len by copying channels */
174  src = stream + small_len;
175  dst = stream + len;
176 
177  switch (settings.mBits) {
178  case 8:
179  for ( i=small_len/2; i; --i ) {
180  src -= 2;
181  dst -= current_output_channels;
182  dst[0] = src[0];
183  dst[1] = src[1];
184  dst[2] = src[0];
185  dst[3] = src[1];
186  if (current_output_channels == 6) {
187  dst[4] = src[0];
188  dst[5] = src[1];
189  }
190  }
191  break;
192  case 16:
193  for ( i=small_len/4; i; --i ) {
194  src -= 4;
195  dst -= 2 * current_output_channels;
196  dst[0] = src[0];
197  dst[1] = src[1];
198  dst[2] = src[2];
199  dst[3] = src[3];
200  dst[4] = src[0];
201  dst[5] = src[1];
202  dst[6] = src[2];
203  dst[7] = src[3];
204  if (current_output_channels == 6) {
205  dst[8] = src[0];
206  dst[9] = src[1];
207  dst[10] = src[2];
208  dst[11] = src[3];
209  }
210  }
211  break;
212  }
213  } else {
214  int i=modplug.ModPlug_Read(music->file, stream, len);
215  if(i<len)
216  {
217  SDL_memset(stream+i,0,len-i);
218  music->playing=0;
219  }
220  }
221  if ( music_swap8 ) {
222  Uint8 *dst;
223  int i;
224 
225  dst = stream;
226  for ( i=len; i; --i ) {
227  *dst++ ^= 0x80;
228  }
229  } else
230  if ( music_swap16 ) {
231  Uint8 *dst, tmp;
232  int i;
233 
234  dst = stream;
235  for ( i=(len/2); i; --i ) {
236  tmp = dst[0];
237  dst[0] = dst[1];
238  dst[1] = tmp;
239  dst += 2;
240  }
241  }
242  return 0;
243 }
