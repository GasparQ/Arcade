143 int MPEGvideo::timeSync( VidStream* vid_stream )
144 {
145  static double correction = -1;
146 
147  /* Update the number of frames displayed */
148  vid_stream->totNumFrames++;
149  vid_stream->current_frame++;
150 
151  /* Do we need to initialize framerate? */
152  if ( vid_stream->rate_deal < 0 ) {
153  switch( framerate ) {
154  case -1: /* Go with stream Value */
155  vid_stream->rate_deal = VidRateNum[ vid_stream->picture_rate ];
156  break;
157 
158  case 0: /* as fast as possible */
159  vid_stream->rate_deal = 0;
160  break;
161 
162  default:
163  vid_stream->rate_deal = framerate;
164  break;
165  }
166  if ( vid_stream->rate_deal ) {
167  vid_stream->_oneFrameTime = 1.0 / vid_stream->rate_deal;
168  }
169  }
170 
171  /* Update the current play time */
172  play_time += vid_stream->_oneFrameTime;
173 
174  /* Synchronize using system timestamps */
175  if(vid_stream->current && vid_stream->current->show_time > 0){
176 #ifdef DEBUG_TIMESTAMP_SYNC
177  fprintf(stderr, "video: time:%.3f shift:%.3f\r",
178  play_time,
179  play_time - vid_stream->current->show_time);
180 #endif
181  if(correction == -1)
182 #ifdef STRANGE_SYNC_TEST
183  /* this forces us to maintain the offset we have at the begining
184  all the time, and is only usefull for testing */
185  correction = play_time - vid_stream->current->show_time;
186 #else
187  correction = 0;
188 #endif
189 #ifdef USE_TIMESTAMP_SYNC
190  play_time = vid_stream->current->show_time + correction ;
191 #endif
192  vid_stream->current->show_time = -1;
193  }
194 
195  /* If we are looking for a particular frame... */
196  if( vid_stream->_jumpFrame > -1 )
197  {
198  if ( vid_stream->totNumFrames != vid_stream->_jumpFrame ) {
199  vid_stream->_skipFrame = 1;
200  } else {
201  vid_stream->_skipFrame = 0;
202  }
203  return vid_stream->_skipFrame;
204  }
205 
206  /* If we're already behind, don't check timing */
207  if ( vid_stream->_skipFrame > 0 )
208  {
209  return --vid_stream->_skipFrame;
210  }
211 
212  /* See if we need to skip frames, based on timing */
213  if ( vid_stream->rate_deal ) {
214  static const double TIMESLICE = 0.01; // Seconds per OS timeslice
215  double time_behind;
216 
217  /* Calculate the frame time relative to real time */
218  time_behind = CurrentTime(vid_stream) - Time();
219 
220 #ifdef DEBUG_MPEG_SCHEDULING
221 printf("Frame %d: frame time: %f, real time: %f, time behind: %f\n", vid_stream->totNumFrames, Time(), CurrentTime(vid_stream), time_behind);
222 #endif
223 
224  /* Allow up to MAX_FUDGE_TIME of delay in output */
225  if ( time_behind < -TIMESLICE ) {
226  time_behind = -time_behind;
227  vid_stream->_skipCount = 0;
228 #ifdef DEBUG_MPEG_SCHEDULING
229 printf("Ahead! Sleeping %f\n", time_behind-TIMESLICE);
230 #endif
231  SDL_Delay((Uint32)((time_behind-TIMESLICE)*1000));
232  } else
233  if ( time_behind < FUDGE_TIME ) {
234  if ( vid_stream->_skipCount > 0 ) {
235  vid_stream->_skipCount /= 2;
236  }
237 #ifdef DEBUG_MPEG_SCHEDULING
238 printf("Just right.\n");
239 #endif
240  } else
241  if ( time_behind < MAX_FUDGE_TIME ) {
242  if ( vid_stream->_skipCount > 0 ) {
243  vid_stream->_skipCount--;
244  }
245  vid_stream->_skipFrame = 1+(int)(vid_stream->_skipCount/2);
246 #ifdef DEBUG_MPEG_SCHEDULING
247 printf("A little behind, skipping %d frames\n", vid_stream->_skipFrame);
248 #endif
249  } else {
250  /* time_behind >= MAX_FUDGE_TIME */
251  if ( (time_behind > (MAX_FUDGE_TIME*2)) &&
252  (vid_stream->_skipCount == MAX_FRAME_SKIP) ) {
253 #ifdef DEBUG_MPEG_SCHEDULING
254 printf("Way too far behind, losing time sync...\n");
255 #endif
256 #if 0 // This results in smoother video, but sync's terribly on slow machines
257  play_time = CurrentTime(vid_stream) - (MAX_FUDGE_TIME*2);
258 #endif
259  }
260 #ifdef SLOW_START_SCHEDULING
261  vid_stream->_skipCount += SLOW_START_INCREMENT;
262 #else
263  vid_stream->_skipCount += 1.0;
264 #endif
265  if( vid_stream->_skipCount > MAX_FRAME_SKIP ) {
266  vid_stream->_skipCount = MAX_FRAME_SKIP;
267  }
268  vid_stream->_skipFrame = (int)(vid_stream->_skipCount+0.9);
269 #ifdef DEBUG_MPEG_SCHEDULING
270 printf("A lot behind, skipping %d frames\n", vid_stream->_skipFrame);
271 #endif
272  }
273  }
274  return(vid_stream->_skipFrame);
275 }
