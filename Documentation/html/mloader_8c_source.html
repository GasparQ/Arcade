123 BOOL ReadLinedComment(UWORD len,UWORD linelen)
124 {
125  CHAR *tempcomment,*line,*storage;
126  UWORD total=0,t,lines;
127  int i;
128 
129  lines = (len + linelen - 1) / linelen;
130  if (len) {
131  if(!(tempcomment=(CHAR*)_mm_malloc(len+1))) return 0;
132  if(!(storage=(CHAR*)_mm_malloc(linelen+1))) {
133  free(tempcomment);
134  return 0;
135  }
136  memset(tempcomment, ' ', len);
137  _mm_read_UBYTES(tempcomment,len,modreader);
138 
139  /* compute message length */
140  for(line=tempcomment,total=t=0;t<lines;t++,line+=linelen) {
141  for(i=linelen;(i>=0)&&(line[i]==' ');i--) line[i]=0;
142  for(i=0;i<linelen;i++) if (!line[i]) break;
143  total+=1+i;
144  }
145 
146  if(total>lines) {
147  if(!(of.comment=(CHAR*)_mm_malloc(total+1))) {
148  free(storage);
149  free(tempcomment);
150  return 0;
151  }
152 
153  /* convert message */
154  for(line=tempcomment,t=0;t<lines;t++,line+=linelen) {
155  for(i=0;i<linelen;i++) if(!(storage[i]=line[i])) break;
156  storage[i]=0; /* if (i==linelen) */
157  strcat(of.comment,storage);strcat(of.comment,"\r");
158  }
159  free(storage);
160  free(tempcomment);
161  }
162  }
163  return 1;
164 }

259 /* Creates a CSTR out of a character buffer of 'len' bytes, but strips any
260  terminating non-printing characters like 0, spaces etc. */
261 CHAR *DupStr(CHAR* s,UWORD len,BOOL strict)
262 {
263  UWORD t;
264  CHAR *d=NULL;
265 
266  /* Scan for last printing char in buffer [includes high ascii up to 254] */
267  while(len) {
268  if(s[len-1]>0x20) break;
269  len--;
270  }
271 
272  /* Scan forward for possible NULL character */
273  if(strict) {
274  for(t=0;t<len;t++) if (!s[t]) break;
275  if (t<len) len=t;
276  }
277 
278  /* When the buffer wasn't completely empty, allocate a cstring and copy the
279  buffer into that string, except for any control-chars */
280  if((d=(CHAR*)_mm_malloc(sizeof(CHAR)*(len+1)))) {
281  for(t=0;t<len;t++) d[t]=(s[t]<32)?'.':s[t];
282  d[len]=0;
283  }
284  return d;
285 }

409 MODULE* Player_LoadGeneric_internal(MREADER *reader,int maxchan,BOOL curious)
410 {
411  int t;
412  MLOADER *l;
413  BOOL ok;
414  MODULE *mf;
415 
416  modreader = reader;
417  _mm_errno = 0;
418  _mm_critical = 0;
419  _mm_iobase_setcur(modreader);
420 
421  /* Try to find a loader that recognizes the module */
422  for(l=firstloader;l;l=l->next) {
423  _mm_rewind(modreader);
424  if(l->Test()) break;
425  }
426 
427  if(!l) {
428  _mm_errno = MMERR_NOT_A_MODULE;
429  if(_mm_errorhandler) _mm_errorhandler();
430  _mm_rewind(modreader);_mm_iobase_revert();
431  return NULL;
432  }
433 
434  /* init unitrk routines */
435  if(!UniInit()) {
436  if(_mm_errorhandler) _mm_errorhandler();
437  _mm_rewind(modreader);_mm_iobase_revert();
438  return NULL;
439  }
440 
441  /* init the module structure with vanilla settings */
442  memset(&of,0,sizeof(MODULE));
443  of.bpmlimit = 33;
444  of.initvolume = 128;
445  for (t = 0; t < UF_MAXCHAN; t++) of.chanvol[t] = 64;
446  for (t = 0; t < UF_MAXCHAN; t++)
447  of.panning[t] = ((t + 1) & 2) ? PAN_RIGHT : PAN_LEFT;
448 
449  /* init module loader and load the header / patterns */
450  if (!l->Init || l->Init()) {
451  _mm_rewind(modreader);
452  ok = l->Load(curious);
453  if (ok) {
454  /* propagate inflags=flags for in-module samples */
455  for (t = 0; t < of.numsmp; t++)
456  if (of.samples[t].inflags == 0)
457  of.samples[t].inflags = of.samples[t].flags;
458  }
459  } else
460  ok = 0;
461 
462  /* free loader and unitrk allocations */
463  if (l->Cleanup) l->Cleanup();
464  UniCleanup();
465 
466  if(!ok) {
467  ML_FreeEx(&of);
468  if(_mm_errorhandler) _mm_errorhandler();
469  _mm_rewind(modreader);_mm_iobase_revert();
470  return NULL;
471  }
472 
473  if(!ML_LoadSamples()) {
474  ML_FreeEx(&of);
475  if(_mm_errorhandler) _mm_errorhandler();
476  _mm_rewind(modreader);_mm_iobase_revert();
477  return NULL;
478  }
479 
480  if(!(mf=ML_AllocUniMod())) {
481  ML_FreeEx(&of);
482  _mm_rewind(modreader);_mm_iobase_revert();
483  if(_mm_errorhandler) _mm_errorhandler();
484  return NULL;
485  }
486 
487  /* If the module doesn't have any specific panning, create a
488  MOD-like panning, with the channels half-separated. */
489  if (!(of.flags & UF_PANNING))
490  for (t = 0; t < of.numchn; t++)
491  of.panning[t] = ((t + 1) & 2) ? PAN_HALFRIGHT : PAN_HALFLEFT;
492 
493  /* Copy the static MODULE contents into the dynamic MODULE struct. */
494  memcpy(mf,&of,sizeof(MODULE));
495 
496  if(maxchan>0) {
497  if(!(mf->flags&UF_NNA)&&(mf->numchn<maxchan))
498  maxchan = mf->numchn;
499  else
500  if((mf->numvoices)&&(mf->numvoices<maxchan))
501  maxchan = mf->numvoices;
502 
503  if(maxchan<mf->numchn) mf->flags |= UF_NNA;
504 
505  if(MikMod_SetNumVoices_internal(maxchan,-1)) {
506  _mm_iobase_revert();
507  Player_Free(mf);
508  return NULL;
509  }
510  }
511  if(SL_LoadSamples()) {
512  _mm_iobase_revert();
513  Player_Free_internal(mf);
514  return NULL;
515  }
516  if(Player_Init(mf)) {
517  _mm_iobase_revert();
518  Player_Free_internal(mf);
519  mf=NULL;
520  }
521  _mm_iobase_revert();
522  return mf;
523 }
