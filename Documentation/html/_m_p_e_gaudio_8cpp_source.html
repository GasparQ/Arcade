50  /* Analyze the MPEG audio stream */
51  if ( loadheader() ) {
52  SDL_AudioSpec wanted;
53  WantedSpec(&wanted);
54

82  }

126 void
127 MPEGaudio:: ActualSpec(const SDL_AudioSpec *actual)
128 {
129  /* Splay can optimize some of the conversion */
130  if ( actual->channels == 1 && outputstereo ) {
131  forcetomonoflag = true;
132  }
133  if ( actual->channels == 2 && !outputstereo ) {
134  forcetostereoflag = true;
135  samplesperframe *= 2;
136  }
137  /* FIXME: Create an audio conversion block */
138  if ( (actual->freq/100) == ((frequencies[version][frequency]/2)/100) ) {
139  downfrequency = 1;
140  } else if ( actual->freq != frequencies[version][frequency] ) {
141 #ifdef VERBOSE_WARNINGS
142  fprintf(stderr, "Warning: wrong audio frequency (wanted %d, got %d)\n",
143  frequencies[version][frequency], actual->freq);
144 #else
145  ;
146 #endif
147  }
148  if ( actual->format != AUDIO_S16SYS )
149  {
150  if ( (actual->format^0x1000) == AUDIO_S16SYS ) {
151  swapendianflag = true;
152  } else {
153  fprintf(stderr, "Warning: incorrect audio format\n");
154  }
155  }
156  rate_in_s=((double)((actual->format&0xFF)/8)*actual->channels*actual->freq);
157  stereo=((actual->channels-1) > 0);
158 }
