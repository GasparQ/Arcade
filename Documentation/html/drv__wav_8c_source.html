54 static void putheader(void)
55 {
56  _mm_fseek(wavout,0,SEEK_SET);
57  _mm_write_string("RIFF",wavout);
58  _mm_write_I_ULONG(dumpsize+44,wavout);
59  _mm_write_string("WAVEfmt ",wavout);
60  _mm_write_I_ULONG(16,wavout); /* length of this RIFF block crap */
61 
62  _mm_write_I_UWORD(1, wavout); /* microsoft format type */
63  _mm_write_I_UWORD((md_mode&DMODE_STEREO)?2:1,wavout);
64  _mm_write_I_ULONG(md_mixfreq,wavout);
65  _mm_write_I_ULONG(md_mixfreq*((md_mode&DMODE_STEREO)?2:1)*
66  ((md_mode&DMODE_16BITS)?2:1),wavout);
67  /* block alignment (8/16 bit) */
68  _mm_write_I_UWORD(((md_mode&DMODE_16BITS)?2:1)*
69  ((md_mode&DMODE_STEREO)?2:1),wavout);
70  _mm_write_I_UWORD((md_mode&DMODE_16BITS)?16:8,wavout);
71  _mm_write_string("data",wavout);
72  _mm_write_I_ULONG(dumpsize,wavout);
73 }

90 static BOOL WAV_Init(void)
91 {
92 #if defined unix || (defined __APPLE__ && defined __MACH__)
93  if (!MD_Access(filename?filename:FILENAME)) {
94  _mm_errno=MMERR_OPENING_FILE;
95  return 1;
96  }
97 #endif
98 
99  if(!(wavfile=fopen(filename?filename:FILENAME,"wb"))) {
100  _mm_errno=MMERR_OPENING_FILE;
101  return 1;
102  }
103  if(!(wavout=_mm_new_file_writer (wavfile))) {
104  fclose(wavfile);unlink(filename?filename:FILENAME);
105  wavfile=NULL;
106  return 1;
107  }
108  if(!(audiobuffer=(SBYTE*)_mm_malloc(BUFFERSIZE))) {
109  _mm_delete_file_writer(wavout);
110  fclose(wavfile);unlink(filename?filename:FILENAME);
111  wavfile=NULL;wavout=NULL;
112  return 1;
113  }
114 
115  md_mode|=DMODE_SOFT_MUSIC|DMODE_SOFT_SNDFX;
116 
117  if (VC_Init()) {
118  _mm_delete_file_writer(wavout);
119  fclose(wavfile);unlink(filename?filename:FILENAME);
120  wavfile=NULL;wavout=NULL;
121  return 1;
122  }
123  dumpsize=0;
124  putheader();
125 
126  return 0;
127 }
