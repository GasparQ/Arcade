126 MPEGstream:: next_packet(bool recurse, bool update_timestamp)
127 {
128  SDL_mutexP(mutex);
129 
130  /* Unlock current buffer */
131  br->Unlock();
132 
133  /* Check for the end of stream mark */
134  next_system_buffer();
135  if(eof())
136  {
137  /* Report eof */
138  SDL_mutexV(mutex);
139  return(false);
140  }
141 
142  /* Lock the buffer */
143  br->Lock();
144 
145  /* Make sure that we have read buffers in advance if possible */
146  if(preread_size < MAX_QUEUE)
147  system->RequestBuffer();
148 
149  /* Update stream datas */
150  data = (Uint8 *) br->Buffer();
151  stop = data + br->Size();
152  if(update_timestamp){
153  timestamp = br->TimeStamp;
154  timestamp_pos = pos;
155  }
156  SDL_mutexV(mutex);
157 
158  return(true);

227 Uint32
228 MPEGstream:: copy_data(Uint8 *area, Sint32 size, bool short_read)
229 {
230  Uint32 copied = 0;
231  bool timestamped = false;
232 
233  while ( (size > 0) && !eof()) {
234  Uint32 len;
235 
236  /* Get new data if necessary */
237  if ( data >= stop ) {
238  /* try to use the timestamp of the first packet */
239  if ( ! next_packet(true, (timestamp == -1) || !timestamped) ) {
240  break;
241  }
242  timestamped = true;
243  }
244 
245  SDL_mutexP(mutex);
246 
247  /* Copy as much as we need */
248  if ( size <= (Sint32)(stop-data) ) {
249  len = size;
250  } else {
251  len = (stop-data);
252  }
253 
254  memcpy(area, data, len);
255 
256  area += len;
257  data += len;
258  size -= len;
259  copied += len;
260  pos += len;
261 
262  SDL_mutexV(mutex);
263 
264  /* Allow 32-bit aligned short reads? */
265  if ( ((copied%4) == 0) && short_read ) {
266  break;
267  }
268  }
269 
270  return(copied);
271 }
