269 static void MedConvert(MODCOMMAND *p, const MMD0SONGHEADER *pmsh)
270 //---------------------------------------------------------------
271 {
272  UINT command = p->command;
273  UINT param = p->param;
274  switch(command)
275  {
276  case 0x00: if (param) command = CMD_ARPEGGIO; else command = 0; break;
277  case 0x01: command = CMD_PORTAMENTOUP; break;
278  case 0x02: command = CMD_PORTAMENTODOWN; break;
279  case 0x03: command = CMD_TONEPORTAMENTO; break;
280  case 0x04: command = CMD_VIBRATO; break;
281  case 0x05: command = CMD_TONEPORTAVOL; break;
282  case 0x06: command = CMD_VIBRATOVOL; break;
283  case 0x07: command = CMD_TREMOLO; break;
284  case 0x0A: if (param & 0xF0) param &= 0xF0; command = CMD_VOLUMESLIDE; if (!param) command = 0; break;
285  case 0x0B: command = CMD_POSITIONJUMP; break;
286  case 0x0C: command = CMD_VOLUME;
287  if (pmsh->flags & MMD_FLAG_VOLHEX)
288  {
289  if (param < 0x80)
290  {
291  param = (param+1) / 2;
292  } else command = 0;
293  } else
294  {
295  if (param <= 0x99)
296  {
297  param = (param >> 4)*10+((param & 0x0F) % 10);
298  if (param > 64) param = 64;
299  } else command = 0;
300  }
301  break;
302  case 0x09: command = (param < 0x20) ? CMD_SPEED : CMD_TEMPO; break;
303  case 0x0D: if (param & 0xF0) param &= 0xF0; command = CMD_VOLUMESLIDE; if (!param) command = 0; break;
304  case 0x0F: // Set Tempo / Special
305  // F.00 = Pattern Break
306  if (!param) command = CMD_PATTERNBREAK; else
307  // F.01 - F.F0: Set tempo/speed
308  if (param <= 0xF0)
309  {
310  if (pmsh->flags & MMD_FLAG_8CHANNEL)
311  {
312  param = (param > 10) ? 99 : bpmvals[param-1];
313  } else
314  // F.01 - F.0A: Set Speed
315  if (param <= 0x0A)
316  {
317  command = CMD_SPEED;
318  } else
319  // Old tempo
320  if (!(pmsh->flags2 & MMD_FLAG2_BPM))
321  {
322  param = _muldiv(param, 5*715909, 2*474326);
323  }
324  // F.0B - F.F0: Set Tempo (assumes LPB=4)
325  if (param > 0x0A)
326  {
327  command = CMD_TEMPO;
328  if (param < 0x21) param = 0x21;
329  if (param > 240) param = 240;
330  }
331  } else
332  switch(param)
333  {
334  // F.F1: Retrig 2x
335  case 0xF1:
336  command = CMD_MODCMDEX;
337  param = 0x93;
338  break;
339  // F.F2: Note Delay 2x
340  case 0xF2:
341  command = CMD_MODCMDEX;
342  param = 0xD3;
343  break;
344  // F.F3: Retrig 3x
345  case 0xF3:
346  command = CMD_MODCMDEX;
347  param = 0x92;
348  break;
349  // F.F4: Note Delay 1/3
350  case 0xF4:
351  command = CMD_MODCMDEX;
352  param = 0xD2;
353  break;
354  // F.F5: Note Delay 2/3
355  case 0xF5:
356  command = CMD_MODCMDEX;
357  param = 0xD4;
358  break;
359  // F.F8: Filter Off
360  case 0xF8:
361  command = CMD_MODCMDEX;
362  param = 0x00;
363  break;
364  // F.F9: Filter On
365  case 0xF9:
366  command = CMD_MODCMDEX;
367  param = 0x01;
368  break;
369  // F.FD: Very fast tone-portamento
370  case 0xFD:
371  command = CMD_TONEPORTAMENTO;
372  param = 0xFF;
373  break;
374  // F.FE: End Song
375  case 0xFE:
376  command = CMD_SPEED;
377  param = 0;
378  break;
379  // F.FF: Note Cut
380  case 0xFF:
381  command = CMD_MODCMDEX;
382  param = 0xC0;
383  break;
384  default:
385 #ifdef MED_LOG
386  Log("Unknown Fxx command: cmd=0x%02X param=0x%02X\n", command, param);
387 #endif
388  param = command = 0;
389  }
390  break;
391  // 11.0x: Fine Slide Up
392  case 0x11:
393  command = CMD_MODCMDEX;
394  if (param > 0x0F) param = 0x0F;
395  param |= 0x10;
396  break;
397  // 12.0x: Fine Slide Down
398  case 0x12:
399  command = CMD_MODCMDEX;
400  if (param > 0x0F) param = 0x0F;
401  param |= 0x20;
402  break;
403  // 14.xx: Vibrato
404  case 0x14:
405  command = CMD_VIBRATO;
406  break;
407  // 15.xx: FineTune
408  case 0x15:
409  command = CMD_MODCMDEX;
410  param &= 0x0F;
411  param |= 0x50;
412  break;
413  // 16.xx: Pattern Loop
414  case 0x16:
415  command = CMD_MODCMDEX;
416  if (param > 0x0F) param = 0x0F;
417  param |= 0x60;
418  break;
419  // 18.xx: Note Cut
420  case 0x18:
421  command = CMD_MODCMDEX;
422  if (param > 0x0F) param = 0x0F;
423  param |= 0xC0;
424  break;
425  // 19.xx: Sample Offset
426  case 0x19:
427  command = CMD_OFFSET;
428  break;
429  // 1A.0x: Fine Volume Up
430  case 0x1A:
431  command = CMD_MODCMDEX;
432  if (param > 0x0F) param = 0x0F;
433  param |= 0xA0;
434  break;
435  // 1B.0x: Fine Volume Down
436  case 0x1B:
437  command = CMD_MODCMDEX;
438  if (param > 0x0F) param = 0x0F;
439  param |= 0xB0;
440  break;
441  // 1D.xx: Pattern Break
442  case 0x1D:
443  command = CMD_PATTERNBREAK;
444  break;
445  // 1E.0x: Pattern Delay
446  case 0x1E:
447  command = CMD_MODCMDEX;
448  if (param > 0x0F) param = 0x0F;
449  param |= 0xE0;
450  break;
451  // 1F.xy: Retrig
452  case 0x1F:
453  command = CMD_RETRIG;
454  param &= 0x0F;
455  break;
456  // 2E.xx: set panning
457  case 0x2E:
458  command = CMD_MODCMDEX;
459  param = ((param + 0x10) & 0xFF) >> 1;
460  if (param > 0x0F) param = 0x0F;
461  param |= 0x80;
462  break;
463  default:
464 #ifdef MED_LOG
465  // 0x2E ?
466  Log("Unknown command: cmd=0x%02X param=0x%02X\n", command, param);
467 #endif
468  command = param = 0;
469  }
470  p->command = command;
471  p->param = param;
472 }
