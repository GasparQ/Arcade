240  DrawCube(gc->getPos(), gc->getColor(), 0.25);

264  }

266  {
267  if (m_render_mode == PERSPECTIVE)
268  {
269  Set2DMode();
270  }
271  const UIComponent *const *var = hsc->getComponentsToDisplay();
272  for (int i = 0; var && var[i]; i++)
273  {
274  DrawText(var[i]->getPos(), var[i]->getText(), var[i]->getColor());
275  }
276  }
277  else if ((ac = dynamic_cast<AnimationComponent *>(stack.top())) != nullptr)
278  {
280  if (m_render_mode == ORTHOGRAPHIC)
281  {
282  Set3DMode();
283  }
284  DrawBackgroundMenu();
285  }
286  stack.pop();
287  }
288  RefreshImage();
289 }
