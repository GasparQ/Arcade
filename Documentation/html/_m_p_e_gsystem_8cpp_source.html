634  if(Match4(pointer, END_CODE, END_MASK) ||
635  Match4(pointer, END2_CODE, END2_MASK))
636  {
637  /* End codes belong to video stream */
638 #ifdef DEBUG_SYSTEM
639  fprintf(stderr, "[%d] MPEG end code\n", read_total - read_size + (pointer - read_buffer));
640 #endif
641  stream_id = exist_stream(VIDEO_STREAMID, 0xF0);
642  packet_size = 4;
643  }
644  else
645  {
646  stream_id = stream_list[0]->streamid;
647 
648  if(!stream_list[1])
649  {
650  //Uint8 * packet_end;
651 
652  packet_size = 0;
653 
654  /* There is no system info in the stream */
655 
656  /* If we're still a system stream, morph to an audio */
657  /* or video stream */
658 
659  /* Sequence header -> gives framerate */
660  while((header_size = sequence_header(pointer+packet_size, read_buffer + read_size - pointer - packet_size, &frametime)) != 0)
661  {
662  stream_id = VIDEO_STREAMID;
663  stream_list[0]->streamid = stream_id;
664  packet_size += header_size;
665 #ifdef DEBUG_SYSTEM
666  fprintf(stderr, "MPEG sequence header frametime: %lf\n", frametime);
667 #endif
668  }
669 
670  /* GOP header */
671  while((header_size = gop_header(pointer+packet_size, read_buffer + read_size - pointer - packet_size, 0)) != 0)
672  {
673  packet_size += header_size;
674 #ifdef DEBUG_SYSTEM
675  fprintf(stderr, "MPEG gop header\n");
676 #endif
677  }
678 
679  /* Picture header */
680  while((header_size = picture_header(pointer+packet_size, read_buffer + read_size - pointer - packet_size)) != 0)
681  {
682  packet_size += header_size; // Warning: header size not quite correct (can be not byte aligned)
683  stream_timestamp += frametime; // but this is compensated by skipping a little more, as we don't need to be header aligned
684  packet_size += 4; // after this, since we then check for the next header to know the slice size.
685 #ifdef DEBUG_SYSTEM
686  fprintf(stderr, "MPEG picture header\n");
687 #endif
688  }
689 
690  /* Slice header */
691  while((header_size = slice_header(pointer+packet_size, read_buffer + read_size - pointer - packet_size)) != 0)
692  {
693  packet_size += header_size;
694 #ifdef DEBUG_SYSTEM
695  fprintf(stderr, "MPEG slice header\n");
696 #endif
697  }
698 
699  /* Audio frame */
700  if(audio_header(pointer+packet_size, &packet_size, &frametime))
701  {
702  stream_id = AUDIO_STREAMID;
703  stream_list[0]->streamid = stream_id;
704  stream_timestamp += frametime;
705 #ifdef DEBUG_SYSTEM
706  fprintf(stderr, "MPEG audio header [%d] time: %lf @ %lf\n", packet_size, frametime, stream_timestamp);
707 #endif
708  }
709  else
710  {
711  /* Check for next slice, picture, gop or sequence header */
712  register Uint8 * p;
713  register Uint8 c;
714 
715  p = pointer + packet_size;
716  state0:
717  c = *p;
718  p++;
719  if(p >= read_buffer + read_size) goto end;
720  if(c != 0) goto state0;
721 /* Not explicitly reached:
722  state1:
723  */
724  c = *p;
725  p++;
726  if(p >= read_buffer + read_size) goto end;
727  if(c != 0) goto state0;
728  state2:
729  c = *p;
730  p++;
731  if(p >= read_buffer + read_size) goto end;
732  if(c == 0) goto state2;
733  if(c != 1) goto state0;
734 /* Not explicitly reached:
735  state3:
736  */
737  c = *p;
738  p++;
739  if(p >= read_buffer + read_size) goto end;
740  if(c <= 0xaf) goto end;
741  if(c == 0xb8) goto end;
742  if(c == 0xb3) goto end;
743  goto state0;
744  end:
745 
746  if(p >= read_buffer + read_size) packet_size = (read_buffer + read_size) - pointer;
747  else packet_size = p - pointer - 4;
748  }
749 
750  if(stream_id == SYSTEM_STREAMID)
751  stream_id = 0;
752  }
753  else
754  {
755 #ifdef DEBUG_SYSTEM
756  fprintf(stderr,
757  "Warning: unexpected header %02x%02x%02x%02x at offset %d\n",
758  pointer[0],
759  pointer[1],
760  pointer[2],
761  pointer[3],
762  read_total - read_size + (pointer - read_buffer));
763 #endif
764  pointer++;
765  stream_list[0]->pos++;
766  seek_next_header();
767  RequestBuffer();
768  return(0);
769  }
770  }
771 
772  if(Eof())
773  {
774  RequestBuffer();
775  return(0);
776  }
777 
778  assert(packet_size <= MPEG_BUFFER_SIZE);
779 
780  if(skip_timestamp > timestamp){
781  int cur_seconds=int(timestamp)%60;
782 
783  if (cur_seconds%5==0){
784  fprintf(stderr, "Skipping to %02d:%02d (%02d:%02d)\r",
785  int(skip_timestamp)/60, int(skip_timestamp)%60,
786  int(timestamp)/60, cur_seconds);
787  }
788  pointer += packet_size;
789  stream_list[0]->pos += packet_size;
790  /* since we skip data, request more */
791  RequestBuffer();
792  return (0);
793  }
794 
795  switch(stream_id)
796  {
797  case 0:
798  /* Unknown stream, just get another packet */
799  pointer += packet_size;
800  stream_list[0]->pos += packet_size;
801  RequestBuffer();
802  return(0);
803 
804  case SYSTEM_STREAMID:
805  /* System header */
806 
807  /* This MPEG contain system information */
808  /* Parse the system header and create MPEG streams */
809 
810  /* Read the stream table */
811  pointer += 5;
812  stream_list[0]->pos += 5;
813 
814  while (pointer[0] & 0x80 )
815  {
816  /* If the stream doesn't already exist */
817  if(!get_stream(pointer[0]))
818  {
819  /* Create a new stream and add it to the list */
820  add_stream(new MPEGstream(this, pointer[0]));
821  }
822  pointer += 3;
823  stream_list[0]->pos += 3;
824  }
825  /* Hack to detect video streams that are not advertised */
826  if ( ! exist_stream(VIDEO_STREAMID, 0xF0) ) {
827  if ( pointer[3] == 0xb3 ) {
828  add_stream(new MPEGstream(this, VIDEO_STREAMID));
829  }
830  }
831  RequestBuffer();
832  return(stream_id);
833 
834  default:
835  MPEGstream * stream;
836 
837  /* Look for the stream the data must be given to */
838  stream = get_stream(stream_id);
839 
840  if(!stream)
841  {
842  /* Hack to detect video or audio streams that are not declared in system header */
843  if ( ((stream_id & 0xF0) == VIDEO_STREAMID) && !exist_stream(stream_id, 0xFF) ) {
844 #ifdef DEBUG_SYSTEM
845  fprintf(stderr, "Undeclared video packet, creating a new video stream\n");
846 #endif
847  stream = new MPEGstream(this, stream_id);
848  add_stream(stream);
849  }
850  else
851  if ( ((stream_id & 0xF0) == AUDIO_STREAMID) && !exist_stream(stream_id, 0xFF) ) {
852 #ifdef DEBUG_SYSTEM
853  fprintf(stderr, "Undeclared audio packet, creating a new audio stream\n");
854 #endif
855  stream = new MPEGstream(this, stream_id);
856  add_stream(stream);
857  }
858  else
859  {
860  /* No stream found for packet, skip it */
861  pointer += packet_size;
862  stream_list[0]->pos += packet_size;
863  RequestBuffer();
864  return(stream_id);
865  }
866  }
867 
868  /* Insert the new data at the end of the stream */
869  if(pointer + packet_size <= read_buffer + read_size)
870  {
871  if(packet_size) stream->insert_packet(pointer, packet_size, stream_timestamp);
872  pointer += packet_size;
873  }
874  else
875  {
876  stream->insert_packet(pointer, 0, stream_timestamp);
877  errorstream = true;
878  pointer = read_buffer + read_size;
879  }
880  return(stream_id);
881  }
882 }

953 double MPEGsystem::TotalTime()
954 {
955  off_t size, pos;
956  off_t file_ptr;
957  Uint8 * buffer, * p;
958  double time;
959 
960  /* Lock to avoid concurrent access to the stream */
961  SDL_mutexP(system_mutex);
962 
963  /* Save current position */
964  if((pos = SDL_RWtell(source)) < 0)
965  {
966  if(errno != ESPIPE)
967  {
968  errorstream = true;
969  SetError(strerror(errno));
970  }
971  SDL_mutexV(system_mutex);
972  return(false);
973  }
974 
975  file_ptr = 0;
976  p = buffer = new Uint8[MPEG_BUFFER_SIZE];
977  time = 0;
978 
979  /* If audio, compute total time according to bitrate of the first header and total size */
980  /* Note: this doesn't work on variable bitrate streams */
981  if(stream_list[0]->streamid == AUDIO_STREAMID)
982  {
983  do
984  {
985  if((size = SDL_RWseek(source, file_ptr, SEEK_SET)) < 0)
986  {
987  if(errno != ESPIPE)
988  {
989  errorstream = true;
990  SetError(strerror(errno));
991  }
992  SDL_mutexV(system_mutex);
993  return(false);
994  }
995 
996  if(SDL_RWread(source, buffer, 1, MPEG_BUFFER_SIZE) < 0) break;
997 
998  /* Search for a valid audio header */
999  for(p = buffer; p < buffer + MPEG_BUFFER_SIZE; p++)
1000  if(audio_aligned(p, buffer + MPEG_BUFFER_SIZE - p)) break;
1001 
1002  file_ptr += MPEG_BUFFER_SIZE;
1003  }
1004  while(p >= MPEG_BUFFER_SIZE + buffer);
1005 
1006  /* Extract time info from the first header */
1007  Uint32 framesize = 0;
1008  double frametime = 0.0;
1009  Uint32 totalsize;
1010 
1011  audio_header(p, &framesize, &frametime);
1012  totalsize = TotalSize();
1013  if(framesize)
1014  time = frametime * totalsize / framesize;
1015  }
1016  else
1017  {
1018  bool last_chance = false;
1019  do
1020  {
1021  /* Otherwise search the stream backwards for a valid header */
1022  file_ptr -= MPEG_BUFFER_SIZE;
1023  if ( file_ptr < -(Sint32)TotalSize() ) {
1024  last_chance = true;
1025  file_ptr = -(Sint32)TotalSize();
1026  }
1027 
1028  if((size = SDL_RWseek(source, file_ptr, SEEK_END)) < 0)
1029  {
1030  if(errno != ESPIPE)
1031  {
1032  errorstream = true;
1033  SetError(strerror(errno));
1034  }
1035  SDL_mutexV(system_mutex);
1036  return(false);
1037  }
1038 
1039  if(SDL_RWread(source, buffer, 1, MPEG_BUFFER_SIZE) < 0) break;
1040 
1041  if(stream_list[0]->streamid == SYSTEM_STREAMID)
1042  for(p = buffer + MPEG_BUFFER_SIZE - 1; p >= buffer;)
1043  {
1044  if(*p-- != 0xba) continue; // Packet header
1045  if(*p-- != 1) continue;
1046  if(*p-- != 0) continue;
1047  if(*p-- != 0) continue;
1048  ++p;
1049  break;
1050  }
1051  if(stream_list[0]->streamid == VIDEO_STREAMID)
1052  for(p = buffer + MPEG_BUFFER_SIZE - 1; p >= buffer;)
1053  {
1054  if(*p-- != 0xb8) continue; // GOP header
1055  if(*p-- != 1) continue;
1056  if(*p-- != 0) continue;
1057  if(*p-- != 0) continue;
1058  ++p;
1059  break;
1060  }
1061  }
1062  while( !last_chance && (p < buffer) );
1063 
1064  if ( p >= buffer ) {
1065  /* Extract time info from the last header */
1066  if(stream_list[0]->streamid == SYSTEM_STREAMID)
1067  packet_header(p, buffer + MPEG_BUFFER_SIZE - p, &time);
1068 
1069  if(stream_list[0]->streamid == VIDEO_STREAMID)
1070  gop_header(p, buffer + MPEG_BUFFER_SIZE - p, &time);
1071  }
1072  }
1073 
1074  delete[] buffer;
1075 
1076  /* Get back to saved position */
1077  if((pos = SDL_RWseek(source, pos, SEEK_SET)) < 0)
1078  {
1079  if(errno != ESPIPE)
1080  {
1081  errorstream = true;
1082  SetError(strerror(errno));
1083  }
1084  time = 0;
1085  }
1086 
1087  SDL_mutexV(system_mutex);
1088 
1089  return(time);
1090 }
