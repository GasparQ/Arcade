62  const FARHEADER2 *pmh2;
63  DWORD dwMemPos = sizeof(FARHEADER1);
64  UINT headerlen, stlen;
65  BYTE samplemap[8];
66 
67  if ((!lpStream) || (dwMemLength < 1024) || (bswapLE32(pmh1->id) != FARFILEMAGIC)
68  || (pmh1->magic2[0] != 13) || (pmh1->magic2[1] != 10) || (pmh1->magic2[2] != 26)) return FALSE;
69  headerlen = bswapLE16(pmh1->headerlen);
70  stlen = bswapLE16( pmh1->stlen );
71  if ((headerlen >= dwMemLength) || (dwMemPos + stlen + sizeof(FARHEADER2) >= dwMemLength)) return FALSE;
72  // Globals
73  m_nType = MOD_TYPE_FAR;
74  m_nChannels = 16;
75  m_nInstruments = 0;
76  m_nSamples = 0;
77  m_nSongPreAmp = 0x20;
78  m_nDefaultSpeed = pmh1->speed;
79  m_nDefaultTempo = 80;
80  m_nDefaultGlobalVolume = 256;

83  // Channel Setting
84  for (UINT nchpan=0; nchpan<16; nchpan++)
85  {
86  ChnSettings[nchpan].dwFlags = 0;
87  ChnSettings[nchpan].nPan = ((pmh1->panning[nchpan] & 0x0F) << 4) + 8;
88  ChnSettings[nchpan].nVolume = 64;
89  }
90  // Reading comment
91  if (stlen)

93  UINT szLen = stlen;
94  if (szLen > dwMemLength - dwMemPos) szLen = dwMemLength - dwMemPos;
95  if ((m_lpszSongComments = new char[szLen + 1]) != NULL)
96  {
97  memcpy(m_lpszSongComments, lpStream+dwMemPos, szLen);
98  m_lpszSongComments[szLen] = 0;
99  }
100  dwMemPos += stlen;
101  }
102  // Reading orders
103  if (sizeof(FARHEADER2) > dwMemLength - dwMemPos) return TRUE;
104  pmh2 = (const FARHEADER2 *)(lpStream + dwMemPos);
105  dwMemPos += sizeof(FARHEADER2);
106  if (dwMemPos >= dwMemLength) return TRUE;
107  for (UINT iorder=0; iorder<MAX_ORDERS; iorder++)
108  {
109  Order[iorder] = (iorder <= pmh2->snglen) ? pmh2->orders[iorder] : 0xFF;
110  }
111  m_nRestartPos = pmh2->loopto;
112  // Reading Patterns
113  dwMemPos += headerlen - (869 + stlen);
114  if (dwMemPos >= dwMemLength) return TRUE;
115 
116  // end byteswap of pattern data
117 
118  WORD *patsiz = (WORD *)pmh2->patsiz;
119  for (UINT ipat=0; ipat<256; ipat++) if (patsiz[ipat])
120  {
121  UINT patlen = bswapLE16(patsiz[ipat]);
122  if ((ipat >= MAX_PATTERNS) || (patlen < 2))
123  {
124  dwMemPos += patlen;
125  continue;
126  }
127  if (dwMemPos + patlen >= dwMemLength) return TRUE;
128  UINT rows = (patlen - 2) >> 6;
129  if (!rows)
130  {
131  dwMemPos += patlen;
132  continue;
133  }
134  if (rows > 256) rows = 256;
135  if (rows < 16) rows = 16;
136  PatternSize[ipat] = rows;
137  if ((Patterns[ipat] = AllocatePattern(rows, m_nChannels)) == NULL) return TRUE;
138  MODCOMMAND *m = Patterns[ipat];
139  UINT patbrk = lpStream[dwMemPos];
140  const BYTE *p = lpStream + dwMemPos + 2;
141  UINT max = rows*16*4;
142  if (max > patlen-2) max = patlen-2;
143  for (UINT len=0; len<max; len += 4, m++)
144  {
145  BYTE note = p[len];
146  BYTE ins = p[len+1];
147  BYTE vol = p[len+2];
148  BYTE eff = p[len+3];
149  if (note)
150  {
151  m->instr = ins + 1;
152  m->note = note + 36;
153  }
154  if (vol & 0x0F)
155  {
156  m->volcmd = VOLCMD_VOLUME;
157  m->vol = (vol & 0x0F) << 2;
158  if (m->vol <= 4) m->vol = 0;
159  }
160  switch(eff & 0xF0)
161  {
162  // 1.x: Portamento Up
163  case 0x10:
164  m->command = CMD_PORTAMENTOUP;
165  m->param = eff & 0x0F;
166  break;
167  // 2.x: Portamento Down
168  case 0x20:
169  m->command = CMD_PORTAMENTODOWN;
170  m->param = eff & 0x0F;
171  break;
172  // 3.x: Tone-Portamento
173  case 0x30:
174  m->command = CMD_TONEPORTAMENTO;
175  m->param = (eff & 0x0F) << 2;
176  break;
177  // 4.x: Retrigger
178  case 0x40:
179  m->command = CMD_RETRIG;
180  m->param = 6 / (1+(eff&0x0F)) + 1;
181  break;
182  // 5.x: Set Vibrato Depth
183  case 0x50:
184  m->command = CMD_VIBRATO;
185  m->param = (eff & 0x0F);
186  break;
187  // 6.x: Set Vibrato Speed
188  case 0x60:
189  m->command = CMD_VIBRATO;
190  m->param = (eff & 0x0F) << 4;
191  break;
192  // 7.x: Vol Slide Up
193  case 0x70:
194  m->command = CMD_VOLUMESLIDE;
195  m->param = (eff & 0x0F) << 4;
196  break;
197  // 8.x: Vol Slide Down
198  case 0x80:
199  m->command = CMD_VOLUMESLIDE;
200  m->param = (eff & 0x0F);
201  break;
202  // A.x: Port to vol
203  case 0xA0:
204  m->volcmd = VOLCMD_VOLUME;
205  m->vol = ((eff & 0x0F) << 2) + 4;
206  break;
207  // B.x: Set Balance
208  case 0xB0:
209  m->command = CMD_PANNING8;
210  m->param = (eff & 0x0F) << 4;
211  break;
212  // F.x: Set Speed
213  case 0xF0:
214  m->command = CMD_SPEED;
215  m->param = eff & 0x0F;
216  break;
217  default:
218  if ((patbrk) && (patbrk+1 == (len >> 6)) && (patbrk+1 != rows-1))
219  {
220  m->command = CMD_PATTERNBREAK;
221  patbrk = 0;
222  }
223  }
224  }
225  dwMemPos += patlen;
226  }
227  // Reading samples
228  if (dwMemPos + 8 >= dwMemLength) return TRUE;
229  memcpy(samplemap, lpStream+dwMemPos, 8);
230  dwMemPos += 8;
231  MODINSTRUMENT *pins = &Ins[1];
232  for (UINT ismp=0; ismp<64; ismp++, pins++) if (samplemap[ismp >> 3] & (1 << (ismp & 7)))
233  {
234  if (dwMemPos + sizeof(FARSAMPLE) > dwMemLength) return TRUE;
235  const FARSAMPLE *pfs = reinterpret_cast<const FARSAMPLE*>(lpStream + dwMemPos);
236  dwMemPos += sizeof(FARSAMPLE);
237  m_nSamples = ismp + 1;
238  memcpy(m_szNames[ismp+1], pfs->samplename, 32);
239  const DWORD length = bswapLE32( pfs->length ) ; /* endian fix - Toad */
240  pins->nLength = length ;
241  pins->nLoopStart = bswapLE32(pfs->reppos) ;
242  pins->nLoopEnd = bswapLE32(pfs->repend) ;
243  pins->nFineTune = 0;
244  pins->nC4Speed = 8363*2;
245  pins->nGlobalVol = 64;
246  pins->nVolume = pfs->volume << 4;
247  pins->uFlags = 0;
248  if ((pins->nLength > 3) && (dwMemPos + 4 < dwMemLength))
249  {
250  if (pfs->type & 1)
251  {
252  pins->uFlags |= CHN_16BIT;
253  pins->nLength >>= 1;
254  pins->nLoopStart >>= 1;
255  pins->nLoopEnd >>= 1;
256  }
257  if ((pfs->loop & 8) && (pins->nLoopEnd > 4)) pins->uFlags |= CHN_LOOP;
258  ReadSample(pins, (pins->uFlags & CHN_16BIT) ? RS_PCM16S : RS_PCM8S,
259  (LPSTR)(lpStream+dwMemPos), dwMemLength - dwMemPos);
260  }
261  dwMemPos += length;
262  }
263  return TRUE;
264 }
