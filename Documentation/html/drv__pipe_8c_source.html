84 static BOOL pipe_Init(void)
85 {
86  if(!target) {
87  _mm_errno=MMERR_OPENING_FILE;
88  return 1;
89  }
90 #if !defined unix && (!defined __APPLE__ || !defined __MACH__)
91 #ifdef __EMX__
92  _fsetmode(stdout, "b");
93 #endif
94 #ifdef __WATCOMC__
95  pipefile = _popen(target, "wb");
96 #else
97  pipefile = popen(target, "wb");
98 #endif
99  if (!pipefile) {
100  _mm_errno=MMERR_OPENING_FILE;
101  return 1;
102  }
103 #else
104  /* poor man's popen() */
105  if (pipe(pipefd)) {
106  _mm_errno = MMERR_OPENING_FILE;
107  return 1;
108  }
109  switch (pid=fork()) {
110  case -1:
111  close(pipefd[0]);
112  close(pipefd[1]);
113  pipefd[0]=pipefd[1]=-1;
114  _mm_errno=MMERR_OPENING_FILE;
115  return 1;
116  case 0:
117  if (pipefd[0]) {
118  dup2(pipefd[0],0);
119  close(pipefd[0]);
120  }
121  close(pipefd[1]);
122  if (!MD_DropPrivileges())
123  execl("/bin/sh","sh","-c",target,NULL);
124  exit(127);
125  }
126  close(pipefd[0]);
127  if (!(pipefile=fdopen(pipefd[1],"wb"))) {
128  _mm_errno=MMERR_OPENING_FILE;
129  return 1;
130  }
131 #endif
132  if(!(pipeout=_mm_new_file_writer(pipefile)))
133  return 1;
134  if(!(audiobuffer=(SBYTE*)_mm_malloc(BUFFERSIZE)))
135  return 1;
136 
137  md_mode|=DMODE_SOFT_MUSIC|DMODE_SOFT_SNDFX;
138 
139  return VC_Init();
140 }
