98 BOOL CSoundFile::ReadPSM(LPCBYTE lpStream, DWORD dwMemLength)
99 //-----------------------------------------------------------
100 {
101  PSMCHUNK *pfh = (PSMCHUNK *)lpStream;
102  DWORD dwMemPos, dwSongPos;
103  DWORD smpnames[MAX_SAMPLES];
104  DWORD patptrs[MAX_PATTERNS];
105  BYTE samplemap[MAX_SAMPLES];
106  UINT nPatterns;
107 
108  // Swap chunk
109  swap_PSMCHUNK(pfh);
110 
111  // Chunk0: "PSM ",filesize,"FILE"
112  if (dwMemLength < 256) return FALSE;
113  if (pfh->id == PSM_ID_OLD)
114  {
115  #ifdef PSM_LOG
116  Log("Old PSM format not supported\n");
117  #endif
118  return FALSE;
119  }
120  if ((pfh->id != PSM_ID_NEW) || (pfh->len+12 > dwMemLength) || (pfh->listid != IFFID_FILE)) return FALSE;
121  m_nType = MOD_TYPE_PSM;
122  m_nChannels = 16;
123  m_nSamples = 0;
124  nPatterns = 0;
125  dwMemPos = 12;
126  dwSongPos = 0;
127  for (UINT iChPan=0; iChPan<16; iChPan++)
128  {
129  UINT pan = (((iChPan & 3) == 1) || ((iChPan&3)==2)) ? 0xC0 : 0x40;
130  ChnSettings[iChPan].nPan = pan;
131  }
132  while (dwMemPos+8 < dwMemLength)
133  {
134  PSMCHUNK *pchunk = (PSMCHUNK *)(lpStream+dwMemPos);
135  swap_PSMCHUNK(pchunk);
136  if ((pchunk->len >= dwMemLength - 8) || (dwMemPos + pchunk->len + 8 > dwMemLength)) break;
137  dwMemPos += 8;
138  PUCHAR pdata = (PUCHAR)(lpStream+dwMemPos);
139  ULONG len = pchunk->len;
140  if (len) switch(pchunk->id)
141  {
142  // "TITL": Song title
143  case IFFID_TITL:
144  if (!pdata[0]) { pdata++; len--; }
145  memcpy(m_szNames[0], pdata, (len>31) ? 31 : len);
146  m_szNames[0][31] = 0;
147  break;
148  // "PBOD": Pattern
149  case IFFID_PBOD:
150  if ((len >= 12) && (nPatterns < MAX_PATTERNS))
151  {
152  patptrs[nPatterns++] = dwMemPos-8;
153  }
154  break;
155  // "SONG": Song description
156  case IFFID_SONG:
157  if ((len >= sizeof(PSMSONGHDR)+8) && (!dwSongPos))
158  {
159  dwSongPos = dwMemPos - 8;
160  }
161  break;
162  // "DSMP": Sample Data
163  case IFFID_DSMP:
164  if ((len >= sizeof(PSMSAMPLE)) && (m_nSamples+1 < MAX_SAMPLES))
165  {
166  m_nSamples++;
167  MODINSTRUMENT *pins = &Ins[m_nSamples];
168  PSMSAMPLE *psmp = (PSMSAMPLE *)pdata;
169  swap_PSMSAMPLE(psmp);
170  smpnames[m_nSamples] = psmp->smpid;
171  memcpy(m_szNames[m_nSamples], psmp->samplename, 31);
172  m_szNames[m_nSamples][31] = 0;
173  samplemap[m_nSamples-1] = (BYTE)m_nSamples;
174  // Init sample
175  pins->nGlobalVol = 0x40;
176  pins->nC4Speed = psmp->samplerate;
177  pins->nLength = psmp->length;
178  pins->nLoopStart = psmp->loopstart;
179  pins->nLoopEnd = psmp->loopend;
180  pins->nPan = 128;
181  pins->nVolume = (psmp->defvol+1) * 2;
182  pins->uFlags = (psmp->flags & 0x80) ? CHN_LOOP : 0;
183  if (pins->nLoopStart > 0) pins->nLoopStart--;
184  // Point to sample data
185  pdata += 0x60;
186  len -= 0x60;
187  // Load sample data
188  if ((pins->nLength > 3) && (len > 3))
189  {
190  ReadSample(pins, RS_PCM8D, (LPCSTR)pdata, len);
191  } else
192  {
193  pins->nLength = 0;
194  }
195  }
196  break;
197  #if 0
198  default:
199  {
200  CHAR s[8], s2[64];
201  *(DWORD *)s = pchunk->id;
202  s[4] = 0;
203  wsprintf(s2, "%s: %4d bytes @ %4d\n", s, pchunk->len, dwMemPos);
204  OutputDebugString(s2);
205  }
206  #endif
207  }
208  dwMemPos += pchunk->len;
209  }
210  // Step #1: convert song structure
211  PSMSONGHDR *pSong = (PSMSONGHDR *)(lpStream+dwSongPos+8);
212  if ((!dwSongPos) || (pSong->channels < 2) || (pSong->channels > 32)) return TRUE;
213  m_nChannels = pSong->channels;
214  // Valid song header -> convert attached chunks
215  {
216  DWORD dwSongEnd = dwSongPos + 8 + *(DWORD *)(lpStream+dwSongPos+4);
217  dwMemPos = dwSongPos + 8 + 11; // sizeof(PSMCHUNK)+sizeof(PSMSONGHDR)
218  while (dwMemPos + 8 < dwSongEnd)
219  {
220  PSMCHUNK *pchunk = (PSMCHUNK *)(lpStream+dwMemPos);
221  swap_PSMCHUNK(pchunk);
222  dwMemPos += 8;
223  if ((pchunk->len > dwSongEnd) || (dwMemPos + pchunk->len > dwSongEnd)) break;
224  PUCHAR pdata = (PUCHAR)(lpStream+dwMemPos);
225  ULONG len = pchunk->len;
226  switch(pchunk->id)
227  {
228  case IFFID_OPLH:
229  if (len >= 0x20)
230  {
231  UINT pos = len - 3;
232  while (pos > 5)
233  {
234  BOOL bFound = FALSE;
235  pos -= 5;
236  DWORD dwName = *(DWORD *)(pdata+pos);
237  for (UINT i=0; i<nPatterns; i++)
238  {
239  DWORD dwPatName = ((PSMPATTERN *)(lpStream+patptrs[i]+8))->name;
240  if (dwName == dwPatName)
241  {
242  bFound = TRUE;
243  break;
244  }
245  }
246  if ((!bFound) && (pdata[pos+1] > 0) && (pdata[pos+1] <= 0x10)
247  && (pdata[pos+3] > 0x40) && (pdata[pos+3] < 0xC0))
248  {
249  m_nDefaultSpeed = pdata[pos+1];
250  m_nDefaultTempo = pdata[pos+3];
251  break;
252  }
253  }
254  UINT iOrd = 0;
255  while ((pos+5<len) && (iOrd < MAX_ORDERS))
256  {
257  DWORD dwName = *(DWORD *)(pdata+pos);
258  for (UINT i=0; i<nPatterns; i++)
259  {
260  DWORD dwPatName = ((PSMPATTERN *)(lpStream+patptrs[i]+8))->name;
261  if (dwName == dwPatName)
262  {
263  Order[iOrd++] = i;
264  break;
265  }
266  }
267  pos += 5;
268  }
269  }
270  break;
271  }
272  dwMemPos += pchunk->len;
273  }
274  }
275 
276  // Step #2: convert patterns
277  for (UINT nPat=0; nPat<nPatterns; nPat++)
278  {
279  PSMPATTERN *pPsmPat = (PSMPATTERN *)(lpStream+patptrs[nPat]+8);
280  swap_PSMPATTERN(pPsmPat);
281  ULONG len = *(DWORD *)(lpStream+patptrs[nPat]+4) - 12;
282  UINT nRows = pPsmPat->rows;
283  if (len > pPsmPat->size) len = pPsmPat->size;
284  if ((nRows < 64) || (nRows > 256)) nRows = 64;
285  PatternSize[nPat] = nRows;
286  if ((Patterns[nPat] = AllocatePattern(nRows, m_nChannels)) == NULL) break;
287  MODCOMMAND *m = Patterns[nPat];
288  BYTE *p = pPsmPat->data;
289  UINT pos = 0;
290  UINT row = 0;
291  UINT oldch = 0;
292  BOOL bNewRow = FALSE;
293  #ifdef PSM_LOG
294  Log("Pattern %d at offset 0x%04X\n", nPat, (DWORD)(p - (BYTE *)lpStream));
295  #endif
296  while ((row < nRows) && (pos+1 < len))
297  {
298  UINT flags = p[pos++];
299  UINT ch = p[pos++];
300 
301  #ifdef PSM_LOG
302  //Log("flags+ch: %02X.%02X\n", flags, ch);
303  #endif
304  if (((flags & 0xf0) == 0x10) && (ch <= oldch) /*&& (!bNewRow)*/)
305  {
306  if ((pos+1<len) && (!(p[pos] & 0x0f)) && (p[pos+1] < m_nChannels))
307  {
308  #ifdef PSM_LOG
309  //if (!nPat) Log("Continuing on new row\n");
310  #endif
311  row++;
312  m += m_nChannels;
313  oldch = ch;
314  continue;
315  }
316  }
317  if ((pos >= len) || (row >= nRows)) break;
318  if (!(flags & 0xf0))
319  {
320  #ifdef PSM_LOG
321  //if (!nPat) Log("EOR(%d): %02X.%02X\n", row, p[pos], p[pos+1]);
322  #endif
323  row++;
324  m += m_nChannels;
325  bNewRow = TRUE;
326  oldch = ch;
327  continue;
328  }
329  bNewRow = FALSE;
330  if (ch >= m_nChannels)
331  {
332  #ifdef PSM_LOG
333  if (!nPat) Log("Invalid channel row=%d (0x%02X.0x%02X)\n", row, flags, ch);
334  #endif
335  ch = 0;
336  }
337  // Note + Instr
338  if ((flags & 0x40) && (pos+1 < len))
339  {
340  UINT note = p[pos++];
341  UINT nins = p[pos++];
342  #ifdef PSM_LOG
343  //if (!nPat) Log("note+ins: %02X.%02X\n", note, nins);
344  if ((!nPat) && (nins >= m_nSamples)) Log("WARNING: invalid instrument number (%d)\n", nins);
345  #endif
346  if ((note) && (note < 0x80)) note = (note>>4)*12+(note&0x0f)+12+1;
347  m[ch].instr = samplemap[nins];
348  m[ch].note = note;
349  }
350  // Volume
351  if ((flags & 0x20) && (pos < len))
352  {
353  m[ch].volcmd = VOLCMD_VOLUME;
354  m[ch].vol = p[pos++] / 2;
355  }
356  // Effect
357  if ((flags & 0x10) && (pos+1 < len))
358  {
359  UINT command = p[pos++];
360  UINT param = p[pos++];
361  // Convert effects
362  switch(command)
363  {
364  // 01: fine volslide up
365  case 0x01: command = CMD_VOLUMESLIDE; param |= 0x0f; break;
366  // 04: fine volslide down
367  case 0x04: command = CMD_VOLUMESLIDE; param>>=4; param |= 0xf0; break;
368  // 0C: portamento up
369  case 0x0C: command = CMD_PORTAMENTOUP; param = (param+1)/2; break;
370  // 0E: portamento down
371  case 0x0E: command = CMD_PORTAMENTODOWN; param = (param+1)/2; break;
372  // 33: Position Jump
373  case 0x33: command = CMD_POSITIONJUMP; break;
374  // 34: Pattern break
375  case 0x34: command = CMD_PATTERNBREAK; break;
376  // 3D: speed
377  case 0x3D: command = CMD_SPEED; break;
378  // 3E: tempo
379  case 0x3E: command = CMD_TEMPO; break;
380  // Unknown
381  default:
382  #ifdef PSM_LOG
383  Log("Unknown PSM effect pat=%d row=%d ch=%d: %02X.%02X\n", nPat, row, ch, command, param);
384  #endif
385  command = param = 0;
386  }
387  m[ch].command = (BYTE)command;
388  m[ch].param = (BYTE)param;
389  }
390  oldch = ch;
391  }
392  #ifdef PSM_LOG
393  if (pos < len)
394  {
395  Log("Pattern %d: %d/%d[%d] rows (%d bytes) -> %d bytes left\n", nPat, row, nRows, pPsmPat->rows, pPsmPat->size, len-pos);
396  }
397  #endif
398  }
399 
400  // Done (finally!)
401  return TRUE;
402 }

482  PROCEDURE PSM_UnpackPattern(VAR Source,Destination;PatternLength : WORD);
483  VAR
484  Witz : ARRAY[00..04] OF WORD;
485  I1,I2 : WORD;
486  I3,I4 : WORD;
487  TopicalByte : ^BYTE;
488  Pattern : PUnpackedPattern;
489  ChannelP : BYTE;
490  NoteP : BYTE;
491  InfoByte : BYTE;
492  CodeByte : BYTE;
493  InfoWord : WORD;
494  Effect : BYTE;
495  Opperand : BYTE;
496  Panning : BYTE;
497  Volume : BYTE;
498  PrevInfo : BYTE;
499  InfoIndex : BYTE;
500  BEGIN
501  Pattern := @Destination;
502  TopicalByte := @Source;
503  { *** Initialize patttern }
504  FOR I2 := 0 TO c_Maximum_NoteIndex DO
505  FOR I3 := 0 TO c_Maximum_ChannelIndex DO
506  BEGIN
507  Pattern^[I2,I3,c_Pattern_NoteIndex] := $FF;
508  Pattern^[I2,I3,c_Pattern_SampleIndex] := $00;
509  Pattern^[I2,I3,c_Pattern_VolumeIndex] := $FF;
510  Pattern^[I2,I3,c_Pattern_PanningIndex] := $FF;
511  Pattern^[I2,I3,c_Pattern_EffectIndex] := $00;
512  Pattern^[I2,I3,c_Pattern_OpperandIndex] := $00;
513  END;
514  { *** Byte-pointer on first pattern-entry }
515  ChannelP := $00;
516  NoteP := $00;
517  InfoByte := $00;
518  PrevInfo := $00;
519  InfoIndex := $02;
520  { *** read notes in pattern }
521  PSM_NotesInPattern := TopicalByte^; INC(TopicalByte); DEC(PatternLength); INC(InfoIndex);
522  PSM_ChannelInPattern := TopicalByte^; INC(TopicalByte); DEC(PatternLength); INC(InfoIndex);
523  { *** unpack pattern }
524  WHILE (INTEGER(PatternLength) > 0) AND (NoteP < c_Maximum_NoteIndex) DO
525  BEGIN
526  { *** Read info-byte }
527  InfoByte := TopicalByte^; INC(TopicalByte); DEC(PatternLength); INC(InfoIndex);
528  IF InfoByte <> $00 THEN
529  BEGIN
530  ChannelP := InfoByte AND $0F;
531  IF InfoByte AND 128 = 128 THEN { note and sample }
532  BEGIN
533  { *** read note }
534  CodeByte := TopicalByte^; INC(TopicalByte); DEC(PatternLength);
535  DEC(CodeByte);
536  CodeByte := CodeByte MOD 12 * 16 + CodeByte DIV 12 + 2;
537  Pattern^[NoteP,ChannelP,c_Pattern_NoteIndex] := CodeByte;
538  { *** read sample }
539  CodeByte := TopicalByte^; INC(TopicalByte); DEC(PatternLength);
540  Pattern^[NoteP,ChannelP,c_Pattern_SampleIndex] := CodeByte;
541  END;
542  IF InfoByte AND 64 = 64 THEN { Volume }
543  BEGIN
544  CodeByte := TopicalByte^; INC(TopicalByte); DEC(PatternLength);
545  Pattern^[NoteP,ChannelP,c_Pattern_VolumeIndex] := CodeByte;
546  END;
547  IF InfoByte AND 32 = 32 THEN { effect AND opperand }
548  BEGIN
549  Effect := TopicalByte^; INC(TopicalByte); DEC(PatternLength);
550  Opperand := TopicalByte^; INC(TopicalByte); DEC(PatternLength);
551  CASE Effect OF
552  c_PSM_SetSpeed:
553  BEGIN
554  Effect := c_I_Set_Speed;
555  END;
556  ELSE
557  BEGIN
558  Effect := c_I_NoEffect;
559  Opperand := $00;
560  END;
561  END;
562  Pattern^[NoteP,ChannelP,c_Pattern_EffectIndex] := Effect;
563  Pattern^[NoteP,ChannelP,c_Pattern_OpperandIndex] := Opperand;
564  END;
565  END ELSE INC(NoteP);
566  END;
567  END;
