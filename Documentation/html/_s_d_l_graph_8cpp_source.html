185 void SDLGraph::drawGameComponent(GameComponent const *component) throw(std::runtime_error)
186 {
187  SDL_Texture *texture;
188 
189  try
190  {
191  texture = loadSprite(component->getSprite2D());
192  }
193  catch (std::runtime_error exception)
194  {
195  SDL_Surface *surface;
196  SDL_Rect rect;
197  Uint32 color = 0;
198  SDL_Color compColor = colors[component->getColor()];
199 
200  color |= compColor.a << 24;
201  color |= compColor.r << 16;
202  color |= compColor.g << 8;
203  color |= compColor.b;
204  rect.x = 0;
205  rect.y = 0;
206  rect.w = SDLGraph::scale;
207  rect.h = SDLGraph::scale;
208  if ((surface = SDL_CreateRGBSurface(0, SDLGraph::scale, SDLGraph::scale, 32, 0, 0, 0, 0)) == NULL ||
209  (SDL_FillRect(surface, &rect, color)) < 0 ||
210  (texture = SDL_CreateTextureFromSurface(render, surface)) == NULL)
211  throw std::runtime_error(SDL_GetError());
212  SDL_FreeSurface(surface);
213  spriteCache[component->getSprite2D()] = texture;
214  }
215  displaySurface(texture, component->getPos());
216 }
