270  }
271  }
272  }
273  return (dwElapsedTime+500) / 1000;

562 UINT CSoundFile::GetNNAChannel(UINT nChn) const
563 //---------------------------------------------
564 {
565  const MODCHANNEL *pChn = &Chn[nChn];
566  // Check for empty channel
567  const MODCHANNEL *pi = &Chn[m_nChannels];
568  for (UINT i=m_nChannels; i<MAX_CHANNELS; i++, pi++) if (!pi->nLength) return i;
569  if (!pChn->nFadeOutVol) return 0;
570  // All channels are used: check for lowest volume
571  UINT result = 0;
572  DWORD vol = 64*65536; // 25%
573  DWORD envpos = 0xFFFFFF;
574  const MODCHANNEL *pj = &Chn[m_nChannels];
575  for (UINT j=m_nChannels; j<MAX_CHANNELS; j++, pj++)
576  {
577  if (!pj->nFadeOutVol) return j;
578  DWORD v = pj->nVolume;
579  if (pj->dwFlags & CHN_NOTEFADE)
580  v = v * pj->nFadeOutVol;
581  else
582  v <<= 16;
583  if (pj->dwFlags & CHN_LOOP) v >>= 1;
584  if ((v < vol) || ((v == vol) && (pj->nVolEnvPosition > envpos)))
585  {
586  envpos = pj->nVolEnvPosition;
587  vol = v;
588  result = j;
589  }
590  }
591  return result;
592 }

595 void CSoundFile::CheckNNA(UINT nChn, UINT instr, int note, BOOL bForceCut)
596 //------------------------------------------------------------------------
597 {
598  MODCHANNEL *pChn = &Chn[nChn];
599  INSTRUMENTHEADER *penv = pChn->pHeader, *pHeader = 0;
600  signed char *pSample;
601  if (note > 0x80) note = 0;
602  if (note < 1) return;
603  // Always NNA cut - using
604  if ((!(m_nType & (MOD_TYPE_IT|MOD_TYPE_MT2))) || (!m_nInstruments) || (bForceCut))
605  {
606  if ((m_dwSongFlags & SONG_CPUVERYHIGH)
607  || (!pChn->nLength) || (pChn->dwFlags & CHN_MUTE)
608  || ((!pChn->nLeftVol) && (!pChn->nRightVol))) return;
609  UINT n = GetNNAChannel(nChn);
610  if (!n) return;
611  MODCHANNEL *p = &Chn[n];
612  // Copy Channel
613  *p = *pChn;
614  p->dwFlags &= ~(CHN_VIBRATO|CHN_TREMOLO|CHN_PANBRELLO|CHN_MUTE|CHN_PORTAMENTO);
615  p->nMasterChn = nChn+1;
616  p->nCommand = 0;
617  // Cut the note
618  p->nFadeOutVol = 0;
619  p->dwFlags |= (CHN_NOTEFADE|CHN_FASTVOLRAMP);
620  // Stop this channel
621  pChn->nLength = pChn->nPos = pChn->nPosLo = 0;
622  pChn->nROfs = pChn->nLOfs = 0;
623  pChn->nLeftVol = pChn->nRightVol = 0;
624  return;
625  }
626  if (instr >= MAX_INSTRUMENTS) instr = 0;
627  pSample = pChn->pSample;
628  pHeader = pChn->pHeader;
629  if ((instr) && (note))
630  {
631  pHeader = Headers[instr];
632  if (pHeader)
633  {
634  UINT n = 0;
635  if (note <= 0x80)
636  {
637  n = pHeader->Keyboard[note-1];
638  note = pHeader->NoteMap[note-1];
639  if ((n) && (n < MAX_SAMPLES)) pSample = Ins[n].pSample;
640  }
641  } else pSample = NULL;
642  }
643  if (!penv) return;
644  MODCHANNEL *p = pChn;
645  for (UINT i=nChn; i<MAX_CHANNELS; p++, i++)
646  if ((i >= m_nChannels) || (p == pChn))
647  {
648  if (((p->nMasterChn == nChn+1) || (p == pChn)) && (p->pHeader))
649  {
650  BOOL bOk = FALSE;
651  // Duplicate Check Type
652  switch(p->pHeader->nDCT)
653  {
654  // Note
655  case DCT_NOTE:
656  if ((note) && (p->nNote == note) && (pHeader == p->pHeader)) bOk = TRUE;
657  break;
658  // Sample
659  case DCT_SAMPLE:
660  if ((pSample) && (pSample == p->pSample)) bOk = TRUE;
661  break;
662  // Instrument
663  case DCT_INSTRUMENT:
664  if (pHeader == p->pHeader) bOk = TRUE;
665  break;
666  }
667  // Duplicate Note Action
668  if (bOk)
669  {
670  switch(p->pHeader->nDNA)
671  {
672  // Cut
673  case DNA_NOTECUT:
674  KeyOff(i);
675  p->nVolume = 0;
676  break;
677  // Note Off
678  case DNA_NOTEOFF:
679  KeyOff(i);
680  break;
681  // Note Fade
682  case DNA_NOTEFADE:
683  p->dwFlags |= CHN_NOTEFADE;
684  break;
685  }
686  if (!p->nVolume)
687  {
688  p->nFadeOutVol = 0;
689  p->dwFlags |= (CHN_NOTEFADE|CHN_FASTVOLRAMP);
690  }
691  }
692  }
693  }
694  if (pChn->dwFlags & CHN_MUTE) return;
695  // New Note Action
696  if ((pChn->nVolume) && (pChn->nLength))
697  {
698  UINT n = GetNNAChannel(nChn);
699  if (n)
700  {
701  MODCHANNEL *p = &Chn[n];
702  // Copy Channel
703  *p = *pChn;
704  p->dwFlags &= ~(CHN_VIBRATO|CHN_TREMOLO|CHN_PANBRELLO|CHN_MUTE|CHN_PORTAMENTO);
705  p->nMasterChn = nChn+1;
706  p->nCommand = 0;
707  // Key Off the note
708  switch(pChn->nNNA)
709  {
710  case NNA_NOTEOFF: KeyOff(n); break;
711  case NNA_NOTECUT:
712  p->nFadeOutVol = 0;
713  case NNA_NOTEFADE: p->dwFlags |= CHN_NOTEFADE; break;
714  }
715  if (!p->nVolume)
716  {
717  p->nFadeOutVol = 0;
718  p->dwFlags |= (CHN_NOTEFADE|CHN_FASTVOLRAMP);
719  }
720  // Stop this channel
721  pChn->nLength = pChn->nPos = pChn->nPosLo = 0;
722  pChn->nROfs = pChn->nLOfs = 0;
723  }
724  }
725 }

728 BOOL CSoundFile::ProcessEffects()
729 //-------------------------------
730 {
731  MODCHANNEL *pChn = Chn;
732  int nBreakRow = -1, nPosJump = -1, nPatLoopRow = -1;
733 
734  for (UINT nChn=0; nChn<m_nChannels; nChn++, pChn++)
735  {
736  UINT instr = pChn->nRowInstr;
737  UINT volcmd = pChn->nRowVolCmd;
738  UINT vol = pChn->nRowVolume;
739  UINT cmd = pChn->nRowCommand;
740  UINT param = pChn->nRowParam;
741  bool bPorta = ((cmd != CMD_TONEPORTAMENTO) && (cmd != CMD_TONEPORTAVOL) && (volcmd != VOLCMD_TONEPORTAMENTO)) ? FALSE : TRUE;
742  UINT nStartTick = 0;
743 
744  pChn->dwFlags &= ~CHN_FASTVOLRAMP;
745  // Process special effects (note delay, pattern delay, pattern loop)
746  if ((cmd == CMD_MODCMDEX) || (cmd == CMD_S3MCMDEX))
747  {
748  if ((!param) && (m_nType & (MOD_TYPE_S3M|MOD_TYPE_IT))) param = pChn->nOldCmdEx; else pChn->nOldCmdEx = param;
749  // Note Delay ?
750  if ((param & 0xF0) == 0xD0)
751  {
752  nStartTick = param & 0x0F;
753  } else
754  if (!m_nTickCount)
755  {
756  // Pattern Loop ?
757  if ((((param & 0xF0) == 0x60) && (cmd == CMD_MODCMDEX))
758  || (((param & 0xF0) == 0xB0) && (cmd == CMD_S3MCMDEX)))
759  {
760  int nloop = PatternLoop(pChn, param & 0x0F);
761  if (nloop >= 0) nPatLoopRow = nloop;
762  } else
763  // Pattern Delay
764  if ((param & 0xF0) == 0xE0)
765  {
766  m_nPatternDelay = param & 0x0F;
767  }
768  }
769  }
770 
771  // Handles note/instrument/volume changes
772  if (m_nTickCount == nStartTick) // can be delayed by a note delay effect
773  {
774  UINT note = pChn->nRowNote;
775  if (instr) pChn->nNewIns = instr;
776  // XM: Key-Off + Sample == Note Cut
777  if (m_nType & (MOD_TYPE_MOD|MOD_TYPE_XM|MOD_TYPE_MT2))
778  {
779  if ((note == 0xFF) && ((!pChn->pHeader) || (!(pChn->pHeader->dwFlags & ENV_VOLUME))))
780  {
781  pChn->dwFlags |= CHN_FASTVOLRAMP;
782  pChn->nVolume = 0;
783  note = instr = 0;
784  }
785  }
786  if ((!note) && (instr))
787  {
788  if (m_nInstruments)
789  {
790  if (pChn->pInstrument) pChn->nVolume = pChn->pInstrument->nVolume;
791  if (m_nType & (MOD_TYPE_XM|MOD_TYPE_MT2))
792  {
793  pChn->dwFlags |= CHN_FASTVOLRAMP;
794  pChn->nVolEnvPosition = 0;
795  pChn->nPanEnvPosition = 0;
796  pChn->nPitchEnvPosition = 0;
797  pChn->nAutoVibDepth = 0;
798  pChn->nAutoVibPos = 0;
799  pChn->dwFlags &= ~CHN_NOTEFADE;
800  pChn->nFadeOutVol = 65536;
801  }
802  } else
803  {
804  if (instr < MAX_SAMPLES) pChn->nVolume = Ins[instr].nVolume;
805  }
806  if (!(m_nType & MOD_TYPE_IT)) instr = 0;
807  }
808  // Invalid Instrument ?
809  if (instr >= MAX_INSTRUMENTS) instr = 0;
810  // Note Cut/Off => ignore instrument
811  if (note >= 0xFE) instr = 0;
812  if ((note) && (note <= 128)) pChn->nNewNote = note;
813  // New Note Action ?
814  if ((note) && (note <= 128) && (!bPorta))
815  {
816  CheckNNA(nChn, instr, note, FALSE);
817  }
818  // Instrument Change ?
819  if (instr)
820  {
821  MODINSTRUMENT *psmp = pChn->pInstrument;
822  InstrumentChange(pChn, instr, bPorta, TRUE);
823  pChn->nNewIns = 0;
824  // Special IT case: portamento+note causes sample change -> ignore portamento
825  if ((m_nType & (MOD_TYPE_S3M|MOD_TYPE_IT))
826  && (psmp != pChn->pInstrument) && (note) && (note < 0x80))
827  {
828  bPorta = FALSE;
829  }
830  }
831  // New Note ?
832  if (note)
833  {
834  if ((!instr) && (pChn->nNewIns) && (note < 0x80))
835  {
836  InstrumentChange(pChn, pChn->nNewIns, bPorta, FALSE, (m_nType & (MOD_TYPE_XM|MOD_TYPE_MT2)) ? FALSE : TRUE);
837  pChn->nNewIns = 0;
838  }
839  NoteChange(nChn, note, bPorta, (m_nType & (MOD_TYPE_XM|MOD_TYPE_MT2)) ? FALSE : TRUE);
840  if ((bPorta) && (m_nType & (MOD_TYPE_XM|MOD_TYPE_MT2)) && (instr))
841  {
842  pChn->dwFlags |= CHN_FASTVOLRAMP;
843  pChn->nVolEnvPosition = 0;
844  pChn->nPanEnvPosition = 0;
845  pChn->nPitchEnvPosition = 0;
846  pChn->nAutoVibDepth = 0;
847  pChn->nAutoVibPos = 0;
848  }
849  }
850  // Tick-0 only volume commands
851  if (volcmd == VOLCMD_VOLUME)
852  {
853  if (vol > 64) vol = 64;
854  pChn->nVolume = vol << 2;
855  pChn->dwFlags |= CHN_FASTVOLRAMP;
856  } else
857  if (volcmd == VOLCMD_PANNING)
858  {
859  if (vol > 64) vol = 64;

861  pChn->dwFlags |= CHN_FASTVOLRAMP;

863  }

909  break;
910

912  PanningSlide(pChn, vol);
913  break;
914 
915  case VOLCMD_PANSLIDERIGHT:
916  PanningSlide(pChn, vol << 4);
917  break;

928  }

932  {

934  case CMD_VOLUME:
935  if (!m_nTickCount)
