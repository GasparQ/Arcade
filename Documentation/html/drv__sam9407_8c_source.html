136 static SWORD sampleLoad(SAMPLOAD *s, int type)
137 {
138  Bank *bankP;
139  SamModSamples modSamples;
140  SWORD handle;
141  SWORD *samples;
142 int rc;
143 
144  for(handle=0; handle<SAM_NUM_BANKS; handle++)
145  if(!banks[handle].inUse)
146  break;
147  if(handle>=SAM_NUM_BANKS) {
148  _mm_errno=MMERR_OUT_OF_HANDLES;
149  return -1;
150  }
151 
152  bankP=banks+handle;
153 
154  bankP->inUse=1;
155 
156  SL_SampleSigned(s);
157  SL_Sample8to16(s);
158 
159  bankP->length=s->sample->length;
160  bankP->loopstart=s->sample->loopstart;
161  bankP->loopend=s->sample->loopend ? s->sample->loopend : s->sample->length;
162  bankP->flags=s->sample->flags;
163 
164  if(!(samples=(SWORD *)malloc((bankP->length+NO_LOOP_SAMPLES)<<1))) {
165  bankP->inUse=0;
166  _mm_errno=MMERR_SAMPLE_TOO_BIG;
167  return -1;
168  }
169 
170  if(SL_Load(samples, s, bankP->length)) {
171  free(samples);
172  bankP->inUse=0;
173  _mm_errno=MMERR_SAMPLE_TOO_BIG;
174  return -1;
175  }
176 
177  if(!(bankP->flags & SF_LOOP)) {
178  memset(samples+bankP->length, 0, NO_LOOP_SAMPLES);
179  bankP->loopstart=bankP->length;
180  bankP->length+=NO_LOOP_SAMPLES;
181  bankP->loopend=bankP->length-1;
182  }
183 
184  modSamples.handle=handle;
185  modSamples.data=samples;
186  modSamples.size=bankP->length;
187  if((rc=ioctl(modfd, SAM_IOC_MOD_SAMPLES_LOAD, &modSamples))<0) {
188  free(samples);
189  bankP->inUse=0;
190  _mm_errno=MMERR_SAMPLE_TOO_BIG;
191  return -1;
192  }
193 
194  free(samples);
195 
196  return handle;
197 }

219 static BOOL init(void)

226  sprintf(devName, "/dev/sam%d_mod", card);
227  if((modfd=open(devName, O_RDWR))<0) {
228  _mm_errno=MMERR_OPENING_AUDIO;

231 
232  if(ioctl(modfd, SAM_IOC_API_INFO, &apiInfo)<0 ||
233  apiInfo.version!=SAM_MOD_API_VERSION) {

237

244

283 static void update(void)
284 {
285  static int wait_echo;
286  static long ticker_idx;
287 
288  int i;
289  Voice *voiceP;
290  SamModEvent event, eventBuf[SAM_NUM_VOICES*10+8], *eventP;
291 
292  eventP = eventBuf;
293 
294  for(i=0; i<md_numchn; i++) {
295  voiceP=voices+i;
296 
297  if((voiceP->pending & PENDING_STOP) ||
298  ((voiceP->pending & PENDING_PLAY) && voiceP->playing)) {
299  voiceP->playing=0;
300 
301  eventP->type=SAM_MOD_EVENT_STOP;
302  eventP->trigger.voice=i;
303  eventP=(SamModEvent *)((char *)eventP+sizeof(SamModTriggerEvent));
304 
305  eventP->type=SAM_MOD_EVENT_CLOSE;
306  eventP->trigger.voice=i;
307  eventP=(SamModEvent *)((char *)eventP+sizeof(SamModTriggerEvent));
308 
309  voiceP->pending&=~PENDING_STOP;
310  }
311 
312  if(voiceP->pending & PENDING_PLAY) {
313  Bank *bankP=banks+voiceP->handle;
314 
315  eventP->type=SAM_MOD_EVENT_OPEN;
316  eventP->trigger.voice=i;
317  eventP=(SamModEvent *)((char *)eventP+sizeof(SamModTriggerEvent));
318 
319  eventP->type=SAM_MOD_EVENT_MEM;
320  eventP->mem.voice=i;
321  eventP->mem.handle=voiceP->handle;
322  eventP->mem.sampleType=SamModSample16Bit;
323  eventP->mem.loopType=(bankP->flags & SF_BIDI) ? SamModLoopReverse : SamModLoopForward;
324  eventP->mem.start=voiceP->start;
325  eventP->mem.loop=bankP->loopstart;
326  eventP->mem.end=bankP->loopend;
327  eventP=(SamModEvent *)((char *)eventP+sizeof(SamModMemEvent));
328 
329  eventP->type=SAM_MOD_EVENT_VOL;
330  eventP->volOut.voice=i;
331  eventP->volOut.volume=0xFF;
332  eventP=(SamModEvent *)((char *)eventP+sizeof(SamModVolOutEvent));
333 
334  voiceP->pending|=PENDING_VOLUME|PENDING_FREQ;
335  }
336 
337  if(voiceP->pending & PENDING_VOLUME) {
338  eventP->type=SAM_MOD_EVENT_MAIN;
339  eventP->volSend.voice=i;
340  eventP->volSend.left=voiceP->vol*(255-voiceP->pan)/255;
341  eventP->volSend.right=voiceP->vol*voiceP->pan/255;
342  eventP=(SamModEvent *)((char *)eventP+sizeof(SamModVolSendEvent));
343 
344  voiceP->pending&=~PENDING_VOLUME;
345  }
346 
347  if(voiceP->pending & PENDING_FREQ) {
348  eventP->type=SAM_MOD_EVENT_PITCH;
349  eventP->pitch.voice=i;
350  eventP->pitch.freq=voiceP->freq;
351  eventP=(SamModEvent *)((char *)eventP+sizeof(SamModPitchEvent));
352 
353  voiceP->pending&=~PENDING_FREQ;
354  }
355 
356  if(voiceP->pending & PENDING_PLAY) {
357  voiceP->playing=1;
358 
359  eventP->type=SAM_MOD_EVENT_START;
360  eventP->trigger.voice=i;
361  eventP=(SamModEvent *)((char *)eventP+sizeof(SamModTriggerEvent));
362 
363  voiceP->pending&=~PENDING_PLAY;
364  }
365  }
366 
367  eventP->type=SAM_MOD_EVENT_WAIT_REL;
368  eventP->timer.time.tv_sec=0;
369  eventP->timer.time.tv_usec=2500000/md_bpm;
370  eventP=(SamModEvent *)((char *)eventP+sizeof(SamModTimerEvent));
371 
372  eventP->type=SAM_MOD_EVENT_ECHO;
373  eventP->echo.closure=(void *)(ticker_idx+1);
374  eventP=(SamModEvent *)((char *)eventP+sizeof(SamModEchoEvent));
375 
376  if(wait_echo)
377  while(read(modfd, &event, sizeof(event))>0 && (long)event.echo.closure!=ticker_idx);
378  wait_echo=1;
379  ticker_idx++;
380 
381  (*md_player)();
382 
383  write(modfd, eventBuf, (char *)eventP-(char *)eventBuf);
384 }
