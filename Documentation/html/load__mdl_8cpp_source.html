45 void ConvertMDLCommand(MODCOMMAND *m, UINT eff, UINT data)
46 //--------------------------------------------------------
47 {
48  UINT command = 0, param = data;
49  switch(eff)
50  {
51  case 0x01: command = CMD_PORTAMENTOUP; break;
52  case 0x02: command = CMD_PORTAMENTODOWN; break;
53  case 0x03: command = CMD_TONEPORTAMENTO; break;
54  case 0x04: command = CMD_VIBRATO; break;
55  case 0x05: command = CMD_ARPEGGIO; break;
56  case 0x07: command = (param < 0x20) ? CMD_SPEED : CMD_TEMPO; break;
57  case 0x08: command = CMD_PANNING8; param <<= 1; break;
58  case 0x0B: command = CMD_POSITIONJUMP; break;
59  case 0x0C: command = CMD_GLOBALVOLUME; break;
60  case 0x0D: command = CMD_PATTERNBREAK; param = (data & 0x0F) + (data>>4)*10; break;
61  case 0x0E:
62  command = CMD_S3MCMDEX;
63  switch(data & 0xF0)
64  {
65  case 0x00: command = 0; break; // What is E0x in MDL (there is a bunch) ?
66  case 0x10: if (param & 0x0F) { param |= 0xF0; command = CMD_PANNINGSLIDE; } else command = 0; break;
67  case 0x20: if (param & 0x0F) { param = (param << 4) | 0x0F; command = CMD_PANNINGSLIDE; } else command = 0; break;
68  case 0x30: param = (data & 0x0F) | 0x10; break; // glissando
69  case 0x40: param = (data & 0x0F) | 0x30; break; // vibrato waveform
70  case 0x60: param = (data & 0x0F) | 0xB0; break;
71  case 0x70: param = (data & 0x0F) | 0x40; break; // tremolo waveform
72  case 0x90: command = CMD_RETRIG; param &= 0x0F; break;
73  case 0xA0: param = (data & 0x0F) << 4; command = CMD_GLOBALVOLSLIDE; break;
74  case 0xB0: param = data & 0x0F; command = CMD_GLOBALVOLSLIDE; break;
75  case 0xF0: param = ((data >> 8) & 0x0F) | 0xA0; break;
76  }
77  break;
78  case 0x0F: command = CMD_SPEED; break;
79  case 0x10: if ((param & 0xF0) != 0xE0) { command = CMD_VOLUMESLIDE; if ((param & 0xF0) == 0xF0) param = ((param << 4) | 0x0F); else param >>= 2; } break;
80  case 0x20: if ((param & 0xF0) != 0xE0) { command = CMD_VOLUMESLIDE; if ((param & 0xF0) != 0xF0) param >>= 2; } break;
81  case 0x30: command = CMD_RETRIG; break;
82  case 0x40: command = CMD_TREMOLO; break;
83  case 0x50: command = CMD_TREMOR; break;
84  case 0xEF: if (param > 0xFF) param = 0xFF; command = CMD_OFFSET; break;
85  }
86  if (command)
87  {
88  m->command = command;
89  m->param = param;
90  }
91 }

94 void UnpackMDLTrack(MODCOMMAND *pat, UINT nChannels, UINT nRows, UINT nTrack, const BYTE *lpTracks)
95 //-------------------------------------------------------------------------------------------------
96 {
97  MODCOMMAND cmd, *m = pat;
98  UINT len = *((WORD *)lpTracks);
99  UINT pos = 0, row = 0, i;
100  lpTracks += 2;
101  for (UINT ntrk=1; ntrk<nTrack; ntrk++)
102  {
103  lpTracks += len;
104  len = *((WORD *)lpTracks);
105  lpTracks += 2;
106  }
107  cmd.note = cmd.instr = 0;
108  cmd.volcmd = cmd.vol = 0;
109  cmd.command = cmd.param = 0;
110  while ((row < nRows) && (pos < len))
111  {
112  UINT xx;
113  BYTE b = lpTracks[pos++];
114  xx = b >> 2;
115  switch(b & 0x03)
116  {
117  case 0x01:
118  for (i=0; i<=xx; i++)
119  {
120  if (row) *m = *(m-nChannels);
121  m += nChannels;
122  row++;
123  if (row >= nRows) break;
124  }
125  break;
126 
127  case 0x02:
128  if (xx < row) *m = pat[nChannels*xx];
129  m += nChannels;
130  row++;
131  break;
132 
133  case 0x03:
134  {
135  cmd.note = (xx & 0x01) ? lpTracks[pos++] : 0;
136  cmd.instr = (xx & 0x02) ? lpTracks[pos++] : 0;
137  cmd.volcmd = cmd.vol = 0;
138  cmd.command = cmd.param = 0;
139  if ((cmd.note < NOTE_MAX-12) && (cmd.note)) cmd.note += 12;
140  UINT volume = (xx & 0x04) ? lpTracks[pos++] : 0;
141  UINT commands = (xx & 0x08) ? lpTracks[pos++] : 0;
142  UINT command1 = commands & 0x0F;
143  UINT command2 = commands & 0xF0;
144  UINT param1 = (xx & 0x10) ? lpTracks[pos++] : 0;
145  UINT param2 = (xx & 0x20) ? lpTracks[pos++] : 0;
146  if ((command1 == 0x0E) && ((param1 & 0xF0) == 0xF0) && (!command2))
147  {
148  param1 = ((param1 & 0x0F) << 8) | param2;
149  command1 = 0xEF;
150  command2 = param2 = 0;
151  }
152  if (volume)
153  {
154  cmd.volcmd = VOLCMD_VOLUME;
155  cmd.vol = (volume+1) >> 2;
156  }
157  ConvertMDLCommand(&cmd, command1, param1);
158  if ((cmd.command != CMD_SPEED)
159  && (cmd.command != CMD_TEMPO)
160  && (cmd.command != CMD_PATTERNBREAK))
161  ConvertMDLCommand(&cmd, command2, param2);
162  *m = cmd;
163  m += nChannels;
164  row++;
165  }
166  break;
167 
168  // Empty Slots
169  default:
170  row += xx+1;
171  m += (xx+1)*nChannels;
172  if (row >= nRows) break;
173  }
174  }
175 }

490 //-----------------------------------------------------------------

498  bitnum += 8;
