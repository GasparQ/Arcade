464  if (8+len > dwMemLength || patpos[npat] > dwMemLength - (8+len)) continue;
465  PatternSize[npat] = rows;
466  if ((Patterns[npat] = AllocatePattern(rows, m_nChannels)) == NULL) continue;
467  memset(lastvalue, 0, sizeof(lastvalue));
468  memset(chnmask, 0, sizeof(chnmask));
469  MODCOMMAND *m = Patterns[npat];
470  UINT i = 0;
471  const BYTE *p = lpStream+patpos[npat]+8;
472  UINT nrow = 0;
473  while (nrow<rows)
474  {
475  if (i >= len) break;
476  BYTE b = p[i++];
477  if (!b)
478  {
479  nrow++;
480  m+=m_nChannels;
481  continue;
482  }
483  UINT ch = b & 0x7F;
484  if (ch) ch = (ch - 1) & 0x3F;
485  if (b & 0x80)
486  {
487  if (i >= len) break;
488  chnmask[ch] = p[i++];
489  }
490  if ((chnmask[ch] & 0x10) && (ch < m_nChannels))
491  {
492  m[ch].note = lastvalue[ch].note;
493  }
494  if ((chnmask[ch] & 0x20) && (ch < m_nChannels))
495  {
496  m[ch].instr = lastvalue[ch].instr;
497  }
498  if ((chnmask[ch] & 0x40) && (ch < m_nChannels))
499  {
500  m[ch].volcmd = lastvalue[ch].volcmd;
501  m[ch].vol = lastvalue[ch].vol;
502  }
503  if ((chnmask[ch] & 0x80) && (ch < m_nChannels))
504  {
505  m[ch].command = lastvalue[ch].command;
506  m[ch].param = lastvalue[ch].param;
507  }
508  if (chnmask[ch] & 1) // Note
509  {
510  if (i >= len) break;
511  UINT note = p[i++];
512  if (ch < m_nChannels)
513  {
514  if (note < 0x80) note++;
515  m[ch].note = note;
516  lastvalue[ch].note = note;
517  channels_used[ch] = TRUE;
518  }
519  }
520  if (chnmask[ch] & 2)
521  {
522  if (i >= len) break;
523  UINT instr = p[i++];
524  if (ch < m_nChannels)
525  {
526  m[ch].instr = instr;
527  lastvalue[ch].instr = instr;
528  }
529  }
530  if (chnmask[ch] & 4)
531  {
532  if (i >= len) break;
533  UINT vol = p[i++];
534  if (ch < m_nChannels)
535  {
536  // 0-64: Set Volume
537  if (vol <= 64) { m[ch].volcmd = VOLCMD_VOLUME; m[ch].vol = vol; } else
538  // 128-192: Set Panning
539  if ((vol >= 128) && (vol <= 192)) { m[ch].volcmd = VOLCMD_PANNING; m[ch].vol = vol - 128; } else
540  // 65-74: Fine Volume Up
541  if (vol < 75) { m[ch].volcmd = VOLCMD_FINEVOLUP; m[ch].vol = vol - 65; } else
542  // 75-84: Fine Volume Down
543  if (vol < 85) { m[ch].volcmd = VOLCMD_FINEVOLDOWN; m[ch].vol = vol - 75; } else
544  // 85-94: Volume Slide Up
545  if (vol < 95) { m[ch].volcmd = VOLCMD_VOLSLIDEUP; m[ch].vol = vol - 85; } else
546  // 95-104: Volume Slide Down
547  if (vol < 105) { m[ch].volcmd = VOLCMD_VOLSLIDEDOWN; m[ch].vol = vol - 95; } else
548  // 105-114: Pitch Slide Up
549  if (vol < 115) { m[ch].volcmd = VOLCMD_PORTADOWN; m[ch].vol = vol - 105; } else
550  // 115-124: Pitch Slide Down
551  if (vol < 125) { m[ch].volcmd = VOLCMD_PORTAUP; m[ch].vol = vol - 115; } else
552  // 193-202: Portamento To
553  if ((vol >= 193) && (vol <= 202)) { m[ch].volcmd = VOLCMD_TONEPORTAMENTO; m[ch].vol = vol - 193; } else
554  // 203-212: Vibrato
555  if ((vol >= 203) && (vol <= 212)) { m[ch].volcmd = VOLCMD_VIBRATOSPEED; m[ch].vol = vol - 203; }
556  lastvalue[ch].volcmd = m[ch].volcmd;
557  lastvalue[ch].vol = m[ch].vol;
558  }
559  }
560  // Reading command/param
561  if (chnmask[ch] & 8)
562  {
563  if (i > len - 2) break;
564  UINT cmd = p[i++];
565  UINT param = p[i++];
566  if (ch < m_nChannels)
567  {
568  if (cmd)
569  {
570  m[ch].command = cmd;
571  m[ch].param = param;
572  S3MConvert(&m[ch], TRUE);
573  lastvalue[ch].command = m[ch].command;
574  lastvalue[ch].param = m[ch].param;
575  }
576  }
577  }
578  }
579  }
580  for (UINT ncu=0; ncu<MAX_BASECHANNELS; ncu++)
581  {
582  if (ncu>=m_nChannels)
583  {
584  ChnSettings[ncu].nVolume = 64;
585  ChnSettings[ncu].dwFlags &= ~CHN_MUTE;
586  }
587  }
588  m_nMinPeriod = 8;
589  m_nMaxPeriod = 0xF000;
590  return TRUE;
591 }
