303 
304  ret = true;

307  if(!feof(f)) {
308  char *p = strchr(line, '\n');
309  if(0 == p) {

311  ret = false;

313  else {

322  ret = false;

325  ret = set_vc_field(filename, block, &field, needs_write, raw);
326  }
327  if(0 != field.field)
