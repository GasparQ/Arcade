108 void CSoundFile::S3MSaveConvert(UINT *pcmd, UINT *pprm, BOOL bIT) const
109 //---------------------------------------------------------------------
110 {
111  UINT command = *pcmd;
112  UINT param = *pprm;
113  switch(command)
114  {
115  case CMD_SPEED: command = 'A'; break;
116  case CMD_POSITIONJUMP: command = 'B'; break;
117  case CMD_PATTERNBREAK: command = 'C'; if (!bIT) param = ((param / 10) << 4) + (param % 10); break;
118  case CMD_VOLUMESLIDE: command = 'D'; break;
119  case CMD_PORTAMENTODOWN: command = 'E'; if ((param >= 0xE0) && (m_nType & (MOD_TYPE_MOD|MOD_TYPE_XM))) param = 0xDF; break;
120  case CMD_PORTAMENTOUP: command = 'F'; if ((param >= 0xE0) && (m_nType & (MOD_TYPE_MOD|MOD_TYPE_XM))) param = 0xDF; break;
121  case CMD_TONEPORTAMENTO: command = 'G'; break;
122  case CMD_VIBRATO: command = 'H'; break;
123  case CMD_TREMOR: command = 'I'; break;
124  case CMD_ARPEGGIO: command = 'J'; break;
125  case CMD_VIBRATOVOL: command = 'K'; break;
126  case CMD_TONEPORTAVOL: command = 'L'; break;
127  case CMD_CHANNELVOLUME: command = 'M'; break;
128  case CMD_CHANNELVOLSLIDE: command = 'N'; break;
129  case CMD_OFFSET: command = 'O'; break;
130  case CMD_PANNINGSLIDE: command = 'P'; break;
131  case CMD_RETRIG: command = 'Q'; break;
132  case CMD_TREMOLO: command = 'R'; break;
133  case CMD_S3MCMDEX: command = 'S'; break;
134  case CMD_TEMPO: command = 'T'; break;
135  case CMD_FINEVIBRATO: command = 'U'; break;
136  case CMD_GLOBALVOLUME: command = 'V'; break;
137  case CMD_GLOBALVOLSLIDE: command = 'W'; break;
138  case CMD_PANNING8:
139  command = 'X';
140  if ((bIT) && (m_nType != MOD_TYPE_IT) && (m_nType != MOD_TYPE_XM))
141  {
142  if (param == 0xA4) { command = 'S'; param = 0x91; } else
143  if (param <= 0x80) { param <<= 1; if (param > 255) param = 255; } else
144  command = param = 0;
145  } else
146  if ((!bIT) && ((m_nType == MOD_TYPE_IT) || (m_nType == MOD_TYPE_XM)))
147  {
148  param >>= 1;
149  }
150  break;
151  case CMD_PANBRELLO: command = 'Y'; break;
152  case CMD_MIDI: command = 'Z'; break;
153  case CMD_XFINEPORTAUPDOWN:
154  if (param & 0x0F) switch(param & 0xF0)
155  {
156  case 0x10: command = 'F'; param = (param & 0x0F) | 0xE0; break;
157  case 0x20: command = 'E'; param = (param & 0x0F) | 0xE0; break;
158  case 0x90: command = 'S'; break;
159  default: command = param = 0;
160  } else command = param = 0;
161  break;
162  case CMD_MODCMDEX:
163  command = 'S';
164  switch(param & 0xF0)
165  {
166  case 0x00: command = param = 0; break;
167  case 0x10: command = 'F'; param |= 0xF0; break;
168  case 0x20: command = 'E'; param |= 0xF0; break;
169  case 0x30: param = (param & 0x0F) | 0x10; break;
170  case 0x40: param = (param & 0x0F) | 0x30; break;
171  case 0x50: param = (param & 0x0F) | 0x20; break;
172  case 0x60: param = (param & 0x0F) | 0xB0; break;
173  case 0x70: param = (param & 0x0F) | 0x40; break;
174  case 0x90: command = 'Q'; param &= 0x0F; break;
175  case 0xA0: if (param & 0x0F) { command = 'D'; param = (param << 4) | 0x0F; } else command=param=0; break;
176  case 0xB0: if (param & 0x0F) { command = 'D'; param |= 0xF0; } else command=param=0; break;
177  }
178  break;
179  default: command = param = 0;
180  }
181  command &= ~0x40;
182  *pcmd = command;
183  *pprm = param;
184 }

197  UINT insnum,patnum,nins,npat;
198  DWORD insfile[MAX_SAMPLES];
199  WORD ptr[256];
200  DWORD dwMemPos;
201  BYTE insflags[MAX_SAMPLES], inspack[MAX_SAMPLES];
202 
203  if ((!lpStream) || (dwMemLength <= sizeof(S3MFILEHEADER)+sizeof(S3MSAMPLESTRUCT)+64)) return FALSE;
204  S3MFILEHEADER psfh = *(S3MFILEHEADER *)lpStream;
205 
206  psfh.reserved1 = bswapLE16(psfh.reserved1);
207  psfh.ordnum = bswapLE16(psfh.ordnum);
208  psfh.insnum = bswapLE16(psfh.insnum);
209  psfh.patnum = bswapLE16(psfh.patnum);
210  psfh.flags = bswapLE16(psfh.flags);
211  psfh.cwtv = bswapLE16(psfh.cwtv);
212  psfh.version = bswapLE16(psfh.version);
213  psfh.scrm = bswapLE32(psfh.scrm);
214  psfh.special = bswapLE16(psfh.special);
215 
216  if (psfh.scrm != 0x4D524353) return FALSE;
217  dwMemPos = 0x60;
218  m_nType = MOD_TYPE_S3M;
219  memset(m_szNames,0,sizeof(m_szNames));
220  memcpy(m_szNames[0], psfh.name, 28);
221  // Speed
222  m_nDefaultSpeed = psfh.speed;
223  if (m_nDefaultSpeed < 1) m_nDefaultSpeed = 6;
224  if (m_nDefaultSpeed > 0x1F) m_nDefaultSpeed = 0x1F;
225  // Tempo
226  m_nDefaultTempo = psfh.tempo;
227  if (m_nDefaultTempo < 40) m_nDefaultTempo = 40;
228  if (m_nDefaultTempo > 240) m_nDefaultTempo = 240;
229  // Global Volume
230  m_nDefaultGlobalVolume = psfh.globalvol << 2;
231  if ((!m_nDefaultGlobalVolume) || (m_nDefaultGlobalVolume > 256)) m_nDefaultGlobalVolume = 256;
232  m_nSongPreAmp = psfh.mastervol & 0x7F;
233  // Channels
234  m_nChannels = 4;
235  for (UINT ich=0; ich<32; ich++)
236  {
237  ChnSettings[ich].nPan = 128;
238  ChnSettings[ich].nVolume = 64;
239 
240  ChnSettings[ich].dwFlags = CHN_MUTE;
241  if (psfh.channels[ich] != 0xFF)
242  {
243  m_nChannels = ich+1;
244  UINT b = psfh.channels[ich] & 0x0F;
245  ChnSettings[ich].nPan = (b & 8) ? 0xC0 : 0x40;
246  ChnSettings[ich].dwFlags = 0;
247  }
248  }
249  if (m_nChannels < 4) m_nChannels = 4;
250  if ((psfh.cwtv < 0x1320) || (psfh.flags & 0x40)) m_dwSongFlags |= SONG_FASTVOLSLIDES;
251  // Reading pattern order
252  UINT iord = psfh.ordnum;
253  if (iord<1) iord = 1;
254  if (iord > MAX_ORDERS) iord = MAX_ORDERS;
255  if (iord)
256  {
257  memcpy(Order, lpStream+dwMemPos, iord);
258  dwMemPos += iord;
259  }
260  if ((iord & 1) && (lpStream[dwMemPos] == 0xFF)) dwMemPos++;
261  // Reading file pointers
262  insnum = nins = psfh.insnum;
263  if (insnum >= MAX_SAMPLES) insnum = MAX_SAMPLES-1;
264  m_nSamples = insnum;
265  patnum = npat = psfh.patnum;
266  if (patnum > MAX_PATTERNS) patnum = MAX_PATTERNS;
267  memset(ptr, 0, sizeof(ptr));
268 
269  // Ignore file if it has a corrupted header.
270  if (nins+npat > 256) return FALSE;
271 
272  if (nins+npat)
273  {
274  memcpy(ptr, lpStream+dwMemPos, 2*(nins+npat));
275  dwMemPos += 2*(nins+npat);
276  for (UINT j = 0; j < (nins+npat); ++j) {
277  ptr[j] = bswapLE16(ptr[j]);
278  }
279  if (psfh.panning_present == 252)
280  {
281  const BYTE *chnpan = lpStream+dwMemPos;
282  for (UINT i=0; i<32; i++) if (chnpan[i] & 0x20)
283  {
284  ChnSettings[i].nPan = ((chnpan[i] & 0x0F) << 4) + 8;
285  }
286  }
287  }
288  if (!m_nChannels) return TRUE;
289  // Reading instrument headers
290  memset(insfile, 0, sizeof(insfile));
291  for (UINT iSmp=1; iSmp<=insnum; iSmp++)
292  {
293  UINT nInd = ((DWORD)ptr[iSmp-1])*16;
294  if ((!nInd) || (nInd + 0x50 > dwMemLength)) continue;
295  S3MSAMPLESTRUCT pSmp;
296  memcpy(&pSmp, lpStream+nInd, 0x50);
297  memcpy(Ins[iSmp].name, &pSmp.dosname, 12);
298  insflags[iSmp-1] = pSmp.flags;
299  inspack[iSmp-1] = pSmp.pack;
300  memcpy(m_szNames[iSmp], pSmp.name, 28);
301  m_szNames[iSmp][28] = 0;
302  if ((pSmp.type==1) && (pSmp.scrs[2]=='R') && (pSmp.scrs[3]=='S'))
303  {
304  Ins[iSmp].nLength = boundInput(bswapLE32(pSmp.length), 4, MAX_SAMPLE_LENGTH);
305  Ins[iSmp].nLoopStart = boundInput(bswapLE32(pSmp.loopbegin), 4, Ins[iSmp].nLength - 1);
306  Ins[iSmp].nLoopEnd = boundInput(bswapLE32(pSmp.loopend), 4, Ins[iSmp].nLength);
307  Ins[iSmp].nVolume = boundInput(pSmp.vol, 0, 64) << 2;
308  Ins[iSmp].nGlobalVol = 64;
309  if (pSmp.flags&1) Ins[iSmp].uFlags |= CHN_LOOP;
310  UINT j = bswapLE32(pSmp.finetune);
311  if (!j) j = 8363;
312  if (j < 1024) j = 1024;
313  Ins[iSmp].nC4Speed = j;
314  insfile[iSmp] = (pSmp.hmem << 20) + (bswapLE16(pSmp.memseg) << 4);
315  // offset is invalid - ignore this sample.
316  if (insfile[iSmp] > dwMemLength) insfile[iSmp] = 0;
317  else if (insfile[iSmp]) {
318  // ignore duplicate samples.
319  for (int z=iSmp-1; z>=0; z--)
320  if (insfile[iSmp] == insfile[z])
321  insfile[iSmp] = 0;
322  }
323  if ((Ins[iSmp].nLoopStart >= Ins[iSmp].nLoopEnd) || (Ins[iSmp].nLoopEnd - Ins[iSmp].nLoopStart < 8))
324  Ins[iSmp].nLoopStart = Ins[iSmp].nLoopEnd = 0;
325  Ins[iSmp].nPan = 0x80;
326  }
327  }
328  // Reading patterns
329  for (UINT iPat=0; iPat<patnum; iPat++)
330  {
331  UINT nInd = ((DWORD)ptr[nins+iPat]) << 4;
332  if (nInd + 0x40 > dwMemLength) continue;
333  WORD len = bswapLE16(*((WORD *)(lpStream+nInd)));
334  nInd += 2;
335  PatternSize[iPat] = 64;
336  if ((!len) || (nInd + len > dwMemLength - 6)
337  || ((Patterns[iPat] = AllocatePattern(64, m_nChannels)) == NULL)) continue;
338  LPBYTE src = (LPBYTE)(lpStream+nInd);
339  // Unpacking pattern
340  MODCOMMAND *p = Patterns[iPat];
341  UINT row = 0;
342  UINT j = 0;
343  while (j < len)
344  {
345  BYTE b = src[j++];
346  if (!b)
347  {
348  if (++row >= 64) break;
349  } else
350  {
351  UINT chn = b & 0x1F;
352  if (chn < m_nChannels)
353  {
354  MODCOMMAND *m = &p[row*m_nChannels+chn];
355  if (b & 0x20)
356  {
357  m->note = src[j++];
358  if (m->note < 0xF0) m->note = (m->note & 0x0F) + 12*(m->note >> 4) + 13;
359  else if (m->note == 0xFF) m->note = 0;
360  m->instr = src[j++];
361  }
362  if (b & 0x40)
363  {
364  UINT vol = src[j++];
365  if ((vol >= 128) && (vol <= 192))
366  {
367  vol -= 128;
368  m->volcmd = VOLCMD_PANNING;
369  } else
370  {
371  if (vol > 64) vol = 64;
372  m->volcmd = VOLCMD_VOLUME;
373  }
374  m->vol = vol;
375  }
376  if (b & 0x80)
377  {
378  m->command = src[j++];
379  m->param = src[j++];
380  if (m->command) S3MConvert(m, FALSE);
381  }
382  } else
383  {
384  if (b & 0x20) j += 2;
385  if (b & 0x40) j++;
386  if (b & 0x80) j += 2;
387  }
388  if (j >= len) break;
389  }
390  }
391  }
392  // Reading samples
393  for (UINT iRaw=1; iRaw<=insnum; iRaw++) if ((Ins[iRaw].nLength) && (insfile[iRaw]))
394  {
395  UINT flags = (psfh.version == 1) ? RS_PCM8S : RS_PCM8U;
396  if (insflags[iRaw-1] & 4) flags += 5;
397  if (insflags[iRaw-1] & 2) flags |= RSF_STEREO;
398  if (inspack[iRaw-1] == 4) flags = RS_ADPCM4;
399  dwMemPos = insfile[iRaw];
400  if (dwMemPos < dwMemLength)
401  dwMemPos += ReadSample(&Ins[iRaw], flags, (LPSTR)(lpStream + dwMemPos), dwMemLength - dwMemPos);
402  }
403  m_nMinPeriod = 64;
404  m_nMaxPeriod = 32767;
405  if (psfh.flags & 0x10) m_dwSongFlags |= SONG_AMIGALIMITS;
406  return TRUE;
407 }

423 BOOL CSoundFile::SaveS3M(LPCSTR lpszFileName, UINT nPacking)
424 //----------------------------------------------------------
425 {
426  FILE *f;
427  BYTE header[0x60];
428  UINT nbo,nbi,nbp,i;
429  WORD patptr[128];
430  WORD insptr[128];
431  BYTE buffer[5*1024];
432  S3MSAMPLESTRUCT insex[128];
433 
434  if ((!m_nChannels) || (!lpszFileName)) return FALSE;
435  if ((f = fopen(lpszFileName, "wb")) == NULL) return FALSE;
436  // Writing S3M header
437  memset(header, 0, sizeof(header));
438  memset(insex, 0, sizeof(insex));
439  memcpy(header, m_szNames[0], 0x1C);
440  header[0x1B] = 0;
441  header[0x1C] = 0x1A;
442  header[0x1D] = 0x10;
443  nbo = (GetNumPatterns() + 15) & 0xF0;
444  if (!nbo) nbo = 16;
445  header[0x20] = nbo & 0xFF;
446  header[0x21] = nbo >> 8;
447  nbi = m_nInstruments;
448  if (!nbi) nbi = m_nSamples;
449  if (nbi > 99) nbi = 99;
450  header[0x22] = nbi & 0xFF;
451  header[0x23] = nbi >> 8;
452  nbp = 0;
453  for (i=0; Patterns[i]; i++) { nbp = i+1; if (nbp >= MAX_PATTERNS) break; }
454  for (i=0; i<MAX_ORDERS; i++) if ((Order[i] < MAX_PATTERNS) && (Order[i] >= nbp)) nbp = Order[i] + 1;
455  header[0x24] = nbp & 0xFF;
456  header[0x25] = nbp >> 8;
457  if (m_dwSongFlags & SONG_FASTVOLSLIDES) header[0x26] |= 0x40;
458  if ((m_nMaxPeriod < 20000) || (m_dwSongFlags & SONG_AMIGALIMITS)) header[0x26] |= 0x10;
459  header[0x28] = 0x20;
460  header[0x29] = 0x13;
461  header[0x2A] = 0x02; // Version = 1 => Signed samples
462  header[0x2B] = 0x00;
463  header[0x2C] = 'S';
464  header[0x2D] = 'C';
465  header[0x2E] = 'R';
466  header[0x2F] = 'M';
467  header[0x30] = m_nDefaultGlobalVolume >> 2;
468  header[0x31] = m_nDefaultSpeed;
469  header[0x32] = m_nDefaultTempo;
470  header[0x33] = ((m_nSongPreAmp < 0x20) ? 0x20 : m_nSongPreAmp) | 0x80; // Stereo
471  header[0x35] = 0xFC;
472  for (i=0; i<32; i++)
473  {
474  if (i < m_nChannels)
475  {
476  UINT tmp = (i & 0x0F) >> 1;
477  header[0x40+i] = (i & 0x10) | ((i & 1) ? 8+tmp : tmp);
478  } else header[0x40+i] = 0xFF;
479  }
480  fwrite(header, 0x60, 1, f);
481  fwrite(Order, nbo, 1, f);
482  memset(patptr, 0, sizeof(patptr));
483  memset(insptr, 0, sizeof(insptr));
484  UINT ofs0 = 0x60 + nbo;
485  UINT ofs1 = ((0x60 + nbo + nbi*2 + nbp*2 + 15) & 0xFFF0) + 0x20;
486  UINT ofs = ofs1;
487 
488  for (i=0; i<nbi; i++) insptr[i] = (WORD)((ofs + i*0x50) / 16);
489  for (i=0; i<nbp; i++) patptr[i] = (WORD)((ofs + nbi*0x50) / 16);
490  fwrite(insptr, nbi, 2, f);
491  fwrite(patptr, nbp, 2, f);
492  if (header[0x35] == 0xFC)
493  {
494  BYTE chnpan[32];
495  for (i=0; i<32; i++)
496  {
497  chnpan[i] = 0x20 | (ChnSettings[i].nPan >> 4);
498  }
499  fwrite(chnpan, 0x20, 1, f);
500  }
501  if ((nbi*2+nbp*2) & 0x0F)
502  {
503  fwrite(S3MFiller, 0x10 - ((nbi*2+nbp*2) & 0x0F), 1, f);
504  }
505  ofs1 = ftell(f);
506  fwrite(insex, nbi, 0x50, f);
507  // Packing patterns
508  ofs += nbi*0x50;
509  for (i=0; i<nbp; i++)
510  {
511  WORD len = 64;
512  memset(buffer, 0, sizeof(buffer));
513  patptr[i] = ofs / 16;
514  if (Patterns[i])
515  {
516  len = 2;
517  MODCOMMAND *p = Patterns[i];
518  for (int row=0; row<64; row++) if (row < PatternSize[i])
519  {
520  for (UINT j=0; j<m_nChannels; j++)
521  {
522  UINT b = j;
523  MODCOMMAND *m = &p[row*m_nChannels+j];
524  UINT note = m->note;
525  UINT volcmd = m->volcmd;
526  UINT vol = m->vol;
527  UINT command = m->command;
528  UINT param = m->param;
529 
530  if ((note) || (m->instr)) b |= 0x20;
531  if (!note) note = 0xFF; else
532  if (note >= 0xFE) note = 0xFE; else
