674  lp->size += sp->data_length + 1;
675 
676  if (1 != fread(sp->data, sp->data_length, 1, fp))
677  goto fail;

684  tmp=newdta=(uint16 *)safe_malloc(sp->data_length*2 + 2);

686  *tmp++ = (uint16)(*cp++) << 8;
687  cp=(uint8 *)(sp->data);
688  sp->data = (sample_t *)newdta;
689  free(cp);
690  sp->data_length *= 2;
691  sp->loop_start *= 2;
692  sp->loop_end *= 2;
693  }
694 #if SDL_BYTEORDER == SDL_BIG_ENDIAN

696  /* convert to machine byte order */

704  }
705  }
706 #endif

709  {
710  int32 i=sp->data_length/2;
711  int16 *tmp=(int16 *)sp->data;
712  while (i--)
713  *tmp++ ^= 0x8000;
714  }

716  /* Reverse reverse loops and pass them off as normal loops */
717  if (sp->modes & MODES_REVERSE)

721  whole sample. We do the same because the GUS does not SUCK. */

724  reverse_data((int16 *)sp->data, 0, sp->data_length/2);
725

727  sp->loop_start=sp->data_length - sp->loop_end;
728  sp->loop_end=sp->data_length - t;
729 
730  sp->modes &= ~MODES_REVERSE;
731  sp->modes |= MODES_LOOPING; /* just in case */
732  }
733 
734  /* If necessary do some anti-aliasing filtering */
735 
736  if (antialiasing_allowed)
737  antialiasing(sp,play_mode->rate);
738 
739 #ifdef ADJUST_SAMPLE_VOLUMES
740  if (amp!=-1)
741  sp->volume=(FLOAT_T)((amp) / 100.0);
742  else if (sf2flag)
743  sp->volume=(FLOAT_T)((sample_volume) / 255.0);
744  else
745  {
746  /* Try to determine a volume scaling factor for the sample.
747  This is a very crude adjustment, but things sound more
748  balanced with it. Still, this should be a runtime option. */
749  uint32 i, numsamps=sp->data_length/2;
750  uint32 higher=0, highcount=0;
751  int16 maxamp=0,a;
752  int16 *tmp=(int16 *)sp->data;
753  i = numsamps;
754  while (i--)
755  {
756  a=*tmp++;
757  if (a<0) a=-a;
758  if (a>maxamp)
759  maxamp=a;
760  }
761  tmp=(int16 *)sp->data;
762  i = numsamps;
763  while (i--)
764  {
765  a=*tmp++;
766  if (a<0) a=-a;
767  if (a > 3*maxamp/4)
768  {
769  higher += a;
770  highcount++;
771  }
772  }
773  if (highcount) higher /= highcount;
774  else higher = 10000;
775  sp->volume = (32768.0 * 0.875) / (double)higher ;
776  ctl->cmsg(CMSG_INFO, VERB_DEBUG, " * volume comp: %f", sp->volume);
777  }
778 #else
779  if (amp!=-1)
780  sp->volume=(double)(amp) / 100.0;
781  else
782  sp->volume=1.0;
783 #endif
784 
785  sp->data_length /= 2; /* These are in bytes. Convert into samples. */
786 
787  sp->loop_start /= 2;
788  sp->loop_end /= 2;
789  sp->data[sp->data_length] = sp->data[sp->data_length-1];
790 
791  /* Then fractional samples */
792  sp->data_length <<= FRACTION_BITS;
793  sp->loop_start <<= FRACTION_BITS;
794  sp->loop_end <<= FRACTION_BITS;
795 
796  /* trim off zero data at end */
797  {
798  int ls = sp->loop_start>>FRACTION_BITS;
799  int le = sp->loop_end>>FRACTION_BITS;
800  int se = sp->data_length>>FRACTION_BITS;
801  while (se > 1 && !sp->data[se-1]) se--;
802  if (le > se) le = se;
803  if (ls >= le) sp->modes &= ~MODES_LOOPING;
804  sp->loop_end = le<<FRACTION_BITS;
805  sp->data_length = se<<FRACTION_BITS;
806  }
807 
808  /* Adjust for fractional loop points. This is a guess. Does anyone
809  know what "fractions" really stands for? */
810  sp->loop_start |=
811  (fractions & 0x0F) << (FRACTION_BITS-4);
812  sp->loop_end |=
813  ((fractions>>4) & 0x0F) << (FRACTION_BITS-4);
814 
815  /* If this instrument will always be played on the same note,
816  and it's not looped, we can resample it now. */
817  if (sp->note_to_use && !(sp->modes & MODES_LOOPING))
818  pre_resample(sp);
819 
820 #ifdef LOOKUP_HACK
821  /* Squash the 16-bit data into 8 bits. */
822  {
823  uint8 *gulp,*ulp;
824  int16 *swp;
825  int l=sp->data_length >> FRACTION_BITS;
826  gulp=ulp=safe_malloc(l+1);
827  swp=(int16 *)sp->data;
828  while(l--)
829  *ulp++ = (*swp++ >> 8) & 0xFF;
830  free(sp->data);
831  sp->data=(sample_t *)gulp;
832  }
833 #endif
834 
835  if (strip_tail==1)
836  {
837  /* Let's not really, just say we did. */
838  ctl->cmsg(CMSG_INFO, VERB_DEBUG, " - Stripping tail");
839  sp->data_length = sp->loop_end;
840  }
841  } /* end of sample loop */
842  } /* end of stereo layer loop */
843  } /* end of vlayer loop */
844 
845 
846  close_file(fp);
847  return headlp;
848 }
