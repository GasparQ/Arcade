126 static void get_entry_tag(GtkEntry * entry, const char *name)
127 {
128  gchar *text;
129  char *utf8;
130 
131  text = gtk_entry_get_text(entry);
132  if (!text || strlen(text) == 0)
133  return;
134  if(flac_cfg.title.convert_char_set)
135  utf8 = convert_from_user_to_utf8(text);
136  else
137  utf8 = text;
138 
139  FLAC_plugin__tags_add_tag_utf8(tags_, name, utf8, /*separator=*/0);
140 
141  if(flac_cfg.title.convert_char_set)
142  free(utf8);
143 }

198 static void show_file_info(void)
199 {
200  FLAC__StreamMetadata streaminfo;
201  struct stat _stat;
202 
203  gtk_label_set_text(GTK_LABEL(flac_samplerate), "");
204  gtk_label_set_text(GTK_LABEL(flac_channels), "");
205  gtk_label_set_text(GTK_LABEL(flac_bits_per_sample), "");
206  gtk_label_set_text(GTK_LABEL(flac_blocksize), "");
207  gtk_label_set_text(GTK_LABEL(flac_filesize), "");
208  gtk_label_set_text(GTK_LABEL(flac_samples), "");
209  gtk_label_set_text(GTK_LABEL(flac_bitrate), "");
210 
211  if(!FLAC__metadata_get_streaminfo(current_filename, &streaminfo)) {
212  return;
213  }
214 
215  label_set_text(flac_samplerate, _("Samplerate: %d Hz"), streaminfo.data.stream_info.sample_rate);
216  label_set_text(flac_channels, _("Channels: %d"), streaminfo.data.stream_info.channels);
217  label_set_text(flac_bits_per_sample, _("Bits/Sample: %d"), streaminfo.data.stream_info.bits_per_sample);
218  if(streaminfo.data.stream_info.min_blocksize == streaminfo.data.stream_info.max_blocksize)
219  label_set_text(flac_blocksize, _("Blocksize: %d"), streaminfo.data.stream_info.min_blocksize);
220  else
221  label_set_text(flac_blocksize, _("Blocksize: variable\n min/max: %d/%d"), streaminfo.data.stream_info.min_blocksize, streaminfo.data.stream_info.max_blocksize);
222 
223  if (streaminfo.data.stream_info.total_samples)
224  label_set_text(flac_samples, _("Samples: %llu\nLength: %d:%.2d"),
225  streaminfo.data.stream_info.total_samples,
226  (int)(streaminfo.data.stream_info.total_samples / streaminfo.data.stream_info.sample_rate / 60),
227  (int)(streaminfo.data.stream_info.total_samples / streaminfo.data.stream_info.sample_rate % 60));
228 
229  if(!stat(current_filename, &_stat) && S_ISREG(_stat.st_mode)) {
230 #if _FILE_OFFSET_BITS == 64
231  label_set_text(flac_filesize, _("Filesize: %lld B"), _stat.st_size);
232 #else
233  label_set_text(flac_filesize, _("Filesize: %ld B"), _stat.st_size);
234 #endif
235  if (streaminfo.data.stream_info.total_samples)
236  label_set_text(flac_bitrate, _("Avg. bitrate: %.1f kb/s\nCompression ratio: %.1f%%"),
237  8.0 * (float)(_stat.st_size) / (1000.0 * (float)streaminfo.data.stream_info.total_samples / (float)streaminfo.data.stream_info.sample_rate),
238  100.0 * (float)_stat.st_size / (float)(streaminfo.data.stream_info.bits_per_sample / 8 * streaminfo.data.stream_info.channels * streaminfo.data.stream_info.total_samples));
239  }
240 }

242 static void show_replaygain(void)
243 {
244  /* known limitation: If only one of gain and peak is set, neither will be shown. This is true for
245  * both track and album replaygain tags. Written so it will be easy to fix, with some trouble. */
246 
247  gtk_label_set_text(GTK_LABEL(replaygain_reference), "");
248  gtk_label_set_text(GTK_LABEL(replaygain_track_gain), "");
249  gtk_label_set_text(GTK_LABEL(replaygain_album_gain), "");
250  gtk_label_set_text(GTK_LABEL(replaygain_track_peak), "");
251  gtk_label_set_text(GTK_LABEL(replaygain_album_peak), "");
252 
253  double reference, track_gain, track_peak, album_gain, album_peak;
254  FLAC__bool reference_set, track_gain_set, track_peak_set, album_gain_set, album_peak_set;
255 
256  if(!FLAC_plugin__replaygain_get_from_file(
257  current_filename,
258  &reference, &reference_set,
259  &track_gain, &track_gain_set,
260  &album_gain, &album_gain_set,
261  &track_peak, &track_peak_set,
262  &album_peak, &album_peak_set
263  ))
264  return;
265 
266  if(reference_set)
267  label_set_text(replaygain_reference, _("ReplayGain Reference Loudness: %2.1f dB"), reference);
268  if(track_gain_set)
269  label_set_text(replaygain_track_gain, _("ReplayGain Track Gain: %+2.2f dB"), track_gain);
270  if(album_gain_set)
271  label_set_text(replaygain_album_gain, _("ReplayGain Album Gain: %+2.2f dB"), album_gain);
272  if(track_peak_set)
273  label_set_text(replaygain_track_peak, _("ReplayGain Track Peak: %1.8f"), track_peak);
274  if(album_peak_set)
275  label_set_text(replaygain_album_peak, _("ReplayGain Album Peak: %1.8f"), album_peak);
276 }
