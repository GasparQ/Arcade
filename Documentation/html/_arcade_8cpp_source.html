128 std::vector<std::string> arcade::Arcade::loadFilesFromDir(std::string const &dirName, regex_t &nameRestric)
129 {
130  std::vector<std::string> names;
131  DIR *directory;
132  struct dirent *pent;
133 
134  if ((directory = opendir(dirName.c_str())) == NULL)
135  throw std::runtime_error("arcade: cannot open directory '" + dirName + "'");
136  while ((pent = readdir(directory)) != NULL)

138  if (pent->d_name[0] == '.' || !isLibNameValid(pent->d_name, nameRestric))
