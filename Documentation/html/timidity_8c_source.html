299 int Timidity_Init(int rate, int format, int channels, int samples)
300 {
301  const char *env = getenv("TIMIDITY_CFG");
302  if (!env || read_config_file(env)<0) {
303  if (read_config_file(CONFIG_FILE)<0) {
304  if (read_config_file(CONFIG_FILE_ETC)<0) {
305  if (read_config_file(CONFIG_FILE_ETC_TIMIDITY_FREEPATS)<0) {
306  return(-1);
307  }
308  }
309  }
310  }
311 
312  if (channels < 1 || channels == 3 || channels == 5 || channels > 6) return(-1);
313 
314  num_ochannels = channels;
315 
316  /* Set play mode parameters */
317  play_mode->rate = rate;
318  play_mode->encoding = 0;
319  if ( (format&0xFF) == 16 ) {
320  play_mode->encoding |= PE_16BIT;
321  }
322  if ( (format&0x8000) ) {
323  play_mode->encoding |= PE_SIGNED;
324  }
325  if ( channels == 1 ) {
326  play_mode->encoding |= PE_MONO;
327  }
328  switch (format) {
329  case AUDIO_S8:
330  s32tobuf = s32tos8;
331  break;
332  case AUDIO_U8:
333  s32tobuf = s32tou8;
334  break;
335  case AUDIO_S16LSB:
336  s32tobuf = s32tos16l;
337  break;
338  case AUDIO_S16MSB:
339  s32tobuf = s32tos16b;
340  break;
341  case AUDIO_U16LSB:
342  s32tobuf = s32tou16l;
343  break;
344  case AUDIO_U16MSB:
345  s32tobuf = s32tou16b;
346  break;
347  default:
348  ctl->cmsg(CMSG_ERROR, VERB_NORMAL, "Unsupported audio format");
349  return(-1);
350  }
351  AUDIO_BUFFER_SIZE = samples;
352 
353  /* Allocate memory for mixing (WARNING: Memory leak!) */
354  resample_buffer = safe_malloc(AUDIO_BUFFER_SIZE*sizeof(resample_t)+100);
355  common_buffer = safe_malloc(AUDIO_BUFFER_SIZE*num_ochannels*sizeof(int32));
356 
357  init_tables();
358 
359  if (ctl->open(0, 0)) {
360  ctl->cmsg(CMSG_ERROR, VERB_NORMAL, "Couldn't open %s\n", ctl->id_name);
361  return(-1);
362  }
363 
364  if (!control_ratio) {
365  control_ratio = play_mode->rate / CONTROLS_PER_SECOND;
366  if(control_ratio<1)
367  control_ratio=1;
368  else if (control_ratio > MAX_CONTROL_RATIO)
369  control_ratio=MAX_CONTROL_RATIO;
370  }
371  if (*def_instr_name)
372  set_default_instrument(def_instr_name);
373  return(0);
374 }
