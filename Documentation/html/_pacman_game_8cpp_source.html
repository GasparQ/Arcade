217 extern "C" void Play(void)
218 {
219  arcade::CommandType c;
220  PacmanGame pacman;
221  struct arcade::GetMap *map;
222  size_t mapSize = sizeof(*map) + ArcadeSystem::winWidth * ArcadeSystem::winHeight * sizeof(uint16_t);
223 
224  if ((map = (arcade::GetMap *)malloc(mapSize)) == NULL)
225  throw std::bad_alloc();
226  map->type = arcade::CommandType::GET_MAP;
227  map->width = ArcadeSystem::winWidth;
228  map->height = ArcadeSystem::winHeight;
229  while (read(0, &c, sizeof(c)))
230  {
231  switch (static_cast<arcade::CommandType>(c))
232  {
233  case arcade::CommandType::WHERE_AM_I:
234  whereAmI(pacman);
235  break;
236  case arcade::CommandType::GET_MAP:
237  updateMap(map, pacman);
238  write(1, map, mapSize);
239  break;
240  case arcade::CommandType::GO_UP:
241  const_cast<Pacman *>(&pacman.getPacman())->goUp(NULL);
242  break;
243  case arcade::CommandType::GO_DOWN:
244  const_cast<Pacman *>(&pacman.getPacman())->goDown(NULL);
245  break;
246  case arcade::CommandType::GO_LEFT:
247  const_cast<Pacman *>(&pacman.getPacman())->goLeft(NULL);
248  break;
249  case arcade::CommandType::GO_RIGHT:
250  const_cast<Pacman *>(&pacman.getPacman())->goRight(NULL);
251  break;
252  case arcade::CommandType::PLAY:
253  pacman.compute(-1);
254  break;
255  default:
256  break;
257  }
258  }
259  free(map);
260 }
