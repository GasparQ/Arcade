49 /*
50  Constant Definitions
51  ====================
52 
53  MAXVOL_FACTOR (was BITSHIFT in virtch.c)
54  Controls the maximum volume of the output data. All mixed data is
55  divided by this number after mixing, so larger numbers result in
56  quieter mixing. Smaller numbers will increase the likeliness of
57  distortion on loud modules.
58 
59  REVERBERATION
60  Larger numbers result in shorter reverb duration. Longer reverb
61  durations can cause unwanted static and make the reverb sound more
62  like a crappy echo.
63 
64  SAMPLING_SHIFT
65  Specified the shift multiplier which controls by how much the mixing
66  rate is multiplied while mixing. Higher values can improve quality by
67  smoothing the sound and reducing pops and clicks. Note, this is a shift
68  value, so a value of 2 becomes a mixing-rate multiplier of 4, and a
69  value of 3 = 8, etc.
70 
71  FRACBITS
72  The number of bits per integer devoted to the fractional part of the
73  number. Generally, this number should not be changed for any reason.
74 
75  !!! IMPORTANT !!! All values below MUST ALWAYS be greater than 0
76 
77 */
78

153 static SLONG Mix32MonoNormal(SWORD* srce,SLONG* dest,SLONG index,SLONG increment,SLONG todo)
154 {
155  SWORD sample=0;
156  SLONG i,f;
157 
158  while(todo--) {
159  i=index>>FRACBITS,f=index&FRACMASK;
160  sample=(((SLONG)(srce[i]*(FRACMASK+1L-f)) +
161  ((SLONG)srce[i+1]*f)) >> FRACBITS);
162  index+=increment;
163 
164  if(vnf->rampvol) {
165  *dest++ += (long)(
166  ( ( (SLONG)(vnf->oldlvol*vnf->rampvol) +
167  (vnf->lvolsel*(CLICK_BUFFER-vnf->rampvol)) ) *
168  (SLONG)sample ) >> CLICK_SHIFT );
169  vnf->rampvol--;
170  } else
171  if(vnf->click) {
172  *dest++ += (long)(
173  ( ( ((SLONG)vnf->lvolsel*(CLICK_BUFFER-vnf->click)) *
174  (SLONG)sample ) +
175  (vnf->lastvalL*vnf->click) ) >> CLICK_SHIFT );
176  vnf->click--;
177  } else
178  *dest++ +=vnf->lvolsel*sample;
179  }
180  vnf->lastvalL=vnf->lvolsel * sample;
181 
182  return index;
183 }

553 static void AddChannel(SLONG* ptr,NATIVE todo)
554 {
555  SLONGLONG end,done;
556  SWORD *s;
557 
558  if(!(s=Samples[vnf->handle])) {
559  vnf->current = vnf->active = 0;
560  vnf->lastvalL = vnf->lastvalR = 0;
561  return;
562  }
563 
564  /* update the 'current' index so the sample loops, or stops playing if it
565  reached the end of the sample */
566  while(todo>0) {
567  SLONGLONG endpos;
568 
569  if(vnf->flags & SF_REVERSE) {
570  /* The sample is playing in reverse */
571  if((vnf->flags&SF_LOOP)&&(vnf->current<idxlpos)) {
572  /* the sample is looping and has reached the loopstart index */
573  if(vnf->flags & SF_BIDI) {
574  /* sample is doing bidirectional loops, so 'bounce' the
575  current index against the idxlpos */
576  vnf->current = idxlpos+(idxlpos-vnf->current);
577  vnf->flags &= ~SF_REVERSE;
578  vnf->increment = -vnf->increment;
579  } else
580  /* normal backwards looping, so set the current position to
581  loopend index */
582  vnf->current=idxlend-(idxlpos-vnf->current);
583  } else {
584  /* the sample is not looping, so check if it reached index 0 */
585  if(vnf->current < 0) {
586  /* playing index reached 0, so stop playing this sample */
587  vnf->current = vnf->active = 0;
588  break;
589  }
590  }
591  } else {
592  /* The sample is playing forward */
593  if((vnf->flags & SF_LOOP) &&
594  (vnf->current >= idxlend)) {
595  /* the sample is looping, check the loopend index */
596  if(vnf->flags & SF_BIDI) {
597  /* sample is doing bidirectional loops, so 'bounce' the
598  current index against the idxlend */
599  vnf->flags |= SF_REVERSE;
600  vnf->increment = -vnf->increment;
601  vnf->current = idxlend-(vnf->current-idxlend);
602  } else
603  /* normal backwards looping, so set the current position
604  to loopend index */
605  vnf->current=idxlpos+(vnf->current-idxlend);
606  } else {
607  /* sample is not looping, so check if it reached the last
608  position */
609  if(vnf->current >= idxsize) {
610  /* yes, so stop playing this sample */
611  vnf->current = vnf->active = 0;
612  break;
613  }
614  }
615  }
616 
617  end=(vnf->flags&SF_REVERSE)?(vnf->flags&SF_LOOP)?idxlpos:0:
618  (vnf->flags&SF_LOOP)?idxlend:idxsize;
619 
620  /* if the sample is not blocked... */
621  if((end==vnf->current)||(!vnf->increment))
622  done=0;
623  else {
624  done=MIN((end-vnf->current)/vnf->increment+1,todo);
625  if(done<0) done=0;
626  }
627 
628  if(!done) {
629  vnf->active = 0;
630  break;
631  }
632 
633  endpos=vnf->current+done*vnf->increment;
634 
635  if(vnf->vol || vnf->rampvol) {
636 #ifndef NATIVE_64BIT_INT
637  /* use the 32 bit mixers as often as we can (they're much faster) */
638  if((vnf->current<0x7fffffff)&&(endpos<0x7fffffff)) {
639  if(vc_mode & DMODE_STEREO) {
640  if((vnf->pan==PAN_SURROUND)&&(vc_mode&DMODE_SURROUND))
641  vnf->current=Mix32StereoSurround
642  (s,ptr,vnf->current,vnf->increment,done);
643  else
644  vnf->current=Mix32StereoNormal
645  (s,ptr,vnf->current,vnf->increment,done);
646  } else
647  vnf->current=Mix32MonoNormal
648  (s,ptr,vnf->current,vnf->increment,done);
649  } else
650 #endif
651  {
652  if(vc_mode & DMODE_STEREO) {
653  if((vnf->pan==PAN_SURROUND)&&(vc_mode&DMODE_SURROUND))
654  vnf->current=MixStereoSurround
655  (s,ptr,vnf->current,vnf->increment,done);
656  else
657  vnf->current=MixStereoNormal
658  (s,ptr,vnf->current,vnf->increment,done);
659  } else
660  vnf->current=MixMonoNormal
661  (s,ptr,vnf->current,vnf->increment,done);
662  }
663  } else {
664  vnf->lastvalL = vnf->lastvalR = 0;
665  /* update sample position */
666  vnf->current=endpos;
667  }

670  ptr +=(vc_mode & DMODE_STEREO)?(done<<1):done;

699 void VC2_WriteSamples(SBYTE* buf,ULONG todo)
700 {
701  int left,portion=0;
702  SBYTE *buffer;
703  int t,pan,vol;
704 
705  todo*=SAMPLING_FACTOR;
706 
707  while(todo) {
708  if(!tickleft) {
709  if(vc_mode & DMODE_SOFT_MUSIC) md_player();
710  tickleft=(md_mixfreq*125L*SAMPLING_FACTOR)/(md_bpm*50L);
711  tickleft&=~(SAMPLING_FACTOR-1);
712  }
713  left = MIN(tickleft, todo);
714  buffer = buf;
715  tickleft -= left;
716  todo -= left;
717  buf += samples2bytes(left)/SAMPLING_FACTOR;
718 
719  while(left) {
720  portion = MIN(left, samplesthatfit);
721  memset(vc_tickbuf,0,portion<<((vc_mode&DMODE_STEREO)?3:2));
722  for(t=0;t<vc_softchn;t++) {
723  vnf = &vinf[t];
724 
725  if(vnf->kick) {
726  vnf->current=((SLONGLONG)(vnf->start))<<FRACBITS;
727  vnf->kick = 0;
728  vnf->active = 1;
729  vnf->click = CLICK_BUFFER;
730  vnf->rampvol = 0;
731  }
732 
733  if(!vnf->frq) vnf->active = 0;
734 
735  if(vnf->active) {
736  vnf->increment=((SLONGLONG)(vnf->frq)<<(FRACBITS-SAMPLING_SHIFT))
737  /md_mixfreq;
738  if(vnf->flags&SF_REVERSE) vnf->increment=-vnf->increment;
739  vol = vnf->vol; pan = vnf->pan;
740 
741  vnf->oldlvol=vnf->lvolsel;vnf->oldrvol=vnf->rvolsel;
742  if(vc_mode & DMODE_STEREO) {
743  if(pan!=PAN_SURROUND) {
744  vnf->lvolsel=(vol*(PAN_RIGHT-pan))>>8;
745  vnf->rvolsel=(vol*pan)>>8;
746  } else {
747  vnf->lvolsel=vnf->rvolsel=(vol * 256L) / 480;
748  }
749  } else
750  vnf->lvolsel=vol;
751 
752  idxsize=(vnf->size)?((SLONGLONG)(vnf->size)<<FRACBITS)-1:0;
753  idxlend=(vnf->repend)?((SLONGLONG)(vnf->repend)<<FRACBITS)-1:0;
754  idxlpos=(SLONGLONG)(vnf->reppos)<<FRACBITS;
755  AddChannel(vc_tickbuf,portion);
756  }
757  }
758 
759  if(md_reverb) {
760  if(md_reverb>15) md_reverb=15;
761  MixReverb(vc_tickbuf,portion);
762  }
763 
764  if(vc_mode & DMODE_16BITS)
765  Mix32to16((SWORD*)buffer,vc_tickbuf,portion);
766  else
767  Mix32to8((SBYTE*)buffer,vc_tickbuf,portion);
768 
769  buffer += samples2bytes(portion) / SAMPLING_FACTOR;
770  left -= portion;
771  }
772  }
773 }
