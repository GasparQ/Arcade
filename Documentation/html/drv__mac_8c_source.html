56 static pascal void MyDoubleBackProc(SndChannelPtr soundChannel,SndDoubleBufferPtr doubleBuffer)
57 {
58 #pragma unused(soundChannel)
59  SBYTE *buffer;
60  long written;
61 #if TARGET_CPU_68K
62  long oldA5=SetA5(doubleBuffer->dbUserInfo[0]);
63 #endif
64 
65  buffer=(SBYTE*)doubleBuffer->dbSoundData;
66 
67  if (Player_Paused()) {
68  MUTEX_LOCK
69  written=VC_SilenceBytes(buffer,(ULONG)SOUND_BUFFER_SIZE);
70  MUTEX_UNLOCK
71  } else {
72  MUTEX_LOCK
73  written=VC_WriteBytes(buffer,(ULONG)SOUND_BUFFER_SIZE);
74  MUTEX_UNLOCK
75  }
76 
77  if (doubleHeader.dbhNumChannels==2) written>>=1;
78  if (doubleHeader.dbhSampleSize==16) written>>=1;
79 
80  doubleBuffer->dbNumFrames=written;
81  doubleBuffer->dbFlags|=dbBufferReady;
82 
83 #if TARGET_CPU_68K
84  SetA5(oldA5);
85 #endif
86 }

99 static BOOL MAC_Init(void)
100 {
101  OSErr err,iErr;
102  int i;
103  SndDoubleBufferPtr doubleBuffer;
104  long rate,maxrate,maxbits;
105  long gestaltAnswer;
106  NumVersion nVers;
107  Boolean Stereo,StereoMixing,Audio16;
108  Boolean NewSoundManager,NewSoundManager31;
109 
110  NewSoundManager31=NewSoundManager=false;
111 
112  nVers=SndSoundManagerVersion();
113  if (nVers.majorRev>=3) {
114  NewSoundManager=true;
115  if (nVers.minorAndBugRev>=0x10)
116  NewSoundManager31=true;
117  } else
118  if (nVers.majorRev<2)
119  return 1; /* failure, need SoundManager 2.0+ for SndPlayDoubleBuffer */
120 
121  iErr=Gestalt(gestaltSoundAttr,&gestaltAnswer);
122  if (iErr==noErr) {
123  Stereo=(gestaltAnswer & (1<<gestaltStereoCapability))!=0;
124  StereoMixing=(gestaltAnswer & (1<<gestaltStereoMixing))!=0;
125  Audio16=(gestaltAnswer & (1<<gestalt16BitSoundIO))!=0;
126  } else {
127  /* failure, couldn't get any sound info at all ? */
128  Stereo=StereoMixing=Audio16=false;
129  }
130 
131 #if !GENERATING68K || !GENERATINGCFM
132  if (NewSoundManager31) {
133  iErr=GetSoundOutputInfo(0L,siSampleRate,(void*)&maxrate);
134  if (iErr==noErr)
135  iErr=GetSoundOutputInfo(0L,siSampleSize,(void*)&maxbits);
136  }
137 
138  if (iErr!=noErr) {
139 #endif
140  maxrate=rate22khz;
141 
142  if (NewSoundManager && Audio16)
143  maxbits=16;
144  else
145  maxbits=8;
146 #if !GENERATING68K || !GENERATINGCFM
147  }
148 #endif
149 
150  switch (md_mixfreq) {
151  case 48000:rate=rate48khz;break;
152  case 44100:rate=rate44khz;break;
153  case 22254:rate=rate22khz;break;
154  case 22050:rate=rate22050hz;break;
155  case 11127:rate=rate11khz;break;
156  case 11025:rate=rate11025hz;break;
157  default: rate=0;break;
158  }
159 
160  if (!rate) {
161  _mm_errno=MMERR_MAC_SPEED;
162  return 1;
163  }
164 
165  md_mode|=DMODE_SOFT_MUSIC|DMODE_SOFT_SNDFX;
166 
167  if ((md_mode&DMODE_16BITS)&&(maxbits<16))
168  md_mode&=~DMODE_16BITS;
169 
170  if (!Stereo || !StereoMixing)
171  md_mode&=~DMODE_STEREO;
172 
173  if (rate>maxrate)
174  rate=maxrate;
175  if (md_mixfreq>(maxrate>>16))
176  md_mixfreq=maxrate>>16;
177 
178  err=SndNewChannel(&soundChannel,sampledSynth,
179  (md_mode&DMODE_STEREO)?initStereo:initMono,NULL);
180  if(err!=noErr) {
181  _mm_errno=MMERR_OPENING_AUDIO;
182  return 1;
183  }
184 
185  doubleHeader.dbhCompressionID=0;
186  doubleHeader.dbhPacketSize =0;
187  doubleHeader.dbhSampleRate =rate;
188  doubleHeader.dbhSampleSize =(md_mode&DMODE_16BITS)?16:8;
189  doubleHeader.dbhNumChannels =(md_mode&DMODE_STEREO)?2:1;
190  doubleHeader.dbhDoubleBack =NewSndDoubleBackProc(&MyDoubleBackProc);
191 
192  for(i=0;i<2;i++) {
193  doubleBuffer=(SndDoubleBufferPtr)NewPtrClear(sizeof(SndDoubleBuffer)+
194  SOUND_BUFFER_SIZE);
195  if(!doubleBuffer) {
196  _mm_errno=MMERR_OUT_OF_MEMORY;
197  return 1;
198  }
199 
200  doubleBuffer->dbNumFrames=0;
201  doubleBuffer->dbFlags=0;
202  doubleBuffer->dbUserInfo[0]=SetCurrentA5();
203  doubleBuffer->dbUserInfo[1]=0;
204 
205  doubleHeader.dbhBufferPtr[i]=doubleBuffer;
206  }
207 
208  return VC_Init();
209 }
