67 BOOL CSoundFile::ReadPTM(const BYTE *lpStream, DWORD dwMemLength)
68 //---------------------------------------------------------------
69 {
70  DWORD dwMemPos;
71  UINT nOrders;
72 
73  if ((!lpStream) || (dwMemLength < sizeof(PTMFILEHEADER))) return FALSE;
74  PTMFILEHEADER pfh = *(LPPTMFILEHEADER)lpStream;
75 
76  pfh.norders = bswapLE16(pfh.norders);
77  pfh.nsamples = bswapLE16(pfh.nsamples);
78  pfh.npatterns = bswapLE16(pfh.npatterns);
79  pfh.nchannels = bswapLE16(pfh.nchannels);
80  pfh.fileflags = bswapLE16(pfh.fileflags);
81  pfh.reserved2 = bswapLE16(pfh.reserved2);
82  pfh.ptmf_id = bswapLE32(pfh.ptmf_id);
83  for (UINT j=0; j<128; j++)
84  {
85  pfh.patseg[j] = bswapLE16(pfh.patseg[j]);
86  }
87 
88  if ((pfh.ptmf_id != 0x464d5450) || (!pfh.nchannels)
89  || (pfh.nchannels > 32)
90  || (pfh.norders > 256) || (!pfh.norders)
91  || (!pfh.nsamples) || (pfh.nsamples > 255)
92  || (!pfh.npatterns) || (pfh.npatterns > 128)
93  || (SIZEOF_PTMFILEHEADER+pfh.nsamples*SIZEOF_PTMSAMPLE >= (int)dwMemLength)) return FALSE;
94  memcpy(m_szNames[0], pfh.songname, 28);
95  m_szNames[0][28] = 0;
96  m_nType = MOD_TYPE_PTM;
97  m_nChannels = pfh.nchannels;
98  m_nSamples = (pfh.nsamples < MAX_SAMPLES) ? pfh.nsamples : MAX_SAMPLES-1;
99  dwMemPos = SIZEOF_PTMFILEHEADER;
100  nOrders = (pfh.norders < MAX_ORDERS) ? pfh.norders : MAX_ORDERS-1;
101  memcpy(Order, pfh.orders, nOrders);
102  for (UINT ipan=0; ipan<m_nChannels; ipan++)
103  {
104  ChnSettings[ipan].nVolume = 64;
105  ChnSettings[ipan].nPan = ((pfh.chnpan[ipan] & 0x0F) << 4) + 4;
106  }
107  for (UINT ismp=0; ismp<m_nSamples; ismp++, dwMemPos += SIZEOF_PTMSAMPLE)
108  {
109  MODINSTRUMENT *pins = &Ins[ismp+1];
110  PTMSAMPLE *psmp = (PTMSAMPLE *)(lpStream+dwMemPos);
111 
112  lstrcpyn(m_szNames[ismp+1], psmp->samplename, 28);
113  memcpy(pins->name, psmp->filename, 12);
114  pins->name[12] = 0;
115  pins->nGlobalVol = 64;
116  pins->nPan = 128;
117  pins->nVolume = psmp->volume << 2;
118  pins->nC4Speed = bswapLE16(psmp->nC4Spd) << 1;
119  pins->uFlags = 0;
120  if ((psmp->sampletype & 3) == 1)
121  {
122  UINT smpflg = RS_PCM8D;
123  pins->nLength = BS2WORD(psmp->length);
124  pins->nLoopStart = BS2WORD(psmp->loopbeg);
125  pins->nLoopEnd = BS2WORD(psmp->loopend);
126  DWORD samplepos = BS2WORD(psmp->fileofs);
127  if (psmp->sampletype & 4) pins->uFlags |= CHN_LOOP;
128  if (psmp->sampletype & 8) pins->uFlags |= CHN_PINGPONGLOOP;
129  if (psmp->sampletype & 16)
130  {
131  pins->uFlags |= CHN_16BIT;
132  pins->nLength >>= 1;
133  pins->nLoopStart >>= 1;
134  pins->nLoopEnd >>= 1;
135  smpflg = RS_PTM8DTO16;
136  }
137  if ((pins->nLength) && (samplepos) && (samplepos < dwMemLength))
138  {
139  ReadSample(pins, smpflg, (LPSTR)(lpStream+samplepos), dwMemLength-samplepos);
140  }
141  }
142  }
143  // Reading Patterns
144  for (UINT ipat=0; ipat<pfh.npatterns; ipat++)
145  {
146  dwMemPos = ((UINT)pfh.patseg[ipat]) << 4;
147  if ((!dwMemPos) || (dwMemPos >= dwMemLength)) continue;
148  PatternSize[ipat] = 64;
149  if ((Patterns[ipat] = AllocatePattern(64, m_nChannels)) == NULL) break;
150  //
151  MODCOMMAND *m = Patterns[ipat];
152  for (UINT row=0; ((row < 64) && (dwMemPos < dwMemLength)); )
153  {
154  UINT b = lpStream[dwMemPos++];
155 
156  if (dwMemPos >= dwMemLength) break;
157  if (b)
158  {
159  UINT nChn = b & 0x1F;
160 
161  if (b & 0x20)
162  {
163  if (dwMemPos + 2 > dwMemLength) break;
164  m[nChn].note = lpStream[dwMemPos++];
165  m[nChn].instr = lpStream[dwMemPos++];
166  }
167  if (b & 0x40)
168  {
169  if (dwMemPos + 2 > dwMemLength) break;
170  m[nChn].command = lpStream[dwMemPos++];
171  m[nChn].param = lpStream[dwMemPos++];
172  if ((m[nChn].command == 0x0E) && ((m[nChn].param & 0xF0) == 0x80))
173  {
174  m[nChn].command = CMD_S3MCMDEX;
175  } else
176  if (m[nChn].command < 0x10)
177  {
178  ConvertModCommand(&m[nChn]);
179  } else
180  {
181  switch(m[nChn].command)
182  {
183  case 16:
184  m[nChn].command = CMD_GLOBALVOLUME;
185  break;
186  case 17:
187  m[nChn].command = CMD_RETRIG;
188  break;
189  case 18:
190  m[nChn].command = CMD_FINEVIBRATO;
191  break;
192  default:
193  m[nChn].command = 0;
194  }
195  }
196  }
197  if (b & 0x80)
198  {
199  if (dwMemPos >= dwMemLength) break;
200  m[nChn].volcmd = VOLCMD_VOLUME;
201  m[nChn].vol = lpStream[dwMemPos++];
202  }
203  } else
204  {
205  row++;
206  m += m_nChannels;
207  }
208  }
209  }
210  return TRUE;
211 }
