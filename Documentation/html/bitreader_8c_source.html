183 static FLaC__INLINE void crc16_update_word_(FLAC__BitReader *br, brword word)
184 {
185  register unsigned crc = br->read_crc16;
186 #if FLAC__BYTES_PER_WORD == 4
187  switch(br->crc16_align) {
188  case 0: crc = FLAC__CRC16_UPDATE((unsigned)(word >> 24), crc);
189  case 8: crc = FLAC__CRC16_UPDATE((unsigned)((word >> 16) & 0xff), crc);
190  case 16: crc = FLAC__CRC16_UPDATE((unsigned)((word >> 8) & 0xff), crc);
191  case 24: br->read_crc16 = FLAC__CRC16_UPDATE((unsigned)(word & 0xff), crc);
192  }
193 #elif FLAC__BYTES_PER_WORD == 8
194  switch(br->crc16_align) {
195  case 0: crc = FLAC__CRC16_UPDATE((unsigned)(word >> 56), crc);
196  case 8: crc = FLAC__CRC16_UPDATE((unsigned)((word >> 48) & 0xff), crc);
197  case 16: crc = FLAC__CRC16_UPDATE((unsigned)((word >> 40) & 0xff), crc);
198  case 24: crc = FLAC__CRC16_UPDATE((unsigned)((word >> 32) & 0xff), crc);
199  case 32: crc = FLAC__CRC16_UPDATE((unsigned)((word >> 24) & 0xff), crc);
200  case 40: crc = FLAC__CRC16_UPDATE((unsigned)((word >> 16) & 0xff), crc);
201  case 48: crc = FLAC__CRC16_UPDATE((unsigned)((word >> 8) & 0xff), crc);
202  case 56: br->read_crc16 = FLAC__CRC16_UPDATE((unsigned)(word & 0xff), crc);
203  }
204 #else
205  for( ; br->crc16_align < FLAC__BITS_PER_WORD; br->crc16_align += 8)
206  crc = FLAC__CRC16_UPDATE((unsigned)((word >> (FLAC__BITS_PER_WORD-8-br->crc16_align)) & 0xff), crc);
207  br->read_crc16 = crc;
208 #endif
209  br->crc16_align = 0;
210 }

213 FLAC__bool bitreader_read_from_client_(FLAC__BitReader *br)
214 {
215  unsigned start, end;
216  size_t bytes;
217  FLAC__byte *target;
218 
219  /* first shift the unconsumed buffer data toward the front as much as possible */
220  if(br->consumed_words > 0) {
221  start = br->consumed_words;
222  end = br->words + (br->bytes? 1:0);
223  memmove(br->buffer, br->buffer+start, FLAC__BYTES_PER_WORD * (end - start));
224 
225  br->words -= start;
226  br->consumed_words = 0;
227  }
228 
229  /*
230  * set the target for reading, taking into account word alignment and endianness
231  */
232  bytes = (br->capacity - br->words) * FLAC__BYTES_PER_WORD - br->bytes;
233  if(bytes == 0)
234  return false; /* no space left, buffer is too small; see note for FLAC__BITREADER_DEFAULT_CAPACITY */
235  target = ((FLAC__byte*)(br->buffer+br->words)) + br->bytes;
236 
237  /* before reading, if the existing reader looks like this (say brword is 32 bits wide)
238  * bitstream : 11 22 33 44 55 br->words=1 br->bytes=1 (partial tail word is left-justified)
239  * buffer[BE]: 11 22 33 44 55 ?? ?? ?? (shown layed out as bytes sequentially in memory)
240  * buffer[LE]: 44 33 22 11 ?? ?? ?? 55 (?? being don't-care)
241  * ^^-------target, bytes=3
242  * on LE machines, have to byteswap the odd tail word so nothing is
243  * overwritten:
244  */
245 #if WORDS_BIGENDIAN
246 #else
247  if(br->bytes)
248  br->buffer[br->words] = SWAP_BE_WORD_TO_HOST(br->buffer[br->words]);
249 #endif
250 
251  /* now it looks like:
252  * bitstream : 11 22 33 44 55 br->words=1 br->bytes=1
253  * buffer[BE]: 11 22 33 44 55 ?? ?? ??
254  * buffer[LE]: 44 33 22 11 55 ?? ?? ??
255  * ^^-------target, bytes=3
256  */
257 
258  /* read in the data; note that the callback may return a smaller number of bytes */
259  if(!br->read_callback(target, &bytes, br->client_data))
260  return false;
261 
262  /* after reading bytes 66 77 88 99 AA BB CC DD EE FF from the client:
263  * bitstream : 11 22 33 44 55 66 77 88 99 AA BB CC DD EE FF
264  * buffer[BE]: 11 22 33 44 55 66 77 88 99 AA BB CC DD EE FF ??
265  * buffer[LE]: 44 33 22 11 55 66 77 88 99 AA BB CC DD EE FF ??
266  * now have to byteswap on LE machines:
267  */
268 #if WORDS_BIGENDIAN
269 #else
270  end = (br->words*FLAC__BYTES_PER_WORD + br->bytes + bytes + (FLAC__BYTES_PER_WORD-1)) / FLAC__BYTES_PER_WORD;
271 # if defined(_MSC_VER) && (FLAC__BYTES_PER_WORD == 4)
272  if(br->cpu_info.type == FLAC__CPUINFO_TYPE_IA32 && br->cpu_info.data.ia32.bswap) {
273  start = br->words;
274  local_swap32_block_(br->buffer + start, end - start);
275  }
276  else
277 # endif
278  for(start = br->words; start < end; start++)
279  br->buffer[start] = SWAP_BE_WORD_TO_HOST(br->buffer[start]);
280 #endif
281 
282  /* now it looks like:
283  * bitstream : 11 22 33 44 55 66 77 88 99 AA BB CC DD EE FF
284  * buffer[BE]: 11 22 33 44 55 66 77 88 99 AA BB CC DD EE FF ??
285  * buffer[LE]: 44 33 22 11 88 77 66 55 CC BB AA 99 ?? FF EE DD
286  * finally we'll update the reader values:
287  */
288  end = br->words*FLAC__BYTES_PER_WORD + br->bytes + bytes;
289  br->words = end / FLAC__BYTES_PER_WORD;
290  br->bytes = end % FLAC__BYTES_PER_WORD;
291 
292  return true;
293 }

440 FLaC__INLINE FLAC__bool FLAC__bitreader_read_raw_uint32(FLAC__BitReader *br, FLAC__uint32 *val, unsigned bits)
441 {
442  FLAC__ASSERT(0 != br);
443  FLAC__ASSERT(0 != br->buffer);
444 
445  FLAC__ASSERT(bits <= 32);
446  FLAC__ASSERT((br->capacity*FLAC__BITS_PER_WORD) * 2 >= bits);
447  FLAC__ASSERT(br->consumed_words <= br->words);
448 
449  /* WATCHOUT: code does not work with <32bit words; we can make things much faster with this assertion */
450  FLAC__ASSERT(FLAC__BITS_PER_WORD >= 32);
451 
452  if(bits == 0) { /* OPT: investigate if this can ever happen, maybe change to assertion */
453  *val = 0;
454  return true;
455  }
456 
457  while((br->words-br->consumed_words)*FLAC__BITS_PER_WORD + br->bytes*8 - br->consumed_bits < bits) {
458  if(!bitreader_read_from_client_(br))
459  return false;
460  }
461  if(br->consumed_words < br->words) { /* if we've not consumed up to a partial tail word... */
462  /* OPT: taking out the consumed_bits==0 "else" case below might make things faster if less code allows the compiler to inline this function */
463  if(br->consumed_bits) {
464  /* this also works when consumed_bits==0, it's just a little slower than necessary for that case */
465  const unsigned n = FLAC__BITS_PER_WORD - br->consumed_bits;
466  const brword word = br->buffer[br->consumed_words];
467  if(bits < n) {
468  *val = (word & (FLAC__WORD_ALL_ONES >> br->consumed_bits)) >> (n-bits);
469  br->consumed_bits += bits;
470  return true;
471  }
472  *val = word & (FLAC__WORD_ALL_ONES >> br->consumed_bits);
473  bits -= n;
474  crc16_update_word_(br, word);
475  br->consumed_words++;
476  br->consumed_bits = 0;
477  if(bits) { /* if there are still bits left to read, there have to be less than 32 so they will all be in the next word */
478  *val <<= bits;
479  *val |= (br->buffer[br->consumed_words] >> (FLAC__BITS_PER_WORD-bits));
480  br->consumed_bits = bits;
481  }
482  return true;
483  }
484  else {
485  const brword word = br->buffer[br->consumed_words];
486  if(bits < FLAC__BITS_PER_WORD) {
487  *val = word >> (FLAC__BITS_PER_WORD-bits);
488  br->consumed_bits = bits;
489  return true;
490  }
491  /* at this point 'bits' must be == FLAC__BITS_PER_WORD; because of previous assertions, it can't be larger */
492  *val = word;
493  crc16_update_word_(br, word);
494  br->consumed_words++;
495  return true;
496  }
497  }
498  else {
499  /* in this case we're starting our read at a partial tail word;
500  * the reader has guaranteed that we have at least 'bits' bits
501  * available to read, which makes this case simpler.
502  */
503  /* OPT: taking out the consumed_bits==0 "else" case below might make things faster if less code allows the compiler to inline this function */
504  if(br->consumed_bits) {
505  /* this also works when consumed_bits==0, it's just a little slower than necessary for that case */
506  FLAC__ASSERT(br->consumed_bits + bits <= br->bytes*8);
507  *val = (br->buffer[br->consumed_words] & (FLAC__WORD_ALL_ONES >> br->consumed_bits)) >> (FLAC__BITS_PER_WORD-br->consumed_bits-bits);
508  br->consumed_bits += bits;
509  return true;
510  }
511  else {
512  *val = br->buffer[br->consumed_words] >> (FLAC__BITS_PER_WORD-bits);
513  br->consumed_bits += bits;
514  return true;
515  }
516  }
