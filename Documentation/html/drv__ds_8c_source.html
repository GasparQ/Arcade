81 static DWORD WINAPI updateBufferProc(LPVOID lpParameter)
82 {
83  LPVOID pBlock1 = NULL, pBlock2 = NULL;
84  DWORD soundBufferCurrentPosition, blockBytes1, blockBytes2;
85  DWORD start;
86 
87  while (threadInUse) {
88  if (WaitForSingleObject(notifyUpdateHandle,INFINITE)==WAIT_OBJECT_0) {
89 
90  if (!threadInUse) break;
91 
92  pSoundBuffer->lpVtbl->GetCurrentPosition
93  (pSoundBuffer,&soundBufferCurrentPosition,NULL);
94 
95  if (soundBufferCurrentPosition < fragsize)
96  start = fragsize;
97  else
98  start = 0;
99 
100  if (pSoundBuffer->lpVtbl->Lock
101  (pSoundBuffer,start,fragsize,&pBlock1,&blockBytes1,
102  &pBlock2,&blockBytes2,0)==DSERR_BUFFERLOST) {
103  pSoundBuffer->lpVtbl->Restore(pSoundBuffer);
104  pSoundBuffer->lpVtbl->Lock
105  (pSoundBuffer,start,fragsize,&pBlock1,&blockBytes1,
106  &pBlock2,&blockBytes2,0);
107  }
108 
109  MUTEX_LOCK(vars);
110  if (Player_Paused_internal()) {
111  VC_SilenceBytes((SBYTE*)pBlock1,(ULONG)blockBytes1);
112  if (pBlock2)
113  VC_SilenceBytes((SBYTE*)pBlock2,(ULONG)blockBytes2);
114  } else {
115  VC_WriteBytes((SBYTE*)pBlock1,(ULONG)blockBytes1);
116  if (pBlock2)
117  VC_WriteBytes((SBYTE*)pBlock2,(ULONG)blockBytes2);
118  }
119  MUTEX_UNLOCK(vars);
120 
121  pSoundBuffer->lpVtbl->Unlock
122  (pSoundBuffer,pBlock1,blockBytes1,pBlock2,blockBytes2);
123  }
124  }
125  return 0;
126 }

156 static BOOL DS_Init(void)
157 {
158  DSBUFFERDESC soundBufferFormat;
159  WAVEFORMATEX pcmwf;
160  DSBPOSITIONNOTIFY positionNotifications[2];
161  DWORD updateBufferThreadID;
162 
163  if (DirectSoundCreate(NULL,&pSoundCard,NULL)!=DS_OK) {
164  _mm_errno=MMERR_OPENING_AUDIO;
165  return 1;
166  }
167 
168  if (pSoundCard->lpVtbl->SetCooperativeLevel
169  (pSoundCard,GetForegroundWindow(),DSSCL_PRIORITY)!=DS_OK) {
170  _mm_errno=MMERR_DS_PRIORITY;
171  return 1;
172  }
173 
174  memset(&soundBufferFormat,0,sizeof(DSBUFFERDESC));
175  soundBufferFormat.dwSize =sizeof(DSBUFFERDESC);
176  soundBufferFormat.dwFlags =DSBCAPS_PRIMARYBUFFER;
177  soundBufferFormat.dwBufferBytes=0;
178  soundBufferFormat.lpwfxFormat =NULL;
179 
180  if (pSoundCard->lpVtbl->CreateSoundBuffer
181  (pSoundCard,&soundBufferFormat,&pPrimarySoundBuffer,NULL)!=DS_OK) {
182  _mm_errno=MMERR_DS_BUFFER;
183  return 1;
184  }
185 
186  memset(&pcmwf,0,sizeof(WAVEFORMATEX));
187  pcmwf.wFormatTag =WAVE_FORMAT_PCM;
188  pcmwf.nChannels =(md_mode&DMODE_STEREO)?2:1;
189  pcmwf.nSamplesPerSec =md_mixfreq;
190  pcmwf.wBitsPerSample =(md_mode&DMODE_16BITS)?16:8;
191  pcmwf.nBlockAlign =(pcmwf.wBitsPerSample * pcmwf.nChannels) / 8;
192  pcmwf.nAvgBytesPerSec=pcmwf.nSamplesPerSec*pcmwf.nBlockAlign;
193 
194  if (pPrimarySoundBuffer->lpVtbl->SetFormat
195  (pPrimarySoundBuffer,&pcmwf)!=DS_OK) {
196  _mm_errno=MMERR_DS_FORMAT;
197  return 1;
198  }
199  pPrimarySoundBuffer->lpVtbl->Play(pPrimarySoundBuffer,0,0,DSBPLAY_LOOPING);
200 
201  memset(&soundBufferFormat,0,sizeof(DSBUFFERDESC));
202  soundBufferFormat.dwSize =sizeof(DSBUFFERDESC);
203  soundBufferFormat.dwFlags =controlflags|DSBCAPS_GETCURRENTPOSITION2 ;
204  soundBufferFormat.dwBufferBytes=fragsize*UPDATES;
205  soundBufferFormat.lpwfxFormat =&pcmwf;
206 
207  if (pSoundCard->lpVtbl->CreateSoundBuffer
208  (pSoundCard,&soundBufferFormat,&pSoundBuffer,NULL)!=DS_OK) {
209  _mm_errno=MMERR_DS_BUFFER;
210  return 1;
211  }
212 
213  pSoundBuffer->lpVtbl->QueryInterface
214  (pSoundBuffer,&IID_IDirectSoundNotify,(LPVOID*)&pSoundBufferNotify);
215  if (!pSoundBufferNotify) {
216  _mm_errno=MMERR_DS_NOTIFY;
217  return 1;
218  }
219 
220  notifyUpdateHandle=CreateEvent
221  (NULL,FALSE,FALSE,"libmikmod DirectSound Driver positionNotify Event");
222  if (!notifyUpdateHandle) {
223  _mm_errno=MMERR_DS_EVENT;
224  return 1;
225  }
226 
227  updateBufferHandle=CreateThread
228  (NULL,0,updateBufferProc,NULL,CREATE_SUSPENDED,&updateBufferThreadID);
229  if (!updateBufferHandle) {
230  _mm_errno=MMERR_DS_THREAD;
231  return 1;
232  }
233 
234  memset(positionNotifications,0,2*sizeof(DSBPOSITIONNOTIFY));
235  positionNotifications[0].dwOffset =0;
236  positionNotifications[0].hEventNotify=notifyUpdateHandle;
237  positionNotifications[1].dwOffset =fragsize;
238  positionNotifications[1].hEventNotify=notifyUpdateHandle;
239  if (pSoundBufferNotify->lpVtbl->SetNotificationPositions
240  (pSoundBufferNotify,2,positionNotifications) != DS_OK) {
241  _mm_errno=MMERR_DS_UPDATE;
242  return 1;
243  }
244 
245  return VC_Init();
246 }

248 static void DS_Exit(void)
249 {
250  DWORD statusInfo;
251 
252  if(updateBufferHandle) {
253 
254  /* Signal thread to exit and wait for the exit */
255  if (threadInUse) {
256  threadInUse = 0;
257  MUTEX_UNLOCK(vars);
258  SetEvent (notifyUpdateHandle);
259  WaitForSingleObject (updateBufferHandle, INFINITE);
260  MUTEX_LOCK(vars);
261  }
262 
263  CloseHandle(updateBufferHandle),
264  updateBufferHandle = 0;
265  }
266  if (notifyUpdateHandle) {
267  CloseHandle(notifyUpdateHandle),
268  notifyUpdateHandle = 0;
269  }
270 
271  SAFE_RELEASE(pSoundBufferNotify);
272  if(pSoundBuffer) {
273  if(pSoundBuffer->lpVtbl->GetStatus(pSoundBuffer,&statusInfo)==DS_OK)
274  if(statusInfo&DSBSTATUS_PLAYING)
275  pSoundBuffer->lpVtbl->Stop(pSoundBuffer);
276  SAFE_RELEASE(pSoundBuffer);
277  }
278 
279  if(pPrimarySoundBuffer) {
280  if(pPrimarySoundBuffer->lpVtbl->GetStatus
281  (pPrimarySoundBuffer,&statusInfo)==DS_OK)
282  if(statusInfo&DSBSTATUS_PLAYING)
283  pPrimarySoundBuffer->lpVtbl->Stop(pPrimarySoundBuffer);
284  SAFE_RELEASE(pPrimarySoundBuffer);
285  }
286 
287  SAFE_RELEASE(pSoundCard);
288 
289  VC_Exit();
290 }

294 static void DS_Update(void)
295 {
296  LPVOID block;
297  DWORD bBytes;
298 
299  /* Do first update in DS_Update() to be consistent with other
300  non threaded drivers. */
301  if (do_update && pSoundBuffer) {
302  do_update = 0;
303 
304  if (pSoundBuffer->lpVtbl->Lock (pSoundBuffer, 0, fragsize,
305  &block, &bBytes, NULL, NULL, 0)
306  == DSERR_BUFFERLOST) {
307  pSoundBuffer->lpVtbl->Restore (pSoundBuffer);
308  pSoundBuffer->lpVtbl->Lock (pSoundBuffer, 0, fragsize,
309  &block, &bBytes, NULL, NULL, 0);
310  }
311 
312  if (Player_Paused_internal()) {
313  VC_SilenceBytes ((SBYTE *)block, (ULONG)bBytes);
314  } else {
315  VC_WriteBytes ((SBYTE *)block, (ULONG)bBytes);
316  }
317 
318  pSoundBuffer->lpVtbl->Unlock (pSoundBuffer, block, bBytes, NULL, 0);
319 
320  pSoundBuffer->lpVtbl->SetCurrentPosition(pSoundBuffer, 0);
321  pSoundBuffer->lpVtbl->Play(pSoundBuffer, 0, 0, DSBPLAY_LOOPING);
322 
323  threadInUse=1;
324  ResumeThread (updateBufferHandle);
325  }
326 }
