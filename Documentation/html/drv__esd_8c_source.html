98 static BOOL ESD_Link(void)
99 {
100  if (libesd) return 0;
101 
102  /* load libesd.so */
103  libesd=dlopen("libesd.so",RTLD_LAZY|RTLD_GLOBAL);
104  if (!libesd) return 1;
105 
106  /* resolve function references */
107  if (!(esd_closestream=dlsym(libesd,"esd_close"))) return 1;
108  if (!(esd_playstream=dlsym(libesd,"esd_play_stream"))) return 1;
109 
110  return 0;
111 }

128 static ssize_t esd_writebuf(int fd,const void *buffer,size_t count)
129 {
130  ssize_t start=0;
131 
132  while (start<count) {
133  ssize_t res;
134 
135  res=write(fd,(char*)buffer+start,count-start);
136  if (res<0) {
137  /* connection lost */
138  if (errno==EPIPE)
139  return -1-start;
140  } else
141  start+=res;
142  }
143  return start;
144 }

156 static BOOL ESD_IsThere(void)
157 {
158  int fd,retval;
159 
160 #ifdef MIKMOD_DYNAMIC
161  if (ESD_Link()) return 0;
162 #endif
163  /* Try to esablish a connection with default esd settings, but we don't
164  want esdlib to spawn esd if esd is not running ! */
165  if (SETENV)
166  retval=0;
167  else {
168  if ((fd=esd_playstream(ESD_BITS16|ESD_STEREO|ESD_STREAM|ESD_PLAY,
169  ESD_DEFAULT_RATE,espeaker,"libmikmod"))<0)
170  retval=0;
171  else {
172  esd_closestream(fd);
173  retval=1;
174  }
175  }
176 #ifdef MIKMOD_DYNAMIC
177  ESD_Unlink();
178 #endif
179  return retval;
180 }

182 static BOOL ESD_Init_internal(void)
183 {
184  format=(md_mode&DMODE_16BITS?ESD_BITS16:ESD_BITS8)|
185  (md_mode&DMODE_STEREO?ESD_STEREO:ESD_MONO)|ESD_STREAM|ESD_PLAY;
186 
187  if (md_mixfreq > ESD_DEFAULT_RATE)
188  md_mixfreq = ESD_DEFAULT_RATE;
189 
190  /* make sure we can open an esd stream with our parameters */
191  if (!(SETENV)) {
192  if ((sndfd=esd_playstream(format,md_mixfreq,espeaker,"libmikmod"))<0) {
193  _mm_errno=MMERR_OPENING_AUDIO;
194  return 1;
195  }
196  } else {
197  _mm_errno=MMERR_OUT_OF_MEMORY;
198  return 1;
199  }
200 
201  if (!(audiobuffer=(SBYTE*)_mm_malloc(ESD_BUF_SIZE*sizeof(char))))
202  return 1;
203

237 static void ESD_Update_internal(int count)
238 {
239  static time_t losttime;
240 
241  if (sndfd>=0) {
242  if (esd_writebuf(sndfd,audiobuffer,count)<0) {
243  /* if we lost our connection with esd, clean up and work as the
244  nosound driver until we can reconnect */
245  esd_closestream(sndfd);
246  sndfd=-1;
247  signal(SIGPIPE,SIG_DFL);
248  losttime=time(NULL);
249  }
250  } else {
251  /* an alarm would be better, but then the library user could not use
252  alarm(2) himself... */
253  if (time(NULL)-losttime>=5) {
254  losttime=time(NULL);
255  /* Attempt to reconnect every 5 seconds */
256  if (!(SETENV))
257  if ((sndfd=esd_playstream(format,md_mixfreq,espeaker,"libmikmod"))>=0) {
258  VC_SilenceBytes(audiobuffer,ESD_BUF_SIZE);
259  esd_writebuf(sndfd,audiobuffer,ESD_BUF_SIZE);
260  }
261  }
262  }
263 }

275 static BOOL ESD_PlayStart(void)
276 {
277  if (sndfd<0)
278  if (!(SETENV))
279  if ((sndfd=esd_playstream(format,md_mixfreq,espeaker,"libmikmod"))<0) {
280  _mm_errno=MMERR_OPENING_AUDIO;
281  return 1;
282  }
283  /* since the default behaviour of SIGPIPE on most Unices is to kill the
284  program, we'll prefer handle EPIPE ourselves should the esd die - recent
285  esdlib use a do-nothing handler, which prevents us from receiving EPIPE,
286  so we override this */
287  signal(SIGPIPE,SIG_IGN);
288 
289  /* silence buffers */
290  VC_SilenceBytes(audiobuffer,ESD_BUF_SIZE);
291  esd_writebuf(sndfd,audiobuffer,ESD_BUF_SIZE);
292 
293  return VC_PlayStart();
294 }
