467  m_nTotalCount++;

471  DWORD nMasterVol;
472  {
473  int nchn32 = (m_nChannels < 32) ? m_nChannels : 31;
474  if ((m_nType & MOD_TYPE_IT) && (m_nInstruments) && (nchn32 < 6)) nchn32 = 6;
475  int realmastervol = m_nMasterVolume;
476  if (realmastervol > 0x80)

479  }

481  DWORD mastervol = (realmastervol * (m_nSongPreAmp + 0x10)) >> 6;
482  if (mastervol > 0x200) mastervol = 0x200;
483  if ((m_dwSongFlags & SONG_GLOBALFADE) && (m_nGlobalFadeMaxSamples))
484  {
485  mastervol = _muldiv(mastervol, m_nGlobalFadeSamples, m_nGlobalFadeMaxSamples);
486  }
487  nMasterVol = (mastervol << 7) / attenuation;
488  if (nMasterVol > 0x180) nMasterVol = 0x180;
489  }
491  // Update channels data
492  m_nMixChannels = 0;
493  MODCHANNEL *pChn = Chn;
494  for (UINT nChn=0; nChn<MAX_CHANNELS; nChn++,pChn++)
495  {
496  if ((pChn->dwFlags & CHN_NOTEFADE) && (!(pChn->nFadeOutVol|pChn->nRightVol|pChn->nLeftVol)))
497  {

499  pChn->nROfs = pChn->nLOfs = 0;
500  }

504  pChn->nVUMeter = 0;
505 #ifdef ENABLE_STEREOVU

511  pChn->nInc = 0;
512  pChn->nRealVolume = 0;
513  pChn->nRealPan = pChn->nPan + pChn->nPanSwing;
514  if (pChn->nRealPan < 0) pChn->nRealPan = 0;
515  if (pChn->nRealPan > 256) pChn->nRealPan = 256;
516  pChn->nRampLength = 0;

518  if ((pChn->nPeriod) && (pChn->nLength))
519  {
520  int vol = pChn->nVolume + pChn->nVolSwing;
521 
522  if (vol < 0) vol = 0;
523  if (vol > 256) vol = 256;
524  // Tremolo
525  if (pChn->dwFlags & CHN_TREMOLO)
526  {

528  if (vol > 0)
529  {

534  vol += (ModRampDownTable[trempos] * (int)pChn->nTremoloDepth) >> tremattn;
535  break;
536  case 2:
537  vol += (ModSquareTable[trempos] * (int)pChn->nTremoloDepth) >> tremattn;
538  break;
539  case 3:
540  vol += (ModRandomTable[trempos] * (int)pChn->nTremoloDepth) >> tremattn;
541  break;
542  default:
543  vol += (ModSinusTable[trempos] * (int)pChn->nTremoloDepth) >> tremattn;
544  }
545  }
546  if ((m_nTickCount) || ((m_nType & (MOD_TYPE_STM|MOD_TYPE_S3M|MOD_TYPE_IT)) && (!(m_dwSongFlags & SONG_ITOLDEFFECTS))))
547  {
548  pChn->nTremoloPos = (trempos + pChn->nTremoloSpeed) & 0x3F;
549  }
550  }
551  // Tremor
552  if (pChn->nCommand == CMD_TREMOR)
553  {

563  }
564  pChn->dwFlags |= CHN_FASTVOLRAMP;
565  }

568  if (vol > 0x100) vol = 0x100;

576  {

579  for (UINT i=0; i<(UINT)(penv->nVolEnv-1); i++)

593  } else
594  if (pt)

598  } else

601  x1 = 0;
602  }

607  }
608  if (envvol < 0) envvol = 0;
609  if (envvol > 256) envvol = 256;

612  // Panning Envelope
613  if ((pChn->dwFlags & CHN_PANENV) && (penv->nPanEnv))

615  int envpos = pChn->nPanEnvPosition;

618  {

620  {
621  pt = i;
622  break;
623  }

634  envpan = penv->PanEnv[pt-1];

636  } else
637  {

639  x1 = 0;
640  }

643  envpan += ((envpos - x1) * (y2 - envpan)) / (x2 - x1);
644  }

651  } else
652  {
653  pan += ((envpan - 32) * (pan)) / 32;
654  }
655  if (pan < 0) pan = 0;
656  if (pan > 256) pan = 256;
657  pChn->nRealPan = pan;
658  }
659  // FadeOut volume
660  if (pChn->dwFlags & CHN_NOTEFADE)

668  } else

672  }
673  }

682  } else

686  {
687  pChn->nFadeOutVol = 0;
688  vol = 0;
689  }
690  }

746  } else

754  int envpitchdest = (((int)penv->PitchEnv[pt]) - 32) * 8;

847  if (pdelta < 0) pdelta = 0;
848  if (pdelta > 256) pdelta = 256;

879  break;

883  case 2: // Ramp Up

886  case 1: // Square

889  default: // Sine

892  int n = ((val * pChn->nAutoVibDepth) >> 8);
893  if (m_nType & MOD_TYPE_IT)
894  {
895  int df1, df2;
896  if (n < 0)
897  {
898  n = -n;
899  UINT n1 = n >> 8;
900  df1 = LinearSlideUpTable[n1];
901  df2 = LinearSlideUpTable[n1+1];
902  } else
903  {
904  UINT n1 = n >> 8;
905  df1 = LinearSlideDownTable[n1];
906  df2 = LinearSlideDownTable[n1+1];
907  }
908  n >>= 2;
909  period = _muldiv(period, df1 + ((df2-df1)*(n&0x3F)>>6), 256);
910  nPeriodFrac = period & 0xFF;
911  period >>= 8;
912  } else
913  {
914  period += (n >> 6);
915  }
916  }
917  // Final Period
918  if (period <= m_nMinPeriod)
919  {
920  if (m_nType & MOD_TYPE_S3M) pChn->nLength = 0;
921  period = m_nMinPeriod;
922  }
923  if (period > m_nMaxPeriod)
924  {
925  if ((m_nType & MOD_TYPE_IT) || (period >= 0x100000))
926  {
927  pChn->nFadeOutVol = 0;
928  pChn->dwFlags |= CHN_NOTEFADE;
929  pChn->nRealVolume = 0;

933  }

936  {
937  pChn->nFadeOutVol = 0;
938  pChn->dwFlags |= CHN_NOTEFADE;
939  pChn->nRealVolume = 0;
940  }
941  UINT ninc = _muldiv(freq, 0x10000, gdwMixingFreq);

943  if (m_nFreqFactor != 128) ninc = (ninc * m_nFreqFactor) >> 7;
944  if (ninc > 0xFF0000) ninc = 0xFF0000;
945  pChn->nInc = (ninc+1) & ~3;
946  }

949  if (pChn->pHeader)
950  {
951  INSTRUMENTHEADER *penv = pChn->pHeader;
952  // Volume Envelope
953  if (pChn->dwFlags & CHN_VOLENV)
954  {
955  // Increase position
956  pChn->nVolEnvPosition++;
957  // Volume Loop ?
958  if (penv->dwFlags & ENV_VOLLOOP)
959  {

969  pChn->nFadeOutVol = 0;
970  }
971  }
972  }

975  {
976  if (pChn->nVolEnvPosition == (UINT)penv->VolPoints[penv->nVolSustainEnd]+1)

987  pChn->nFadeOutVol = 0;
988

991  }
