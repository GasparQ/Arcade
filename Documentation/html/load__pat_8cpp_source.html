383 void pat_init_patnames(void)
384 {
385  int z, i, nsources, isdrumset, nskip, pfnlen;
386  char *p, *q;
387  char line[PATH_MAX];
388  char cfgsources[5][PATH_MAX] = {{0}, {0}, {0}, {0}, {0}};
389  MMSTREAM *mmcfg;
390  strcpy(pathforpat, PATHFORPAT);
391  strcpy(timiditycfg, TIMIDITYCFG);
392  p = getenv(PAT_ENV_PATH2CFG);
393  if( p ) {
394  strcpy(timiditycfg,p);
395  strcpy(pathforpat,p);
396  strcat(timiditycfg,"/timidity.cfg");
397  strcat(pathforpat,"/instruments");
398  }
399  strncpy(cfgsources[0], timiditycfg, PATH_MAX);
400  nsources = 1;
401 
402  for( i=0; i<MAXSMP; i++ ) midipat[i][0] = '\0';
403 
404  for ( z=0; z<5; z++ ) {
405  if (cfgsources[z][0] == 0) continue;
406  mmcfg = _mm_fopen(cfgsources[z],"r");
407  if( !mmcfg ) {
408  pat_message("can not open %s, use environment variable " PAT_ENV_PATH2CFG " for the directory", cfgsources[z]);
409  }
410  else {
411  // read in bank 0 and drum patches
412  isdrumset = 0;
413  _mm_fgets(mmcfg, line, PATH_MAX);
414  while( !_mm_feof(mmcfg) ) {
415  if( isdigit(line[0]) || (isblank(line[0]) && isdigit(line[1])) ) {
416  p = line;
417  // get pat number
418  while ( isspace(*p) ) p ++;
419  i = atoi(p);
420  while ( isdigit(*p) ) p ++;
421  while ( isspace(*p) ) p ++;
422  // get pat file name
423  if( *p && i < MAXSMP && i >= 0 && *p != '#' ) {
424  q = isdrumset ? midipat[pat_gm_drumnr(i)-1] : midipat[i];
425  pfnlen = 0;
426  while( *p && !isspace(*p) && *p != '#' && pfnlen < 128 ) {
427  pfnlen ++;
428  *q++ = *p++;
429  }
430  if( isblank(*p) && *(p+1) != '#' && pfnlen < 128 ) {
431  *q++ = ':'; pfnlen ++;
432  while( isspace(*p) ) {
433  while( isspace(*p) ) p++;
434  if ( *p == '#' ) { // comment
435 
436  } else while( *p && !isspace(*p) && pfnlen < 128 ) {
437  pfnlen ++;
438  *q++ = *p++;
439  }
440  if( isspace(*p) ) { *q++ = ' '; pfnlen++; }
441  }
442  }
443  *q++ = '\0';
444  }
445  }
446  if( !strncmp(line,"drumset",7) ) isdrumset = 1;
447  if( !strncmp(line,"source",6) && nsources < 5 ) {
448  q = cfgsources[nsources];
449  p = &line[7];
450  while ( isspace(*p) ) p ++;
451  pfnlen = 0;
452  while ( *p && *p != '#' && !isspace(*p) && pfnlen < 128 ) {
453  pfnlen ++;
454  *q++ = *p++;
455  }
456  *q = 0; // null termination
457  nsources++;
458  }
459  _mm_fgets(mmcfg, line, PATH_MAX);
460 
461  } /* end file parsing */
462  _mm_fclose(mmcfg);
463  }
464  }
465  q = midipat[0];
466  nskip = 0;
467  // make all empty patches duplicates the previous valid one.
468  for( i=0; i<MAXSMP; i++ ) {
469  if( midipat[i][0] ) q = midipat[i];
470  else {
471  if( midipat[i] != q)
472  strcpy(midipat[i], q);
473  if( midipat[i][0] == '\0' ) nskip++;
474  }
475  }
476  if( nskip ) {
477  for( i=MAXSMP; i-- > 0; ) {
478  if( midipat[i][0] ) q = midipat[i];
479  else if( midipat[i] != q )
480  strcpy(midipat[i], q);
481  }
482  }
483 }

642 static void pat_amplify(char *b, int num, int amp, int m)
643 {
644  char *pb;
645  BYTE *pu;
646  short int *pi;
647  WORD *pw;
648  int i,n,v;
649  n = num;
650  if( m & PAT_16BIT ) { // 16 bit
651  n >>= 1;
652  if( m & 2 ) { // unsigned
653  pw = (WORD *)b;
654  for( i=0; i<n; i++ ) {
655  v = (((int)(*pw) - 0x8000) * amp) / 100;
656  if( v < -0x8000 ) v = -0x8000;
657  if( v > 0x7fff ) v = 0x7fff;
658  *pw++ = v + 0x8000;
659  }
660  }
661  else {
662  pi = (short int *)b;
663  for( i=0; i<n; i++ ) {
664  v = ((*pi) * amp) / 100;
665  if( v < -0x8000 ) v = -0x8000;
666  if( v > 0x7fff ) v = 0x7fff;
667  *pi++ = v;
668  }
669  }
670  }
671  else {
672  if( m & 2 ) { // unsigned
673  pu = (BYTE *)b;

1048 // return relative position in samples for the rate starting with offset start ending with offset end
1049 static int pat_envelope_rpos(int rate, int start, int end)
1050 {
1051  int r, p, t, s;
1052  // rate byte is 3 bits exponent and 6 bits increment size
1053  // eeiiiiii
1054  // every 8 to the power ee the volume is incremented/decremented by iiiiii
1055  // Thank you Gravis for this weirdness...
1056  r = 3 - ((rate >> 6) & 3) * 3;
1057  p = rate & 0x3f;
1058  if( !p ) return 0;
1059  t = end - start;
1060  if( !t ) return 0;
1061  if (t < 0) t = -t;
1062  s = (t << r)/ p;
1063  return s;
1064 }

1066 static void pat_modenv(WaveHeader *hw, int mpos[6], int mvol[6])
1067 {
1068  int i, sum, s;
1069  BYTE *prate = hw->envelope_rate, *poffset = hw->envelope_offset;
1070  for( i=0; i<6; i++ ) {
1071  mpos[i] = 0;
1072  mvol[i] = 64;
1073  }
1074  if( !memcmp(prate, "??????", 6) || poffset[5] >= 100 ) return; // weird rates or high env end volume
1075  if( !(hw->modes & PAT_SUSTAIN) ) return; // no sustain thus no need for envelope
1076  s = hw->wave_size;
1077  if (s == 0) return;
1078  if( hw->modes & PAT_16BIT )
1079  s >>= 1;
1080  // offsets 0 1 2 3 4 5 are distributed over 0 2 4 6 8 10, the odd numbers are set in between
1081  sum = 0;
1082  for( i=0; i<6; i++ ) {
1083  mvol[i] = poffset[i];
1084  mpos[i] = pat_envelope_rpos(prate[i], i? poffset[i-1]: 0, poffset[i]);
1085  sum += mpos[i];
1086  }
1087  if( sum == 0 ) return;
1088  if( sum > s ) {
1089  for( i=0; i<6; i++ )
1090  mpos[i] = (s * mpos[i]) / sum;
1091  }
1092  for( i=1; i<6; i++ )
1093  mpos[i] += mpos[i-1];
1094  for( i=0; i<6 ; i++ ) {
1095  mpos[i] = (256 * mpos[i]) / s;
1096  mpos[i]++;
1097  if( i > 0 && mpos[i] <= mpos[i-1] ) {
1098  if( mvol[i] == mvol[i-1] ) mpos[i] = mpos[i-1];
1099  else mpos[i] = mpos[i-1] + 1;
1100  }
1101  if( mpos[i] > 256 ) mpos[i] = 256;
1102  }
1103  mvol[5] = 0; // kill Bill....
1104 }
