109 static void OSS_CommandLine(CHAR *cmdline)
110 {
111  CHAR *ptr;
112 
113 #ifdef SNDCTL_DSP_SETFRAGMENT
114  if((ptr=MD_GetAtom("buffer",cmdline,0))) {
115  fragsize=atoi(ptr);
116  if((fragsize<7)||(fragsize>17)) fragsize=DEFAULT_FRAGSIZE;
117  free(ptr);
118  }
119  if((ptr=MD_GetAtom("count",cmdline,0))) {
120  numfrags=atoi(ptr);
121  if((numfrags<2)||(numfrags>255)) numfrags=DEFAULT_NUMFRAGS;
122  free(ptr);
123  }
124 #endif
125  if((ptr=MD_GetAtom("card",cmdline,0))) {
126  card = atoi(ptr);
127  if((card<0)||(card>99)) card=DEFAULT_CARD;
128  free(ptr);
129  }

132 static char *OSS_GetDeviceName(void)
133 {
134  static char sounddevice[20];
135 
136  /* First test for devfs enabled Linux sound devices */
137  if (card)
138  sprintf(sounddevice,"/dev/sound/dsp%d",card);
139  else
140  strcpy(sounddevice,"/dev/sound/dsp");
141  if(!access(sounddevice,F_OK))
142  return sounddevice;
143 
144  sprintf(sounddevice,"/dev/dsp%d",card);
145  if (!card) {
146  /* prefer /dev/dsp0 over /dev/dsp, as /dev/dsp might be a symbolic link
147  to something else than /dev/dsp0. Revert to /dev/dsp if /dev/dsp0
148  does not exist. */
149  if(access("/dev/dsp0",F_OK))
150  strcpy(sounddevice,"/dev/dsp");
151  }
152 
153  return sounddevice;
154 }

156 static BOOL OSS_IsThere(void)
157 {
158  /* under Linux, and perhaps other Unixes, access()ing the device is not
159  enough since it won't fail if the machine doesn't have sound support
160  in the kernel or sound hardware */
161  int fd;
162 
163  if((fd=open(OSS_GetDeviceName(),O_WRONLY))>=0) {
164  close(fd);
165  return 1;
166  }
167  return (errno==EACCES?1:0);
168 }

170 static BOOL OSS_Init_internal(void)
171 {
172  int play_stereo,play_rate;
173  int orig_precision,orig_stereo;
174  long formats;
175 #if SOUND_VERSION >= 301
176  audio_buf_info buffinf;
177 #endif
178 
179 #ifdef SNDCTL_DSP_GETFMTS
180  /* Ask device for supported formats */
181  if(ioctl(sndfd,SNDCTL_DSP_GETFMTS,&formats)<0) {
182  _mm_errno=MMERR_OPENING_AUDIO;
183  return 1;
184  }
185 #else
186  formats=AFMT_S16_NE|AFMT_U8;
187 #endif
188 
189  orig_precision=play_precision=(md_mode&DMODE_16BITS)?AFMT_S16_NE:AFMT_U8;
190 
191  /* Device does not support the format we would prefer... */
192  if(!(formats & play_precision)) {
193  /* We could try 8 bit sound if available */
194  if(play_precision==AFMT_S16_NE &&(formats&AFMT_U8)) {
195  _mm_errno=MMERR_8BIT_ONLY;
196  return 1;
197  }
198 #ifdef AFMT_MU_LAW
199  /* We could try uLaw if available */
200  if(formats&AFMT_MU_LAW) {
201  if((md_mode&DMODE_STEREO)||(md_mode&DMODE_16BITS)||
202  md_mixfreq!=8000) {
203  _mm_errno=MMERR_ULAW;
204  return 1;
205  } else
206  orig_precision=play_precision=AFMT_MU_LAW;
207  } else
208 #endif
209  /* Otherwise, just abort */
210  {
211  _mm_errno=MMERR_OSS_SETSAMPLESIZE;
212  return 1;
213  }
214  }
215 
216  if((ioctl(sndfd,SNDCTL_DSP_SETFMT,&play_precision)<0)||
217  (orig_precision!=play_precision)) {
218  _mm_errno=MMERR_OSS_SETSAMPLESIZE;
219  return 1;
220  }
221 #ifdef SNDCTL_DSP_CHANNELS
222  orig_stereo=play_stereo=(md_mode&DMODE_STEREO)?2:1;
223  if((ioctl(sndfd,SNDCTL_DSP_CHANNELS,&play_stereo)<0)||
224  (orig_stereo!=play_stereo)) {
225  _mm_errno=MMERR_OSS_SETSTEREO;
226  return 1;
227  }
228 #else
229  orig_stereo=play_stereo=(md_mode&DMODE_STEREO)?1:0;
230  if((ioctl(sndfd,SNDCTL_DSP_STEREO,&play_stereo)<0)||
231  (orig_stereo!=play_stereo)) {
232  _mm_errno=MMERR_OSS_SETSTEREO;
233  return 1;
234  }
235 #endif
236 
237  play_rate=md_mixfreq;
238  if((ioctl(sndfd,SNDCTL_DSP_SPEED,&play_rate)<0)) {
239  _mm_errno=MMERR_OSS_SETSPEED;
240  return 1;
241  }
242  md_mixfreq=play_rate;
243 
244 #if SOUND_VERSION >= 301
245  /* This call fails on Linux/PPC */
246  if((ioctl(sndfd,SNDCTL_DSP_GETOSPACE,&buffinf)<0))
247  ioctl(sndfd,SNDCTL_DSP_GETBLKSIZE,&buffinf.fragsize);
248  if(!(audiobuffer=(SBYTE*)_mm_malloc(buffinf.fragsize)))
249  return 1;
250 
251  buffersize = buffinf.fragsize;
252 #else
253  ioctl(sndfd,SNDCTL_DSP_GETBLKSIZE,&buffersize);
254  if(!(audiobuffer=(SBYTE*)_mm_malloc(buffersize)))
255  return 1;
256 #endif
257 
258  return VC_Init();
259 }

261 static BOOL OSS_Init(void)
262 {
263 #ifdef SNDCTL_DSP_SETFRAGMENT
264  int fragmentsize;
265 #endif
266 
267  if((sndfd=open(OSS_GetDeviceName(),O_WRONLY))<0) {
268  _mm_errno=MMERR_OPENING_AUDIO;
269  return 1;
270  }
271 
272 #ifdef SNDCTL_DSP_SETFRAGMENT
273  if((fragsize==DEFAULT_FRAGSIZE)&&(getenv("MM_FRAGSIZE"))) {
274  fragsize=atoi(getenv("MM_FRAGSIZE"));
275  if((fragsize<7)||(fragsize>17)) fragsize=DEFAULT_FRAGSIZE;
276  }
277  if((numfrags==DEFAULT_NUMFRAGS)&&(getenv("MM_NUMFRAGS"))) {
278  numfrags=atoi(getenv("MM_NUMFRAGS"));
279  if((numfrags<2)||(numfrags>255)) numfrags=DEFAULT_NUMFRAGS;
280  }
281 
282  fragmentsize=(numfrags<<16)|fragsize;
283 
284  if(ioctl(sndfd,SNDCTL_DSP_SETFRAGMENT,&fragmentsize)<0) {
285  _mm_errno=MMERR_OSS_SETFRAGMENT;
286  return 1;
287  }
288 #endif
289 
290  return OSS_Init_internal();
291 }

316 static void OSS_Update(void)
317 {
318  int done;
319 
320 #if SOUND_VERSION >= 301
321  audio_buf_info buffinf;
322 
323  buffinf.fragments = 2;
324  for(;;) {
325  /* This call fails on Linux/PPC */
326  if ((ioctl(sndfd,SNDCTL_DSP_GETOSPACE,&buffinf)<0)) {
327  buffinf.fragments--;
328  buffinf.fragsize = buffinf.bytes = buffersize;
329  }
330  if(!buffinf.fragments)
331  break;
332  done=VC_WriteBytes(audiobuffer,buffinf.fragsize>buffinf.bytes?
333  buffinf.bytes:buffinf.fragsize);
334 #ifdef AFMT_MU_LAW
335  if (play_precision==AFMT_MU_LAW)
336  unsignedtoulaw(audiobuffer,done);
337 #endif
338  write(sndfd,audiobuffer,done);
339  }
340 #else
341  done=VC_WriteBytes(audiobuffer,buffersize);
342 #ifdef AFMT_MU_LAW
343  if (play_precision==AFMT_MU_LAW)
344  unsignedtoulaw(audiobuffer,done);
345 #endif
346  write(sndfd,audiobuffer,done);
347 #endif
348 }
